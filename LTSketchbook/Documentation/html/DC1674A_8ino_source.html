103 void loop()
104 {
105  /* the API functions can be used to read and write individual bit fields within a command code */
106  Serial.println(F("Press any key to arm ship-and-store mode."));
107  while (Serial.available() == 0);
108  while (Serial.available() > 0)
109  Serial.read();
110  LTC4155_write_register(chip, LTC4155_ARM_SHIPMODE, LTC4155_ARM_SHIPMODE_PRESET_ARM);
111  Serial.println(F("Ship-and-store mode armed.\n"));
112 
113  Serial.println(F("Press any key to disarm ship-and-store mode."));
114  while (Serial.available() == 0);
115  while (Serial.available() > 0)
116  Serial.read();
117  LTC4155_read_register(chip, LTC4155_REG6, &data);
118  LTC4155_write_register(chip, LTC4155_REG6, data); // Write to register 6 to clear ship-mode without changing its contents.
119  Serial.println(F("Ship-and-store mode disarmed.\n"));
120 }
