7 NOTES
8  Setup:
9  Set the terminal baud rate to 115200 and select the newline terminator.
10  Ensure all jumpers on the demo board are installed in their default positions from the factory.
11  Refer to Demo Manual D1894B.
12 
13 
14  Menu Entry 1: Write Configuration
15  Writes the configuration register of the LTC6804s on the stack. This command can be used to turn on
16  the reference and shorten ADC conversion Times.
17 
18  Menu Entry 2: Read Configuration
19  Reads the configuration register of the LTC6804, the read configuration can differ from the written configuration.
20  The GPIO pins will reflect the state of the pin
21 
22  Menu Entry 3: Start Cell voltage conversion
23  Starts a LTC6804 cell channel adc conversion.
24 
25  Menu Entry 4: Read cell voltages
26  Reads the LTC6804 cell voltage registers and prints the results to the serial port.
27 
28  Menu Entry 5: Start Auxiliary voltage conversion
29  Starts a LTC6804 GPIO channel adc conversion.
30 
31  Menu Entry 6: Read Auxiliary voltages
32  Reads the LTC6804 axiliary registers and prints the GPIO voltages to the serial port.
33 
34  Menu Entry 7: Start cell voltage measurement loop
35  The command will continuously measure the LTC6804 cell voltages and print the results to the serial port.
36  The loop can be exited by sending the MCU a 'm' character over the serial link.
37 
38 USER INPUT DATA FORMAT:

179 /*!*****************************************
180  \brief executes the user inputted command
181 
182  Menu Entry 1: Write Configuration \n
183  Writes the configuration register of the LTC6804. This command can be used to turn on the reference
184  and increase the speed of the ADC conversions.
185 
186  Menu Entry 2: Read Configuration \n
187  Reads the configuration register of the LTC6804, the read configuration can differ from the written configuration.
188  The GPIO pins will reflect the state of the pin
189 
190  Menu Entry 3: Start Cell voltage conversion \n
191  Starts a LTC6804 cell channel adc conversion.
192 
193  Menu Entry 4: Read cell voltages
194  Reads the LTC6804 cell voltage registers and prints the results to the serial port.
195 
196  Menu Entry 5: Start Auxiliary voltage conversion
197  Starts a LTC6804 GPIO channel adc conversion.
198 
199  Menu Entry 6: Read Auxiliary voltages6118
200  Reads the LTC6804 axiliary registers and prints the GPIO voltages to the serial port.
201 
202  Menu Entry 7: Start cell voltage measurement loop
203  The command will continuously measure the LTC6804 cell voltages and print the results to the serial port.
204  The loop can be exited by sending the MCU a 'm' character over the serial link.
205 
206 *******************************************/
207 void run_command(uint32_t cmd)
208 {
209  int8_t error = 0;
210 
211  char input = 0;
212  switch (cmd)
213  {
214 
215  case 1:
216  wakeup_sleep();
217  LTC6804_wrcfg(TOTAL_IC,tx_cfg);
218  print_config();
219  break;
220 
221  case 2:
222  wakeup_sleep();
223  error = LTC6804_rdcfg(TOTAL_IC,rx_cfg);
224  if (error == -1)
225  {
226  Serial.println("A PEC error was detected in the received data");
227  }
228  print_rxconfig();
229  break;
230 
231  case 3:
232  wakeup_sleep();
233  LTC6804_adcv();
234  delay(3);
235  Serial.println("cell conversion completed");
236  Serial.println();
237  break;
238 
239  case 4:
240  wakeup_sleep();
241  error = LTC6804_rdcv(0, TOTAL_IC,cell_codes); // Set to read back all cell voltage registers
242  if (error == -1)
243  {
244  Serial.println("A PEC error was detected in the received data");
245  }
246  print_cells();
247  break;
248 
249  case 5:
250  wakeup_sleep();
251  LTC6804_adax();
252  delay(3);
253  Serial.println("aux conversion completed");
254  Serial.println();
255  break;
256 
257  case 6:
258  wakeup_sleep();
259  error = LTC6804_rdaux(0,TOTAL_IC,aux_codes); // Set to read back all aux registers
260  if (error == -1)
261  {
262  Serial.println("A PEC error was detected in the received data");
263  }
264  print_aux();
265  break;
266 
267  case 7:
268  Serial.println("transmit 'm' to quit");
269  wakeup_sleep();
270  LTC6804_wrcfg(TOTAL_IC,tx_cfg);
271  while (input != 'm')
272  {
273  if (Serial.available() > 0)
274  {
275  input = read_char();
276  }
277  wakeup_idle();
278  LTC6804_adcv();
279  delay(10);
280  wakeup_idle();
281  error = LTC6804_rdcv(0, TOTAL_IC,cell_codes);
282  if (error == -1)
283  {
284  Serial.println("A PEC error was detected in the received data");
285  }
286  print_cells();
287  delay(500);
288  }
289  print_menu();
290  break;
291 
292  default:
293  Serial.println("Incorrect Option");
294  break;
295  }
296 }
