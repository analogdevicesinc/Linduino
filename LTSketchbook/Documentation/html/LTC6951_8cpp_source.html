124 /* -------------------------------------------------------------------------
125  FUNCTION: LTC6951_read_field
126  For SPI FIELDS located in 1 or multiple address location
127  - reads specific address locations
128  - identifies and returns specific field in question
129  - can handle SPI fields in multiple addresses, if MSB bit is in the lower number address
130 --------------------------------------------------------------------------- */
131 long LTC6951_read_field(uint8_t cs, uint8_t address, uint8_t MSB_loc, uint8_t numbits)
132 {
133  int bit_shift, i, num_reg;
134  long field_val, maskbits, pow2;
135 
136  num_reg=0;
137  field_val=0;
138 // determines how many register are used
139  do
140  {
141  bit_shift = (MSB_loc+1)- (numbits-num_reg*8); // determines bit_shift for last register location
142  field_val=LTC6951_read(cs, (address+num_reg))+(field_val<<8); // reads current address locations, shifts previous address location 8 bits
143  num_reg++;
144  }
145  while ((bit_shift<0) && (num_reg<4));
146 
147 // creates a bit mask for complete word,
148  maskbits = 1;
149  pow2=1;
150  for (i=1, maskbits=1; i<numbits; i++)
151  {
152  pow2=pow2*2;
153  maskbits = maskbits+pow2;
154  }
155 
156  field_val=(field_val >>bit_shift) &maskbits;
157  return field_val;
158 }

187 /* -------------------------------------------------------------------------
188  FUNCTION: LTC6951_write_field
189  For SPI FIELDS
190  - reads specific address location
191  - identifies and returns specific field in question
192  - can handle SPI fields in multiple addresses, if MSB bit is in the lower number address
193 ---------------------------------------------------------------------------- */
194 uint8_t LTC6951_write_field(uint8_t cs, long field_data, uint8_t address, uint8_t MSB_loc, uint8_t numbits)
195 {
196  long current_content, desired_content, reg_val;
197  int LSB_loc, i, j, num_reg, bit_shift;
198  long temp_arr[32];
199 
200  for (i=0; i<32 ; i++) temp_arr[i]=0; // init temp_arr
201 
202 // read data in current address location and put in a bit array
203  num_reg=0;
204  current_content=0;
205  do
206  {
207  bit_shift=(MSB_loc+1)-(numbits-num_reg*8);
208  current_content=LTC6951_read(cs, (address+num_reg)) + (current_content<<8);
209 
210  num_reg++;
211  }
212  while ((bit_shift<0) && (num_reg<4));
213  for (i=0; i<(8*num_reg); i++)
214  {
215  temp_arr[i]=(current_content>>i) & 1;
216  }
217 
218 // exchange current bits with desired bits
219  LSB_loc = 8*(num_reg-1)+MSB_loc-numbits+1;
220  for (i=LSB_loc, j=0; i<=(MSB_loc+(num_reg-1)*8); i++, j++)
221  {
222  temp_arr[i] = (field_data>>j) &1;
223  } // end of for loop
224 
225 // reconstruct bits into an integer
226  desired_content = 0;
227  for (i=0; i<(8*num_reg); i++)
228  {
229  desired_content = desired_content | (temp_arr[i]<<i);
230  } // end of for loop
231 
232 // write new field value to part
233  for (i=0; i<num_reg; i++)
234  {
235  reg_val = (desired_content >> 8*(num_reg-1-i)) & 0xff;
236  LTC6951_write(cs, (address+i), reg_val);
237  } // end of for loop
238 } // end of LTC6951_write_field

318 /* -------------------------------------------------------------------------
319  FUNCTION: set_LTC6951_REGS_lkup_tbl
320  - writes LTC6951 registers from a look up table
321  - uses the Multi-byte write option for more efficient data transfer
322  - Calibrates VCO based on RAO and Autocal bit vales
323  - Either SPI Syncs or recommends the use to SYNC pin based on RAO, SN and SR values
324 --------------------------------------------------------------------------- */
325 void set_LTC6951_REGS_lkup_tbl(uint8_t lkup_tbl_row)
326 {
327  uint8_t val_temp, i, RAO_bit, AUTOCAL_bit, SN_bit, SR_bit;
328  uint8_t val_reg2, val_reg3, val_regA;
329  uint8_t *rx;
330 
331 //! ************** SPI REGISTER WRITE SECTION ************
332  output_low(QUIKEVAL_CS); //! 1) CS low, enable SPI
333  val_temp = 1<<1; //! 2) addr 1, shift 1 bit to account for RW bit (W=0)
334  *rx = SPI.transfer(val_temp); //! 3) send ADDR1 byte
335  for (i=1; i<19; i++)
336  {
337  val_temp=LTC6951_lkup_tbl[lkup_tbl_row][i];
338  *rx = SPI.transfer(val_temp); //! 4) send REG byte
339  }
340  output_high(QUIKEVAL_CS); //! 5) CS high, disable SPI
341 
342 //! **************** CALIBRATION SECTION ******************
343 //! Calibration required after Register writes if RAO=1 or if RAO=AUTOCAL=0.
344  val_reg2 = LTC6951_lkup_tbl[lkup_tbl_row][2]; //! CAL bit is located @ reg 2, bit 0, AUTOCAL bit reg 2, bit 3
345  val_reg3 = LTC6951_lkup_tbl[lkup_tbl_row][3]; //! RAO bit is located @ reg 3, bit 2; SSYNC bit @ reg 3, bit 4
346  RAO_bit = (val_reg3 & 0x04)>>2;
347  AUTOCAL_bit = (val_reg3 & 0x08)>>3;
348 
349  val_regA = LTC6951_lkup_tbl[lkup_tbl_row][10]; //! SN, SR bit are located @ reg A, bits 7.6
350  SN_bit = (val_regA & 0x80)>>7;
351  SR_bit = (val_regA & 0x40)>>6;
352 
353  if ((RAO_bit ==1) || (AUTOCAL_bit ==0)) //! checks if RAO=1 or AUTOCAL bit = 0
354  {
355  val_reg2 = val_reg2 | 0x01; //! sets CAL bit = 1
356  output_low(QUIKEVAL_CS); //! CS low, enable SPI
357  val_temp = 2<<1; //! addr 2, shift 1 bit to account for RW bit (W=0)
358  *rx = SPI.transfer(val_temp); //! send ADDR2 byte + RW byte
359  *rx = SPI.transfer(val_reg2); //! send REG2 byte w/ CAL=1
360  output_high(QUIKEVAL_CS); //! CS high, disable SPI
361  val_reg2 = LTC6951_lkup_tbl[lkup_tbl_row][2]; //! CAL bit auto-clears after Calibration, set val_reg2 back to look-up table value
362  }
363  delay(1); //! wait 1ms for Calibration to complete, calibration usually takes <100us
364 
365 //! **************** OUTPUT SYNC SECTION ******************
366 //! SSYNC can be used for EZSYNC StandAlone, EZSync Ref aligned standalone, EZParallelSYnc, EZ204Sync
367 //! - EZSync Standalone, RAO=0
368 //! - EZParallelSync, EZ204Sync, EZSync ref aligned: RA0=1, SN=SR=0
369 //! For ParallelSync or EZSync Controller to work the the SYNC pin should be used
370 //! - EZSync Controller, RAO=0 (User will have to remember if its a Controller as there is not a bit to distinguish this)
371 //! - ParallelSync: RA0=1, SN or SR= 1
372 //! See datasheet for timing requirements
373  if ((RAO_bit==0) || ((RAO_bit == 1) && ((SN_bit+SR_bit)==0)))
374  {
375  val_reg2 = val_reg2 | 0x04; //! Sets SSYNC bit =1
376  output_low(QUIKEVAL_CS); //! CS low, enable SPI
377  val_temp = 2<<1; //! addr 2, shift 1 bit to account for RW bit (W=0)
378  *rx = SPI.transfer(val_temp); //! send ADDR2 byte + RW byte
379  *rx = SPI.transfer(val_reg2); //! send REG2 byte w/ CAL=1
380  output_high(QUIKEVAL_CS); //! CS high, disable SPI
381  delay(1); //! SSYNC bit remains high for a minimum of 1ms
382 
383  val_reg2 = val_reg2 - 0x04; //! Sets SSYNC bit back to 0
384  output_low(QUIKEVAL_CS); //! CS low, enable SPI
385  val_temp = 2<<1; //! addr 2, shift 1 bit to account for RW bit (W=0)
386  *rx = SPI.transfer(val_temp); //! send ADDR2 byte + RW byte
387  *rx = SPI.transfer(val_reg2); //! send REG2 byte w/ CAL=1
388  output_high(QUIKEVAL_CS); //! CS high, disable SPI
389  }
390 
391 //! **************** INFO SECTION ******************
392 //! INFO ONLY: Print Register Map written to part
393  for (i=1; i<19; i++)
394  {
395  val_temp=LTC6951_lkup_tbl[lkup_tbl_row][i];
396  Serial.print(F("ADDR"));
397  if (i<16) Serial.print("0");
398  Serial.print(i, HEX);
399  Serial.print(F(" = 0x"));
400  Serial.println(val_temp, HEX);
401  }
402 
403  if (RAO_bit == 0)
404  {
405  Serial.println(F(" ***************************************************************************** "));
406  Serial.println(F("If this is a EZSync Standalone Device ignore this message - LTC6951 has been SYNC'd via SPI "));
407  Serial.println(F("If this is a EZSync Controller Device toggle all EZSync Device's Sync pins "));
408  Serial.println(F("EZSync TIMING: Sync Pins must remain high for 1ms, and with <10us skew between all EZSync Sync pins "));
409  Serial.println(F(" ***************************************************************************** "));
410  }

413  Serial.println(F(" ***************************************************************************** "));
414  Serial.println(F("ParallelSync Timing: See Datasheet for setup and hold times "));

417  Serial.println(F(" - DC2430's J20-J22 circuitry can be used to create correct SYNC to REF setup and hold times"));
418  Serial.println(F(" ***************************************************************************** "));

452  {
453  //! ************** INIT: SPI REGISTERS (ALL) WRITE SECTION ************
454  output_low(QUIKEVAL_CS); //! 1) CS low, enable SPI
455  start_addr = 1;
456  val_temp = start_addr<<1; //! 2) addr 1, shift 1 bit to account for RW bit (W=0)
457  *rx = SPI.transfer(val_temp); //! 3) send ADDR1 byte
458  for (i=start_addr; i<19; i++)
459  {
460  val_temp=local6951_lkup_tbl[lkup_tbl_row][i];
461  *rx = SPI.transfer(val_temp); //! 4) send REG byte
462  }
463  output_high(QUIKEVAL_CS); //! 5) CS high, disable SPI
464  } // end if(init_setup)
465 
466 // checks if b or r-divider change?
467  REGx3toREGx5_change=false;
468  for (i=3; i<=5; i++)
469  {
470  if (local6951_lkup_tbl[0][i] != local6951_lkup_tbl[1][i]) REGx3toREGx5_change=true;
471  }
472 
473 // checks if p-divider change?
474  REGx7toREGx8_change=false;
475  for (i=7; i<=8; i++)
476  {
477  if (local6951_lkup_tbl[0][i] != local6951_lkup_tbl[1][i]) REGx7toREGx8_change=true;
478  }
479 
480 // checks if output divider change?
481  REGx9toREGx12_change=false;
482  for (i=9; i<=0x12; i++)
483  {
484  if (local6951_lkup_tbl[0][i] != local6951_lkup_tbl[1][i]) REGx9toREGx12_change=true;
485  }
486 
487 //! **************** SELECTION SECTION ******************
488 //! Calibration required after Register writes if RAO=1 or if RAO=AUTOCAL=0.
489  val_reg2 = local6951_lkup_tbl[lkup_tbl_row][2]; //! CAL bit is located @ reg 2, bit 0, AUTOCAL bit reg 2, bit 3
490  val_reg3 = local6951_lkup_tbl[lkup_tbl_row][3]; //! RAO bit is located @ reg 3, bit 2; SSYNC bit @ reg 3, bit 4
491  RAO_bit = (val_reg3 & 0x04)>>2;
492  AUTOCAL_bit = (val_reg3 & 0x08)>>3;
493 
494 //! All Settings of RAO and AUTOCAL Run this section
495 //! ************** SPI REGISTERS 4-19 WRITE SECTION ************
496 //! it possible the that register 6 is the only register that needs to change (N-divider),
497 //! but this codes was written for any and all frequency changes.
498  start_addr = 6; //! ADDR6 includes include ND
499  if (REGx3toREGx5_change==true) start_addr=3; //! ADDR4 includes include BD (cal-divider), ADDR 5 includes RD
500  end_addr=6; //! ADDR6 includes include ND
501  if (REGx7toREGx8_change==true) end_addr=8; //! ADDR8 includes include PD
502  if (REGx9toREGx12_change==true) end_addr=12; //! ADDR9-12 include output dividers
503 
504  output_low(QUIKEVAL_CS); //! CS low, enable SPI
505  val_temp = start_addr<<1; //! addr 4, shift 1 bit to account for RW bit (W=0)
506  *rx = SPI.transfer(val_temp); //! send ADDR byte
507  for (i=start_addr; i<=end_addr; i++)
508  {
509  val_temp=local6951_lkup_tbl[lkup_tbl_row][i];
510  *rx = SPI.transfer(val_temp); //! send REG bytes
511  }
512  output_high(QUIKEVAL_CS); //! CS high, disable SPI
513 
514 
515 //! If Autocal=0 or RAO_bit=1 (ParallelSync, EZParallelSync, EZ204Sync, EZSync Ref Aligned
516 //! Then the cal bit needs to be toggle, with Linduino it maybe fastest to continue
517 //! The multi-byte write operation. (reason: The CS L-H transistion has a delay)
518  if (RAO_bit || !AUTOCAL_bit)
519  {
520  start_addr = 2;
521  val_temp = start_addr<<1; //! ADDR, shift 1 bit to account for RW bit (W=0)
522  val_reg2 = val_reg2 | 0x01; //! sets CAL bit loc = 1
523 
524  output_low(QUIKEVAL_CS); //! CS low, enable SPI
525  *rx = SPI.transfer(val_temp); //! send ADDR2 byte
526  *rx = SPI.transfer(val_reg2); //! send REG2 byte
527  output_high(QUIKEVAL_CS); //! CS high, disable SPI
528  }
529 
530 } // end of set_LTC6951_REGS_freq_jump
