77 void loop()

104 
105

107  {
108  oldcolor = currentcolor;
109

111  {
112  currentcolor = ILI9341_RED;
113  tft.drawRect(0, 0, BOXSIZE, BOXSIZE, ILI9341_WHITE);
114  }
115  else if (p.x < BOXSIZE*2)
116  {
117  currentcolor = ILI9341_YELLOW;
118  tft.drawRect(BOXSIZE, 0, BOXSIZE, BOXSIZE, ILI9341_WHITE);
119  }
120  else if (p.x < BOXSIZE*3)
121  {
122  currentcolor = ILI9341_GREEN;
123  tft.drawRect(BOXSIZE*2, 0, BOXSIZE, BOXSIZE, ILI9341_WHITE);
124  }

126  {
127  currentcolor = ILI9341_CYAN;
128  tft.drawRect(BOXSIZE*3, 0, BOXSIZE, BOXSIZE, ILI9341_WHITE);
129  }
130  else if (p.x < BOXSIZE*5)
131  {
132  currentcolor = ILI9341_BLUE;
133  tft.drawRect(BOXSIZE*4, 0, BOXSIZE, BOXSIZE, ILI9341_WHITE);
134  }
135  else if (p.x < BOXSIZE*6)

137  currentcolor = ILI9341_MAGENTA;
138  tft.drawRect(BOXSIZE*5, 0, BOXSIZE, BOXSIZE, ILI9341_WHITE);
139  }
140 
141  if (oldcolor != currentcolor)
142  {
143  if (oldcolor == ILI9341_RED)
144  tft.fillRect(0, 0, BOXSIZE, BOXSIZE, ILI9341_RED);
145  if (oldcolor == ILI9341_YELLOW)
146  tft.fillRect(BOXSIZE, 0, BOXSIZE, BOXSIZE, ILI9341_YELLOW);
147  if (oldcolor == ILI9341_GREEN)
148  tft.fillRect(BOXSIZE*2, 0, BOXSIZE, BOXSIZE, ILI9341_GREEN);
149  if (oldcolor == ILI9341_CYAN)
150  tft.fillRect(BOXSIZE*3, 0, BOXSIZE, BOXSIZE, ILI9341_CYAN);
151  if (oldcolor == ILI9341_BLUE)
152  tft.fillRect(BOXSIZE*4, 0, BOXSIZE, BOXSIZE, ILI9341_BLUE);
153  if (oldcolor == ILI9341_MAGENTA)
154  tft.fillRect(BOXSIZE*5, 0, BOXSIZE, BOXSIZE, ILI9341_MAGENTA);
155  }
156  }
157  if (((p.y-PENRADIUS) > BOXSIZE) && ((p.y+PENRADIUS) < tft.height()))
158  {
159  tft.fillCircle(p.x, p.y, PENRADIUS, currentcolor);
160  }
161 }
