111 //! Repeats Linduino loop
112 void loop()
113 {
114  if (1)
115  {
116  int8_t ack = 0; // I2C acknowledge indicator
117  static uint8_t user_command; // The user input command
118  if (Serial.available()) //! Do nothing if serial is not available
119  {
120  user_command = read_int(); //! Read user input command
121  if (user_command != 'm')
122  Serial.println(user_command);
123  Serial.println();
124  ack = 0;
125  switch (user_command) //! Prints the appropriate submenu
126  {
127  case 1:
128  ack |= main_menu_1_continuous_mode(); // continous mode
129  break;
130  case 2:
131  ack |= main_menu_2_read_and_clear_faults(); // read and clear faults
132  break;
133  case 3:
134  ack |= main_menu_3_send_ARA();// read address of alert
135  break;
136  case 4:
137  ack |= main_menu_4_manage_alerts(); // manage alerts
138  break;
139  case 5:
140  ack |= main_menu_5_settings(); // settings
141  break;
142  case 6:
143  ack |= main_menu_6_read_all_registers();
144  break;
145  default:
146  Serial.println("Incorrect Option");
147  break;
148  }
149 
150  print_prompt();
151  }
152  }

181 //! Function to read Rsense current, ADIN voltage and ADIN2 voltage in continous mode
182 int8_t main_menu_1_continuous_mode()
183 //! @return Returns the state of the acknowledge bit after the I2C address write. 0=acknowledge, 1=no acknowledge.
184 {
185  int8_t ack = 0;
186  float sense_voltage, source_voltage, adin_voltage;
187  uint8_t sense_code, source_code, adin_code;
188  float adin_lsb = (1.23/255), adin_resisive_ratio = (155.4/12.4);
189  float source_lsb = (15.44/255), source_resisive_ratio = 1; // Set voltage divider ratio to 1, for no resistor divider circuit.
190  float sense_lsb = (0.03845/255), sense_resisive_ratio = 1;
191 
192  do
193  {
194  uint8_t faults;
195  Serial.print(F("********** Press Enter to Exit ***********\n\n"));
196  ack |= LTC4280_read(LTC4280_I2C_ADDRESS, LTC4280_FAULT_REG, &faults);
197  if (faults != 0)
198  {
199  Serial.println(F("Faults Detected :"));
200  if (faults & LTC4280_FET_SHORT_FAULT )
201  Serial.println(F(" FET SHORT DETECTED"));
202  if (faults & LTC4280_EN_STATE_CHANGE)
203  Serial.println(F(" !EN STATE CHANGE DETECTED"));
204  if (faults & LTC4280_POWER_BAD_FAULT )
205  Serial.println(F(" BAD POWER DETECTED"));
206  if (faults & LTC4280_OVERCURRENT_FAULT )
207  Serial.println(F(" OVERCURRENT DETECTED"));
208  if (faults & LTC4280_UNDERVOLTAGE_FAULT )
209  Serial.println(F(" UNDERVOLTAGE DETECTED"));
210  if (faults & LTC4280_OVERVOLTAGE_FAULT )
211  Serial.println(F(" OVERVOLTAGE DETECTED"));
212  }
213  else
214  Serial.println(F(" NO FAULTS DETECTED"));
215 
216  ack |= LTC4280_read(LTC4280_I2C_ADDRESS, LTC4280_SENSE_REG, &sense_code);
217  ack |= LTC4280_read(LTC4280_I2C_ADDRESS, LTC4280_SOURCE_REG, &source_code);
218  ack |= LTC4280_read(LTC4280_I2C_ADDRESS, LTC4280_ADIN_REG, &adin_code);
219 
220  sense_voltage = LTC4280_code_to_voltage(sense_code, sense_lsb, sense_resisive_ratio);
221  source_voltage = LTC4280_code_to_voltage(source_code, source_lsb, source_resisive_ratio);
222  adin_voltage = LTC4280_code_to_voltage(adin_code, adin_lsb, adin_resisive_ratio);
223 
224  Serial.print(F("\nSENSE VOLTAGE CODE: "));
225  Serial.println(sense_code, HEX);
226  Serial.print(F("SENSE CURRENT CALCULATED: "));
227  Serial.println((sense_voltage / SENSE_RESISTOR), 4);
228  Serial.print(F("\nSOURCE VOLTAGE CODE: "));
229  Serial.println(source_code, HEX);
230  Serial.print(F("SOURCE VOLTAGE CALCULATED: "));
231  Serial.println(source_voltage, 4);
232  Serial.print(F("\nADIN CODE: "));
233  Serial.println(adin_code, HEX);
234  Serial.print(F("VOLTAGE CALCULATED AT ADIN: "));
235  Serial.println(adin_voltage, 4);
236  Serial.print("\n");
237  delay(2000);
238  }
239  while (Serial.available()==false && ack== 0);
240  if (Serial.available())
241  read_int(); // clears the Serial.available()
242  return (ack);
243 }
