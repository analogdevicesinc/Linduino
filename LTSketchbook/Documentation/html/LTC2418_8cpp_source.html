6 The LTC2414/LTC2418 are 8-/16-channel (4-/8-differential) micropower 24-bit
7 delta-sigma analog-to-digital converters. They operate from 2.7V to 5.5V and
8 include an integrated oscillator, 2ppm INL and 0.2ppm RMS noise. They use
9 delta-sigma technology and provide single cycle settling time for multiplexed
10 applications. Through a single pin, the LTC2414/LTC2418 can be configured for
11 better than 110dB differential mode rejection at 50Hz or 60Hz +/-2%, or they can
12 be driven by an external oscillator for a user-defined rejection frequency. The
13 internal oscillator requires no external frequency setting components.

76 int8_t LTC2418_EOC_timeout(uint8_t cs, uint16_t miso_timeout)
77 // Checks for EOC with a specified timeout
78 {
79  uint16_t timer_count = 0; // Timer count for MISO
80  output_low(cs); //! 1) Pull CS low
81  while (1) //! 2) Wait for SDO (MISO) to go low
82  {
83  if (input(MISO) == 0) break; //! 3) If SDO is low, break loop
84  if (timer_count++>miso_timeout) // If timeout, return 1 (failure)
85  {
86  output_high(cs); // Pull CS high
87  return(1);
88  }
89  else
90  delay(1);
91  }
92  return(0);
93 }

96 // Reads the LTC2418

110 float LTC2418_code_to_voltage(int32_t adc_code, float LTC2418_lsb , int32_t LTC2418_offset_code)
111 // Calculates the LTC2418 input bipolar voltage
112 {
113  float adc_voltage;
114  adc_code = adc_code>>6; //! 1) Bit-shift ADC code to the right 6 bits
115  adc_code -= 8388608; //! 2) Convert ADC code from offset binary to binary
116  adc_voltage=((float)adc_code+LTC2418_offset_code)*(LTC2418_lsb); //! 3) Calculate voltage from ADC code, lsb, offset.
117  return(adc_voltage);
118 }

120 void LTC2418_cal_voltage(int32_t zero_code, int32_t fs_code, float zero_voltage, float fs_voltage, float *LTC2418_lsb , int32_t *LTC2418_offset_code)
121 // Calibrate the lsb
122 {
123  zero_code = zero_code >> 6; //! 1) Bit-shift zero code to the right 6 bits
124  zero_code -= 8388608; //! 2) Convert zero code from offset binary to binary
125  fs_code = fs_code >> 6; //! 3) Bit-shift full scale code to the right 6 bits
126  fs_code -= 8388608; //! 4) Convert full scale code from offset binary to binary
127 
128  float temp_offset;
129  *LTC2418_lsb = (fs_voltage-zero_voltage)/((float)(fs_code - zero_code)); //! 5) Calculate the LSB
130 
131  temp_offset = (zero_voltage/ *LTC2418_lsb) - zero_code; //! 6) Calculate Unipolar offset
132  temp_offset = (temp_offset > (floor(temp_offset) + 0.5)) ? ceil(temp_offset) : floor(temp_offset); //! 7) Round
133  *LTC2418_offset_code = (int32_t)temp_offset; //! 8) Cast as int32_t
134 }
