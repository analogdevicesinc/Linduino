112 void LT_PMBusDevice::setSpeed(uint32_t speed //!< Speed
113  )
114 {
115  uint8_t configAll;
116 
117  if (speed > 100000)
118  {
119  configAll = pmbus_->smbus()->readByte(address_, 0xD1);
120  configAll |= 1 << 1;
121  pmbus_->smbus()->writeByte(address_, 0xD1, configAll);
122  }
123  else
124  {
125  configAll = pmbus_->smbus()->readByte(address_, 0xD1);
126  configAll &= ~(1 << 1);
127  pmbus_->smbus()->writeByte(address_, 0xD1, configAll);
128  }
129  pmbus_->smbus()->i2cbus()->changeSpeed(speed);
130 }
