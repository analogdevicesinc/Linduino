95 //Generic function to write 68xx commands and write payload data. Function calculated PEC for tx_cmd data
96 void write_68(uint8_t total_ic , uint8_t tx_cmd[2], uint8_t data[])
97 {
98  const uint8_t BYTES_IN_REG = 6;
99  const uint8_t CMD_LEN = 4+(8*total_ic);
100  uint8_t *cmd;
101  uint16_t data_pec;
102  uint16_t cmd_pec;
103  uint8_t cmd_index;
104 
105  cmd = (uint8_t *)malloc(CMD_LEN*sizeof(uint8_t));
106  cmd[0] = tx_cmd[0];
107  cmd[1] = tx_cmd[1];
108  cmd_pec = pec15_calc(2, cmd);
109  cmd[2] = (uint8_t)(cmd_pec >> 8);
110  cmd[3] = (uint8_t)(cmd_pec);
111  cmd_index = 4;
112  for (uint8_t current_ic = total_ic; current_ic > 0; current_ic--) // executes for each LTC681x in daisy chain, this loops starts with
113  {
114  // the last IC on the stack. The first configuration written is
115  // received by the last IC in the daisy chain
116 
117  for (uint8_t current_byte = 0; current_byte < BYTES_IN_REG; current_byte++)
118  {
119  cmd[cmd_index] = data[((current_ic-1)*6)+current_byte];
120  cmd_index = cmd_index + 1;
121  }
122 
123  data_pec = (uint16_t)pec15_calc(BYTES_IN_REG, &data[(current_ic-1)*6]); // calculating the PEC for each Iss configuration register data
124  cmd[cmd_index] = (uint8_t)(data_pec >> 8);
125  cmd[cmd_index + 1] = (uint8_t)data_pec;
126  cmd_index = cmd_index + 2;
127  }
128 
129 
130  cs_low(CS_PIN);
131  spi_write_array(CMD_LEN, cmd);
132  cs_high(CS_PIN);
133  free(cmd);
134 }

136 //Generic function to write 68xx commands and read data. Function calculated PEC for tx_cmd data
137 int8_t read_68( uint8_t total_ic, uint8_t tx_cmd[2], uint8_t *rx_data)
138 {
139  const uint8_t BYTES_IN_REG = 8;
140  uint8_t cmd[4];
141  uint8_t data[256];
142  int8_t pec_error = 0;
143  uint16_t cmd_pec;
144  uint16_t data_pec;
145  uint16_t received_pec;
146 
147  // data = (uint8_t *) malloc((8*total_ic)*sizeof(uint8_t)); // This is a problem because it can fail
148 
149  cmd[0] = tx_cmd[0];
150  cmd[1] = tx_cmd[1];
151  cmd_pec = pec15_calc(2, cmd);
152  cmd[2] = (uint8_t)(cmd_pec >> 8);
153  cmd[3] = (uint8_t)(cmd_pec);
154 
155 
156  cs_low(CS_PIN);
157  spi_write_read(cmd, 4, data, (BYTES_IN_REG*total_ic)); //Read the configuration data of all ICs on the daisy chain into
158  cs_high(CS_PIN); //rx_data[] array
159 
160  for (uint8_t current_ic = 0; current_ic < total_ic; current_ic++) //executes for each LTC681x in the daisy chain and packs the data
161  {
162  //into the r_comm array as well as check the received Config data
163  //for any bit errors
164  for (uint8_t current_byte = 0; current_byte < BYTES_IN_REG; current_byte++)
165  {
166  rx_data[(current_ic*8)+current_byte] = data[current_byte + (current_ic*BYTES_IN_REG)];
167  }
168  received_pec = (rx_data[(current_ic*8)+6]<<8) + rx_data[(current_ic*8)+7];
169  data_pec = pec15_calc(6, &rx_data[current_ic*8]);
170  if (received_pec != data_pec)
171  {
172  pec_error = -1;
173  }
174  }
175 
176 
177  return(pec_error);
178 }

181 /*
182  Calculates and returns the CRC15
183  */
184 uint16_t pec15_calc(uint8_t len, //Number of bytes that will be used to calculate a PEC
185  uint8_t *data //Array of data that will be used to calculate a PEC
186  )
187 {
188  uint16_t remainder,addr;
189 
190  remainder = 16;//initialize the PEC
191  for (uint8_t i = 0; i<len; i++) // loops for each byte in data array
192  {
193  addr = ((remainder>>7)^data[i])&0xff;//calculate PEC table address
194 #ifdef MBED
195  remainder = (remainder<<8)^crc15Table[addr];
196 #else
197  remainder = (remainder<<8)^pgm_read_word_near(crc15Table+addr);
198 #endif
199  }
200  return(remainder*2);//The CRC15 has a 0 in the LSB so the remainder must be multiplied by 2
201 }

512 //helper function that parses voltage measurement registers
513 int8_t parse_cells(uint8_t current_ic, uint8_t cell_reg, uint8_t cell_data[], uint16_t *cell_codes, uint8_t *ic_pec)
514 {
515 
516  const uint8_t BYT_IN_REG = 6;
517  const uint8_t CELL_IN_REG = 3;
518  int8_t pec_error = 0;
519  uint16_t parsed_cell;
520  uint16_t received_pec;
521  uint16_t data_pec;
522  uint8_t data_counter = current_ic*NUM_RX_BYT; //data counter
523 
524 
525  for (uint8_t current_cell = 0; current_cell<CELL_IN_REG; current_cell++) // This loop parses the read back data into cell voltages, it
526  {
527  // loops once for each of the 3 cell voltage codes in the register
528 
529  parsed_cell = cell_data[data_counter] + (cell_data[data_counter + 1] << 8);//Each cell code is received as two bytes and is combined to
530  // create the parsed cell voltage code
531  cell_codes[current_cell + ((cell_reg - 1) * CELL_IN_REG)] = parsed_cell;
532  data_counter = data_counter + 2; //Because cell voltage codes are two bytes the data counter
533  //must increment by two for each parsed cell code
534  }
535 
536  received_pec = (cell_data[data_counter] << 8) | cell_data[data_counter+1]; //The received PEC for the current_ic is transmitted as the 7th and 8th
537  //after the 6 cell voltage data bytes
538  data_pec = pec15_calc(BYT_IN_REG, &cell_data[(current_ic) * NUM_RX_BYT]);
539 
540  if (received_pec != data_pec)
541  {
542  pec_error = 1; //The pec_error variable is simply set negative if any PEC errors
543  ic_pec[cell_reg-1]=1;
544  }
545  else
546  {
547  ic_pec[cell_reg-1]=0;
548  }
549  data_counter=data_counter+2;
550  return(pec_error);
551 }
