204 void wait_for_ready(uint8_t address)
205 {
206  uint8_t status;
207  uint8_t mask;
208  uint8_t page;
209 
210  // All devices poll on Ack, but the LTC2978/A will assert ALERTB if busy.
211  // By waiting 1ms, the only ALERTBs will be during a store or restore user all.
212  // This value can be tuned lower based on the clock rate, but should be well
213  // tested if the expectation is no faults except on store/restore.
214  delay(1);
215  wait_for_ack(address, MFR_COMMON);
216  if (is_ltc2978(address))
217  {
218  // Check if there is a CML or BUSY fault, and clear faults,
219  // but only if there are no other faults, because another
220  // thread or function may be interested in them.
221 
222  // Make sure we can use PAGE 0xFF to clear faults, but save
223  // off the mask and page so it can be restored.
224  mask = smbus->readByte(address, MFR_PAGE_FF_MASK);
225  smbus->writeByte(address, MFR_PAGE_FF_MASK, 0xFF);
226  page = smbus->readByte(address, PAGE);
227  smbus->writeByte(address, PAGE, 0xFF);
228 
229  // Clear if CML/BUSY but not other faults.
230  status = smbus->readByte(address, STATUS_BYTE);
231  if ((status & ((1 << CML) | (1 << BUSY))) &&
232  !(status & ~((1 << CML) | (1 << BUSY))))
233  smbus->sendByte(address, CLEAR_FAULTS);
234 
235  // Restore page and mask.
236  smbus->writeByte(address, PAGE, page);
237  smbus->writeByte(address, MFR_PAGE_FF_MASK, mask);
238  }
239  else
240  // All other devices have a busy bit to poll.
241  // Controllers also have a EEPROM register that can be
242  // polled to wait for internal operations to complete.
243  // This is not in general polling, because controllers
244  // still operate during this time.
245  wait_for_not_busy(address);
246 }

248 uint8_t wait_for_not_busy(uint8_t address)
249 {
250  uint8_t mfr_common;
251  // A real application should timeout at 4.1 seconds.
252  uint16_t timeout = 8192;
253 
254  while (timeout-- > 0)
255  {
256  mfr_common = smbus->readByte(address, MFR_COMMON);
257  // If too busy to answer, poll again.
258  if (mfr_common == 0xFF)
259  continue;
260  if ((mfr_common & (NOT_BUSY | NOT_PENDING | NOT_TRANS)) == (NOT_BUSY | NOT_PENDING | NOT_TRANS))
261  return 1;
262  }
263  return 0;
264 }

266 uint8_t wait_for_nvm_done(uint8_t address)
267 {
268  uint8_t mfr_eeprom_status;
269  // A real application should timeout at 4.1 seconds.
270  uint16_t timeout = 8192;
271 
272  while (timeout-- > 0)
273  {
274  wait_for_ack(address, 0x00);
275  wait_for_not_busy(address);
276  mfr_eeprom_status = smbus->readByte(address, MFR_EEPROM_STATUS);
277  if (mfr_eeprom_status == 0xFF)
278  continue;
279  if ((mfr_eeprom_status & 0xC0) == 0)
280  return 1;
281  }
282  return 0;
283 }

348 bool check_nvm_data_checksum(uint8_t address)
349 {
350  uint16_t w;
351  uint8_t bs[32];
352  uint16_t count;
353  uint8_t bpos = 0, wpos;
354  bool nok = false;
355 
356  // Unlock
357  unlock_nvm(address);
358 
359  // Read preamble and count
360  smbus->readWord(address, MFR_EE_DATA);
361  wait_for_ready(address);
362  count = smbus->readWord(address, MFR_EE_DATA);
363  wait_for_ready(address);
364 
365  for (wpos = 0; wpos < count/16; wpos++)
366  {
367  while (bpos < 32)
368  {
369  w = smbus->readWord(address, MFR_EE_DATA);
370  wait_for_ready(address);
371  bs[bpos++] = w & 0xFF;
372  bs[bpos++] = (w >> 8) & 0xFF;
373  }
374  // Compute CRC on 32 byte boundary
375  nok |= smbus->checkCRC(bs);
376  bpos = 0;
377  }
378 
379  lock_nvm(address);
380  return nok;
381 }

400 // If a device has a memory fault, and has been through a power cycle or reset, communcation
401 // will probably fail during the check, as the address of the device may not be correct.
402 bool check_health(uint8_t address)
403 {
404  bool res = true;
405 
406  if (is_ltc2978(address))
407  {
408  if (smbus->readByte(address, STATUS_CML) & STATUS_CML_MEM_FAULT)
409  {
410  res = false;
411  Serial.println(F(" Device has a Memory Fault"));
412  }
413  else if (smbus->readWord(address, STATUS_WORD) & ~STATUS_WORD_POWER_GOOD_B) // Assumes chann
414  {
415  res = false;
416  Serial.println(F(" Device has a Fault"));
417  }
418  else if (readItemp2977_8(address) > 85.0) // Really ITemp on LTC297X
419  {
420  res = false;
421  Serial.println(F(" Device is too hot"));
422  }
423  else
424  {
425  Serial.println(F(" Checksum not verified (LTC2978/A does not support direct read of NVM)"));
426  Serial.println(F(" Device is Ok (No Memory Fault, Not Hot)"));
427  }
428  }
429  else
430  {
431  if (smbus->readByte(address, STATUS_CML) & STATUS_CML_MEM_FAULT)
432  {
433  res = false;
434  Serial.println(F(" Device has a Memory Fault"));
435  }
436  else if (smbus->readWord(address, STATUS_WORD) & ~STATUS_WORD_POWER_GOOD_B)
437  {
438  res = false;
439  Serial.println(F(" Device has a Fault"));
440  }
441  // This could be converted to a positive check rather than a negative check.
442  // This depends on left to right execution.
443  else if ((!is_ltc297x(address)) && ((smbus->readByte(address, STATUS_MFR_SPECIFIC) & STATUS_MFR_SPECIFIC_NVM_CRC)))
444  {
445  res = false;
446  Serial.println(F(" Device has a Memory Fault"));
447  }
448  // This depends on left to right execution.
449  else if ((is_ltc2977(address) && (readItemp2977_8(address) > 85.0)) || (!is_ltc2977(address) && (readItemp(address) > 85.0)))
450  {
451  res = false;
452  Serial.println(F(" Device is hot"));
453  }
454  else if (check_nvm_data_checksum(address))
455  {
456  res = false;
457  Serial.println(F(" Device has an invalid CRC"));
458  }
459  else
460  Serial.println(F(" Device is Ok (No Memory Fault, Not Hot, CRC is Good)"));
461  }
462  return res;
463 }

520 void read_bytes(uint8_t address, uint16_t **data, uint16_t *count)
521 {
522  int i;
523 
524  // Ensure we are not busy
525  wait_for_ready(address);
526 
527  nvm_lock_reset(address);
528  unlock_nvm(address);
529 
530  // Read preamble
531  smbus->readWord(address, MFR_EE_DATA); // Read ID
532  wait_for_ready(address);
533  *count = smbus->readWord(address, MFR_EE_DATA);
534  wait_for_ready(address);
535 
536  // If we don't have memory yet, get some.
537  if (*data == NULL)
538  *data = (uint16_t *) malloc((*count) * sizeof(uint16_t));
539  else
540  {
541  free(*data);
542  *data = (uint16_t *) malloc((*count) * sizeof(uint16_t));
543  }
544 
545  // Read the data word at a time.
546  for (i = 0; i < *count; i++)
547  {
548  (*data)[i] = smbus->readWord(address, MFR_EE_DATA);
549  wait_for_ready(address);
550  }
551 
552  lock_nvm(address);
553 }

555 void write_bytes(uint8_t address, uint16_t **data, uint16_t *count)
556 {
557  int i;
558  uint16_t d;
559 
560  if (*data == NULL)
561  {
562  Serial.println(F(" No Data, please read data before writing"));
563  }
564  else
565  {
566  // Ensure we are not busy
567  wait_for_ready(address);
568 
569  // To improve reliability
570  nvm_lock_reset(address);
571 
572  // Erase the data to enable write
573  unlock_nvm(address);
574  smbus->writeByte(address, MFR_EE_ERASE, 0x2B); // Erase
575  // Very important to wait here
576  wait_for_ready(address);
577  lock_nvm(address);
578 
579  // Write the data
580  unlock_nvm(address);
581  for (i = 0; i < *count; i++)
582  {
583  smbus->writeWord(address, MFR_EE_DATA, (*data)[i]);
584  // Very important to wait here
585  wait_for_ready(address);
586  }
587  lock_nvm(address);
588 
589  // Verify the data
590  unlock_nvm(address); // Read preamble
591  smbus->readWord(address, MFR_EE_DATA); // Read ID
592  wait_for_ready(address);
593  *count = smbus->readWord(address, MFR_EE_DATA);
594  wait_for_ready(address);
595  // Read the data word at a time.
596  for (i = 0; i < *count; i++)
597  {
598  if (d = (smbus->readWord(address, MFR_EE_DATA) != (*data)[i]))
599  {
600  Serial.print(F(" Data Mismatch ("));
601  Serial.print(i, DEC);
602  Serial.print(F(","));
603  Serial.print(d, HEX);
604  Serial.print(F(","));
605  Serial.print((*data)[i], HEX);
606  Serial.println(F(")"));
607  }
608  }
609  lock_nvm(address);
610  }
611 }

829  wait_for_ready(LTC2977_8_I2C_ADDRESS);

831  // All is ok, so clear the fault logs so they can be used if enabled.
832  clear_fault_logs();
833  wait_for_ready(LTC3880_I2C_ADDRESS);
834  wait_for_ready(LTC2974_I2C_ADDRESS);
835  wait_for_ready(LTC2977_8_I2C_ADDRESS);
836

838  // a power cycle and rails with turn off and on.
839  Serial.println(F("Check NVM"));
840  if (!check_health(LTC3880_I2C_ADDRESS))

872  {
873  Serial.print(F("ADDR 0x"));
874  Serial.println(*addresses++, HEX);
875  }
876  break;
877  case 12:
878  // Group protocol is a way to send multiple devices in one command as an atomic transaction.
879  // It begins with a start, follows with address/command transactions, and ends in a stop.
880  // At the stop, all devices respond to their command at the same time. It is not legal
881  // or defined how things behave if it sends more than one command to a device withing one
882  // group protocol transaction..
883 
884  // The reason for using this is that the managers do not have a reset command like the controllers,
885  // so the code next to mix restore commands to managers with reset commands to controllers.
886 
887  // Tell the PMBus object to start saving commands, but don't do anything on PMBus.
888  smbus->beginStoring();

890  smbus->sendByte(LTC3880_I2C_ADDRESS, MFR_RESET);
891  smbus->sendByte(LTC2974_I2C_ADDRESS, RESTORE_USER_ALL);
892  smbus->sendByte(LTC2977_8_I2C_ADDRESS, RESTORE_USER_ALL);
893  // Tell the PMBus object to send all the comands atomically as a Group Command Protocol transaction.
894  // At the stop the controller will reset, and the managers will do a restore. This effectivly restarts
895  // all devices at the same time so that proper sequencing takes place.
896  smbus->execute();
897 
898  // Wait for each device to finish.
899  wait_for_ready(LTC3880_I2C_ADDRESS);
900  wait_for_ready(LTC2974_I2C_ADDRESS);
901  wait_for_ready(LTC2977_8_I2C_ADDRESS);
902 
903  break;
904  default:
905  Serial.println(F("Incorrect Option"));
906  break;
907  }
908  print_prompt();
909  }
