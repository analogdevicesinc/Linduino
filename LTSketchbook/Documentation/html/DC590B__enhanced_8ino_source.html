121 void byte_to_hex(byte value)
122 // convert a byte to two hex characters
123 {
124  byte_to_hex_buffer[0]=hex_digits[value>>4]; // get upper nibble
125  byte_to_hex_buffer[1]=hex_digits[(value & 0x0F)]; // get lower nibble
126  byte_to_hex_buffer[2]='\0'; // add NULL at end
127 }

129 byte read_hex()
130 // read 2 hex characters from the serial buffer and convert
131 // them to a byte
132 {
133  byte data;
134  hex_to_byte_buffer[2]=get_char();
135  hex_to_byte_buffer[3]=get_char();
136  data = strtol(hex_to_byte_buffer, NULL, 0);
137  return(data);
138 }

140 char get_char()
141 // get the next character either from the serial port
142 // or the recording buffer
143 {
144  char command='\0';
145  if (recording_mode != playback)
146  {
147  // read a command from the serial port
148  while (Serial.available() <= 0);
149  return(Serial.read());
150  }
151  else
152  {
153  // read a command from the recording buffer
154  if (recording_index < RECORDING_SIZE)
155  {
156  command = recording_buffer[recording_index++];
157  // disregard loop commands during playback
158  if (command == 'w') command='\1';
159  if (command == 't') command='\1';
160  if (command == 'v') command='\1';
161  if (command == 'u') command='\1';
162  }
163  else
164  command = '\0';
165  if (command == '\0')
166  {
167  recording_index = 0;
168  recording_mode = off;
169  }
170  return(command);
171  }
172 }

222  delay_value<<=8;
223  delay_value|=read_hex();
224  delay(delay_value);
225  break;
226  case 'g':
227  // IO pin low
228  output_low(QUIKEVAL_GPIO);
229  break;
230  case 'G':
231  // IO pin high
232  output_high(QUIKEVAL_GPIO);
233  break;
234  case 'H': // wait for MISO to go high with a timeout
235  delay_count = 0;
236  while (1)
237  {
238  if (input(MISO)==1) break; // MISO is high so quit
239  if (delay_count++>MISO_TIMEOUT)
240  {
241  //Serial.print('T'); // timeout occurred. Print 'T'
242  break;
243  }
244  else delay(1);
245  }
246  break;
247  case 'i':
248  // send controller id string
249  pseudo_reset = 0;
250  Serial.print(id_string);
251  Serial.print('\0');
252  Serial.flush();
253  break;
254  case 'I':
255 #ifdef spoof_board_id
256  Serial.print(spoof_board_id);
257  Serial.print('\0');
258  Serial.print('\0');
259 #else

262  pseudo_reset = 0;
263  byte_count = read_quikeval_id_string(&ui_buffer[0]);
264  if (byte_count!=0)
265  {
266  Serial.print(ui_buffer);
267  Serial.print("\n\0");
268  Serial.flush();
269  }
270 #endif
271  break;
272  case 'K':
273  // Bang pin. The pin assignments are :
274  // 0: PIND2, Arduino 2
275  // 1: PIND3, Arduino 3
276  // 2: PIND4, Arduino 4
277  // 3: PIND5, Arduino 5
278  // 4: PIND6, Arduino 6
279  // 5: PIND6, Arduino 7
280 
281  pin_value = get_char(); // read the value
282 // pin = get_char()-0x30; // read the pin
283  pin = get_char(); // read the pin
284 // if (pin_value == '0') digitalWrite(pin+2, LOW);
285 // else digitalWrite(pin+2, HIGH);
286 
287  digitalWrite(pin-0x30+2, pin_value == '0' ? LOW : HIGH);
288  break;
289  case 'k':
290  // Get pin state. The digial pin assignments are the same as for 'K',
291  // Analog pin mappings as follows:
292  // 6: Analog A0 (read only)
293  // 7: Analog A1 (read only)
294  // 8: Analog A2 (read only)
295  // 9: Analog A3 (read only)
296  pin = get_char(); // read the pin
297  if (pin <= '5')
298  {
299  Serial.print(digitalRead(pin-0x30+2));
300  }
301  else if (pin == 'C') // To read MISO pin
302  {
303  int val;
304  //pin = read_hex();
305  val = digitalRead(12);
306  byte_to_hex(val);
307  Serial.print(byte_to_hex_buffer);
308  }
309  else
310  {
311  int val = analogRead(pin - '6');
312  if (val < 10)
313  {
314  Serial.print("000");
315  }
316  else if (val < 100)
317  {
318  Serial.print("00");
319  }
320  else if (val < 1000)
321  {
322  Serial.print("0");
323  }
324  Serial.print(val);
325  break;
326  }
327  break;
328  case 'j':
329  // Set pin mode. The pin assignments are the same as for 'K':
330  pin_value = get_char(); // read the value
331  pin = get_char(); // read the pin
332  pinMode(pin-0x30+2, pin_value == '0' ? INPUT : OUTPUT);
333  break;
334  case 'L':
335  // wait for MISO to go low with a timeout
336  delay_count = 0;
337  while (1)
338  {
339  if (input(MISO)==0) break; // MISO is low so quit
340  if (delay_count++>MISO_TIMEOUT)
341  {
342  //Serial.print('T'); // timeout occurred. Print 'T'
343  break;
344  }
345  else delay(1);
346  }
347  break;
348  case 'M':
349  // change the serial mode
350  command = get_char();
351  switch (command)
352  {
353  case 'I':
354  // I2C mode
355  serial_mode = i2c_mode;
356  // enable_main_I2C();
357  quikeval_I2C_connect();
358  break;
359  case 'S':
360  // spi mode
361  serial_mode = spi_mode;
362  // Need to send command to disable LTC4302
363  // enable_main_SPI();
364  quikeval_SPI_connect();
365  break;
366  case 'X': // axillary I2C mode - no hardware action necessary, always available.
367  serial_mode = i2c_auxiliary_mode;
368  quikeval_SPI_connect();
369  break;
370  case '0': // New for Enhanced version - set SPI mode on the fly. (Mainly for Altera SPI-Avalon MM bridge)
371  SPI.setDataMode(SPI_MODE0);
372  break;
373  case '1':
374  SPI.setDataMode(SPI_MODE1);
375  break;

377  SPI.setDataMode(SPI_MODE2);
378  break;
379  case '3':

384  break;
385  case 'b':
386  SPI.setClockDivider(SPI_CLOCK_DIV4);
387  break;
388  case 'c':
389  SPI.setClockDivider(SPI_CLOCK_DIV8);
390  break;

392  SPI.setClockDivider(SPI_CLOCK_DIV16);

403  }

405  break;
406  case 'p':
407  // I2C stop
408  // Switching BACK to the i2c_stop() function call. See Rev 1405, LT_i2c.cpp, Linduino One branch
409  if (serial_mode!=spi_mode) i2c_stop(); // TWCR=(1<<TWINT) | (1<<TWEN) | (1<<TWSTO); // I2C stop //i2c_stop();
410  // if(serial_mode == i2c_auxiliary_mode) i2c_stop();
411  break;
412  case 'P':
413  // ping
414  Serial.print('P');
415  delay(5);
416  break;
417  case 'Q':
418  // Read byte in I2C mode only. Add ACK
419  switch (serial_mode)
420  {
421  case i2c_mode:
422  rx_data = i2c_read(WITH_ACK);
423  byte_to_hex(rx_data);
424  Serial.print(byte_to_hex_buffer);

426  case i2c_auxiliary_mode:
427  rx_data = i2c_read(WITH_ACK);

431  }
432  break;
433 
434  case 'r':
435  rx_data = spi_read(0);
436  byte_to_hex(rx_data);
437  Serial.print(byte_to_hex_buffer);
438  break;
439 
440  case 'R':
441  // Read byte, add NACK in I2C mode
442  switch (serial_mode)
443  {
444  case spi_mode:
445  rx_data = spi_read(0);
446  byte_to_hex(rx_data);
447  Serial.print(byte_to_hex_buffer);
448  break;
449  case i2c_mode:
450  rx_data = i2c_read(WITH_NACK);
451  byte_to_hex(rx_data);
452  Serial.print(byte_to_hex_buffer);
453  break;
454  case i2c_auxiliary_mode:
455  rx_data = i2c_read(WITH_NACK);
456  byte_to_hex(rx_data);
457  Serial.print(byte_to_hex_buffer);
458  break;
459  }
460  break;
461  case 's': // I2C start
462  if (serial_mode == i2c_mode) i2c_start();
463  if (serial_mode == i2c_auxiliary_mode) i2c_start();
464  break;
465  case 'S': // send byte
466  tx_data = read_hex();
467  switch (serial_mode)
468  {
469  case spi_mode:
470  spi_write(tx_data);
471  break;
472  case i2c_mode:
473  if (i2c_write(tx_data)==1) Serial.print('N');
474  break;
475  case i2c_auxiliary_mode:
476  if (i2c_write(tx_data)==1) Serial.print('N');
477  break;
478  }
479  break;
480  case 't': // recording loop
481  recording_index = 0;
482  do
483  {
484  command = get_char();
485  if (command == 'u') // stop recording
486  {
487  recording_buffer[recording_index]='\0';
488  recording_index = 0;
489  break;
490  }
491  else // add character to recording buffer
492  {
493  if (recording_index < RECORDING_SIZE) recording_buffer[recording_index++]=command;
494  }
495  }
496  while (1);
497  break;
498  case 'T': // transceive byte
499  tx_data = read_hex();
500  if (serial_mode == spi_mode)
501  {
502  rx_data = spi_read(tx_data);
503  byte_to_hex(rx_data);
504  Serial.print(byte_to_hex_buffer);
505  }
506  break;
507  case 'v': // echo recording loop
508  Serial.print(recording_buffer);
509  break;
510  case 'w':
511  recording_mode = playback;
512  break;
513  case 'x':
514  output_low(QUIKEVAL_CS);
515  break;
516  case 'X':
517  output_high(QUIKEVAL_CS);
518  break;
519  case 'Z': // line feed

522  break;
