137 LT_PMBusMath::fl32_t LT_PMBusMath::lin11_to_fl32 (LT_PMBusMath::lin11_t xin)
138 {
139 
140  // Repackage the Linear11 in an IEEE754 format (sign, exponent, mantissa)
141  LT_PMBusMath::lin11_t lin11_sign, lin11_exp, lin11_exp_sign, lin11_mant, lin11_nmant;
142  LT_PMBusMath::fl32_t xout;
143 
144  // Extract the sign, exponent and mantissa
145  lin11_sign = (xin & ((lin11_t) lin11_mant_sign_mask)) << lin11_exp_width;
146  lin11_exp = (xin >> lin11_mant_width) & ((lin11_t) lin11_exp_mask);
147  lin11_exp_sign = lin11_exp & ((lin11_t) lin11_exp_sign_mask);
148  lin11_mant = (xin & ((lin11_t) lin11_mant_mask));
149 
150  // Add the exponent bias of the target IEEE 754 format to the L11 exponent
151  if (lin11_exp_sign != 0)
152  {
153  lin11_exp = (lin11_exp ^ ((lin11_t) lin11_exp_mask)) + 1;
154  lin11_exp = fl32_exp_bias - lin11_exp + lin11_mant_width - 2;
155  }
156  else
157  {
158  lin11_exp = fl32_exp_bias + lin11_exp + lin11_mant_width - 2;
159  }
160 
161  // Invert the mantissa sign for the two's complement representation
162  if (lin11_sign != 0)
163  {
164  lin11_nmant = (lin11_mant ^ ((lin11_t) lin11_mant_mask)) + 1;
165 
166  // The most negative mantissa is a corner case and has to be shifted
167  if (lin11_nmant == lin11_mant)
168  {
169  lin11_mant = lin11_nmant >> 1;
170  lin11_exp++;
171  }
172  else
173  {
174  lin11_mant = lin11_nmant;
175  }
176  }
177 
178  // Normalize the Linear11 mantissa to fit the IEEE 754 format structure
179  if (lin11_mant != 0 )
180  {
181  while ((lin11_mant & (~((lin11_t) lin11_ieee_mant_mask))) == 0 )
182  {
183  lin11_mant = lin11_mant << 1;
184  lin11_exp--;
185  }
186  }
187  else
188  {
189  lin11_exp = 0; // Special case of Zero
190  }
191  lin11_mant = lin11_mant & ((lin11_t) lin11_ieee_mant_mask);
192 
193  // Shift the L11 sign, exponent and mantissa to the target IEEE format
194  xout = ((fl32_t) lin11_sign) << (fl32_width - lin11_width);
195  xout |= ((fl32_t) lin11_exp) << fl32_mant_width;
196  xout |= ((fl32_t) lin11_mant) << (fl32_mant_width - lin11_ieee_mant_width);
197 
198  return xout;
199 }

208 LT_PMBusMath::fl32_t LT_PMBusMath::lin16_to_fl32 (LT_PMBusMath::lin16_t lin16_mant, LT_PMBusMath::lin16m_t vout_mode)
209 {
210 
211  // Repackage the Linear16 in an IEEE 754 format (sign, exponent, mantissa)
212  LT_PMBusMath::lin16m_t lin16_exp_sign;
213  LT_PMBusMath::fl32_t lin16_exp,xout;
214 
215  // Extract the exponent and mantissa (sign = 0)
216  lin16_exp = (fl32_t) (vout_mode & ((lin16m_t) lin16_exp_mask));
217  lin16_exp_sign = vout_mode & ((lin16m_t) lin16_exp_sign_mask);
218 
219  // Add the exponent bias of the target IEEE 754 format to the L16 exponent
220  if (lin16_exp_sign != (lin16m_t) 0)
221  {
222  lin16_exp = (lin16_exp ^ ((fl32_t) lin16_exp_mask)) + 1;
223  lin16_exp = (fl32_t)(fl32_exp_bias + lin16_ieee_mant_width) - lin16_exp;
224  }
225  else
226  {
227  lin16_exp = (fl32_t)(fl32_exp_bias + lin16_ieee_mant_width) + lin16_exp;
228  }
229 
230  // Normalize the Linear16 mantissa to fit the IEEE 754 format structure
231  if (lin16_mant != 0 )
232  {
233  while ((lin16_mant & (~((lin16_t) lin16_ieee_mant_mask))) == 0 )
234  {
235  lin16_mant = lin16_mant << 1;
236  lin16_exp--;
237  }
238  }
239  else
240  {
241  lin16_exp = 0; // Special case of Zero
242  }
243  lin16_mant = lin16_mant & ((lin16_t) lin16_ieee_mant_mask);
244 
245  // Shift the L16 mantissa and exponent to the target IEEE format (sign=0)
246  xout = lin16_exp << fl32_mant_width;
247  xout |= ((fl32_t) lin16_mant) << (fl32_mant_width - lin16_ieee_mant_width);
248 
249  return xout;
250 }

260 LT_PMBusMath::lin16_t LT_PMBusMath::fl32_to_lin16 (LT_PMBusMath::fl32_t xin, LT_PMBusMath::lin16m_t vout_mode)
261 {
262 
263  // Variables for the exponent, mantissa and mantissa shift
264  LT_PMBusMath::fl32_t fl32_exp, fl32_mant, rnd_msb, tie_mask;
265  LT_PMBusMath::lin16m_t lin16_exp_sign, lin16_exp;
266  LT_PMBusMath::lin16_t xout;
267  int shift;
268  unsigned int ushift;
269 
270  // Extract the sign, exponent and mantissa of the Float32 number
271  if ( ( xin >> (fl32_width - 1)) != 0 )
272  {
273  xout = 0; // All negative Float32 values get mapped to zero
274  }
275  else
276  {
277  // Extract the exponent and mantissa of the Float32 number
278  fl32_exp = (((fl32_t) fl32_exp_mask) & xin) >> fl32_mant_width;
279  fl32_mant = ((fl32_t) fl32_mant_mask) & xin;
280 
281  // Add the implicit 1.mmm to the mantissa (L16 is not normalized)
282  fl32_mant |= ((fl32_t) (((fl32_t) (1)) << fl32_mant_width));
283 
284  // Calculate the required shift for the mantissa
285  lin16_exp = vout_mode & ((lin16m_t) lin16_exp_mask);
286  lin16_exp_sign = lin16_exp & ((lin16m_t) lin16_exp_sign_mask);
287  shift = (int) (fl32_exp_bias + fl32_mant_width) - (int) fl32_exp;
288  if (lin16_exp_sign != (lin16m_t) 0)
289  {
290  shift -= (int)(lin16_exp ^ ((lin16m_t) lin16_exp_mask)) + 1;
291  }
292  else
293  {
294  shift += (int) lin16_exp;
295  }
296 
297  // Right shift the mantissa and round
298  if (shift < (fl32_mant_width + 1 - lin16_width))
299  {
300  xout = (lin16_t) lin16_max_value; // Saturate to 0xFFFF
301  }
302  else if (shift > (fl32_mant_width + 1))
303  {
304  xout = 0; // Underflow to zero
305  }
306  else // Shift and round
307  {
308  ushift = (unsigned) shift;
309  xout = (lin16_t) (fl32_mant >> ushift);
310  rnd_msb = (((fl32_t) (1)) << (ushift - 1));
311  tie_mask = ((((fl32_t) (1)) << (ushift + 1)) - 1);
312 
313  // When rounding, mantissa overflow is not possible (right shift>=1)
314  if (((rnd_msb & fl32_mant)!=0)&&((tie_mask & fl32_mant)!= rnd_msb))
315  {
316  xout++;
317  }
318  }
319  }
320  return xout;
321 }

330 LT_PMBusMath::lin11_t LT_PMBusMath::fl32_to_lin11 (LT_PMBusMath::fl32_t xin)
331 {
332 
333  // Variables for the floating point sign, exponent and mantissa bit fields
334  LT_PMBusMath::fl32_t fl32_exp, fl32_mant, rnd_msb, tie_mask;
335  LT_PMBusMath::lin11_t lin11_sign, lin11_mant;
336  LT_PMBusMath::slin11_t lin11_exp;
337  uchar_t mant_shift;
338  const LT_PMBusMath::lin11_t normal_n1 = (lin11_t) lin11_norm_n1;
339  const LT_PMBusMath::lin11_t denormal_n1 = (lin11_t) lin11_den_n1;
340 
341  // Extract the sign, exponent and mantissa of the Float32 number
342  fl32_exp = (((fl32_t) fl32_exp_mask) & xin) >> fl32_mant_width;
343  fl32_mant = ((fl32_t) fl32_mant_mask) & xin;
344  fl32_mant |= (((fl32_t) (1)) << fl32_mant_width); // Add the 1.mmm
345 
346  // Convert the extracted sign and signed exponent to Linear11
347  lin11_sign = ((lin11_t)(xin >> (fl32_width - lin11_width)));
348  lin11_sign &= (((lin11_t) (1)) << (lin11_width-1));
349  lin11_exp = (slin11_t)(fl32_exp) - fl32_exp_bias - lin11_ieee_mant_width;
350 
351  // Determine if the corresponding Linear11 number is normalized or denormal
352  mant_shift = (uchar_t) (fl32_mant_width - lin11_ieee_mant_width);
353  if (lin11_exp < (-(slin11_t) lin11_max_n_exp)) // Denormal Linear11
354  {
355  mant_shift -= (uchar_t) (lin11_exp + ((slin11_t) lin11_max_n_exp));
356  lin11_exp = -((slin11_t) lin11_max_n_exp);
357  }
358 
359  // Calculate the Linear11 mantissa and check if rounding up is needed
360  // If the mantissa overflowed after the rounding, increment the exponent
361  if (mant_shift > (uchar_t)(fl32_mant_width + 1))
362  {
363  lin11_mant = 0;
364  }
365  else
366  {
367  lin11_mant = (lin11_t) (fl32_mant >> mant_shift);
368  rnd_msb = (((fl32_t) (1)) << (fl32_t) (mant_shift - (uchar_t) 1));
369  tie_mask = ((((fl32_t) (1)) << (fl32_t) (mant_shift + (uchar_t) 1))-1);
370 
371  if (((rnd_msb & fl32_mant) !=0) && ((tie_mask & fl32_mant) != rnd_msb))
372  {
373  // Increment the mantissa and check for mantissa overflow
374  lin11_mant++;
375  if ((lin11_mant & ((lin11_t) lin11_ieee_mant_mask)) == 0)
376  {
377  lin11_exp++;
378  lin11_mant = lin11_mant >> 1;
379  }
380  }
381  }
382 
383  // Convert the mantissa to two's complement
384  if (lin11_sign != 0)
385  {
386  lin11_mant = ((~lin11_mant + 1) & (lin11_t) lin11_mant_mask);
387  }
388 
389  // Normalize the special case of -1 x 2^N
390  if ((lin11_mant == denormal_n1) && (lin11_exp > -(slin11_t)lin11_max_n_exp))
391  {
392  lin11_mant = normal_n1;
393  lin11_exp--;
394  }
395 
396  // Saturate to maximum if the exponent is too large
397  if (lin11_exp > (slin11_t) lin11_max_p_exp)
398  {
399  lin11_exp = (slin11_t) lin11_max_p_exp;
400  if (lin11_sign != 0)
401  {
402  lin11_mant = normal_n1;
403  }
404  else
405  {
406  lin11_mant = (((lin11_t) (1)) << (lin11_mant_width - 1)) - 1;
407  }
408  }
409 
410  // Select a unique representation for zero - if the mantissa is zero,
411  // return 0x0000 (zero exponent)
412  if (lin11_mant == 0) lin11_exp = 0;
413 
414  lin11_exp = (slin11_t)((lin11_t) lin11_exp << (lin11_t) lin11_mant_width);
415  lin11_exp &= ((lin11_t) lin11_bit_mask);
416  return ((lin11_t)(lin11_exp | lin11_mant));
417 }
