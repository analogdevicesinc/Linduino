152 //! Repeats Linduino loop
153 void loop()
154 {
155  int8_t ack = 0; // I2C acknowledge indicator
156  static uint8_t user_command; // The user input command
157 
158  static int8_t VDD_or_SENSE_monitoring = 1; // Choose whether VIN is sensed at VDD pin or SENSE pin (A2 bit of CONTROL Register A)
159  static int8_t ADIN_or_SENSE = 1; // Choose whether power multiplier uses ADIN pin or SENSE pin as voltage input (A0 bit in CONTROL Register A)
160 
161  static float scale; // Stores division ration for resistive divider on ADIN pin. Configured inside "Settings" menu.
162 
163  if (demo_board_connected) //! Do nothing if the demo board is not connected
164  {
165  if (Serial.available()) //! Do nothing if serial is not available
166  {
167  user_command = read_int(); //! Read user input command
168  if (user_command != 'm')
169  Serial.println(user_command);
170  Serial.println();
171  ack = 0;
172  switch (user_command) //! Prints the appropriate submenu
173  {
174  case 1:
175  ack |= menu_1_continuous_mode(VDD_or_SENSE_monitoring, ADIN_or_SENSE, scale); // Continuous Mode
176  break;
177 
178  case 2:
179  ack |= menu_2_snapshot_mode(VDD_or_SENSE_monitoring, ADIN_or_SENSE);
180  break;
181 
182  case 3:
183  ack |= menu_3_alert(ADIN_or_SENSE); // Alert
184  break;
185 
186  case 4:
187  ack |= menu_4_shutdown(); // Shutdown Down Mode
188  break;
189 
190  case 5:
191  menu_5_settings(&VDD_or_SENSE_monitoring, &ADIN_or_SENSE, &scale); // Settings
192  break;
193 
194  case 6:
195  ack |= menu_6_clear_min_max(); // Clear Min/Max
196  break;
197 
198  default:
199  Serial.println("Incorrect Option");
200  break;
201  }
202  if (ack != 0)
203  Serial.println(ack_error);
204  Serial.print(F("*************************"));
205  print_prompt();
206  }
207  }
208 }
