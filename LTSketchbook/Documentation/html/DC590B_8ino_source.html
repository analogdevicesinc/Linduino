3 DC590B USB to Serial Controller
4 
5 This file contains the routines to emulate the DC590B USB to Serial Converter. All commands
6 are supported except Uxxy the Write Port D bus. Added the 'D' delay ms command.
7 With this program, the Linduino can be used by the QuikEval program running on a PC
8 to communicate with QuikEval compatible demo boards.
9 
10 The Kxy bit bang command uses the following pin mappings :
11 0-Linduino 2
12 1-Linduino 3

14 3-Linduino 5

110 void byte_to_hex(byte value)
111 // convert a byte to two hex characters
112 {
113  byte_to_hex_buffer[0]=hex_digits[value>>4]; // get upper nibble
114  byte_to_hex_buffer[1]=hex_digits[(value & 0x0F)]; // get lower nibble
115  byte_to_hex_buffer[2]='\0'; // add NULL at end
116 }

118 byte read_hex()
119 // read 2 hex characters from the serial buffer and convert
120 // them to a byte
121 {
122  byte data;
123  hex_to_byte_buffer[2]=get_char();
124  hex_to_byte_buffer[3]=get_char();
125  data = strtol(hex_to_byte_buffer, NULL, 0);
126  return(data);
127 }

129 char get_char()
130 // get the next character either from the serial port
131 // or the recording buffer
132 {
133  char command='\0';
134  if (recording_mode != playback)
135  {
136  // read a command from the serial port
137  while (Serial.available() <= 0);
138  return(Serial.read());
139  }
140  else
141  {
142  // read a command from the recording buffer
143  if (recording_index < RECORDING_SIZE)
144  {
145  command = recording_buffer[recording_index++];
146  // disregard loop commands during playback
147  if (command == 'w') command='\1';
148  if (command == 't') command='\1';
149  if (command == 'v') command='\1';
150  if (command == 'u') command='\1';
151  }
152  else
153  command = '\0';
154  if (command == '\0')
155  {
156  recording_index = 0;
157  recording_mode = off;
158  }
159  return(command);
160  }
161 }

195 void loop()
196 {
197  byte tx_data;
198  byte rx_data;
199  byte pin_value;
200  int delay_value;
201  int pin;
202  char command;
203  int byte_count;
204  long delay_count;
205  command = get_char();
206  switch (command)
207  {
208  case 'D':
209  // delay milliseconds
210  delay_value = read_hex();
211  delay_value<<=8;
212  delay_value|=read_hex();
213  delay(delay_value);
214  break;
215  case 'g':
216  // IO pin low
217  output_low(QUIKEVAL_GPIO);
218  break;
219  case 'G':
220  // IO pin high
221  output_high(QUIKEVAL_GPIO);
222  break;
223  case 'H': // wait for MISO to go high with a timeout
224  delay_count = 0;
225  while (1)
226  {
227  if (input(MISO)==1) break; // MISO is high so quit
228  if (delay_count++>MISO_TIMEOUT)
229  {
230  //Serial.print('T'); // timeout occurred. Print 'T'
231  break;
232  }
233  else delay(1);
234  }
235  break;
236  case 'i':
237  // send controller id string
238  pseudo_reset = 0;
239  Serial.print(id_string);
240  Serial.print('\0');
241  Serial.flush();
242  break;
243  case 'I':
244  // get controller id string
245  quikeval_SPI_connect(); // Connect SPI to main data port
246  pseudo_reset = 0;
247  byte_count = read_quikeval_id_string(&ui_buffer[0]);
248  if (byte_count!=0)
249  {
250  Serial.print(ui_buffer);
251  //Serial.print("LTC4261,Cls,D4261,01,01,DC,DC998A,--------------\n\0");
252  Serial.print('\0');
253  Serial.flush();
254  }
255  break;
256  case 'K':
257  // Bang pin. The pin assignments are :
258  // 0: PIND2, Arduino 2
259  // 1: PIND3, Arduino 3
260  // 2: PIND4, Arduino 4
261  // 3: PIND5, Arduino 5
262  // 4: PIND6, Arduino 6
263  // 5: PIND6, Arduino 7
264  pin_value = get_char(); // read the value
265  pin = get_char()-0x30; // read the pin
266  if (pin_value == '0') digitalWrite(pin+2, LOW);
267  else digitalWrite(pin+2, HIGH);
268  break;
269  case 'L':
270  // wait for MISO to go low with a timeout
271  delay_count = 0;
272  while (1)
273  {
274  if (input(MISO)==0) break; // MISO is low so quit
275  if (delay_count++>MISO_TIMEOUT)
276  {
277  //Serial.print('T'); // timeout occurred. Print 'T'
278  break;
279  }
280  else delay(1);
281  }
282  break;
283  case 'M':
284  // change the serial mode
285  command = get_char();
286  switch (command)
287  {
288  case 'I':
289  // I2C mode
290  serial_mode = i2c_mode;
291  // enable_main_I2C();
292  quikeval_I2C_connect();
293  break;
294  case 'S':
295  // spi mode
296  serial_mode = spi_mode;
297  // Need to send command to disable LTC4302
298  // enable_main_SPI();
299  quikeval_SPI_connect();
300  break;
301  case 'X':
302  // axillary I2C mode - no hardware action necessary, always available.
303  serial_mode = i2c_auxiliary_mode;
304  quikeval_SPI_connect();
305  break;
306  }
307  //delay(1);
308  break;
309  case 'p':
310  // I2C stop
311  if (serial_mode!=spi_mode) TWCR=(1<<TWINT) | (1<<TWEN) | (1<<TWSTO); // I2C stop //i2c_stop();
312  // if(serial_mode == i2c_auxiliary_mode) i2c_stop();
313  break;
314  case 'P':
315  // ping
316  Serial.print('P');
317  delay(5);
318  break;
319  case 'Q':
320  // Read byte in I2C mode only. Add ACK
321  switch (serial_mode)
322  {
323  case i2c_mode:
324  rx_data = i2c_read(WITH_ACK);
325  byte_to_hex(rx_data);
326  Serial.print(byte_to_hex_buffer);
327  break;
328  case i2c_auxiliary_mode:
329  rx_data = i2c_read(WITH_ACK);
330  byte_to_hex(rx_data);
331  Serial.print(byte_to_hex_buffer);
332  break;
333  }
334  break;
335 
336  case 'r':
337  rx_data = spi_read(0);
338  byte_to_hex(rx_data);
339  Serial.print(byte_to_hex_buffer);
340  break;
341 
342  case 'R':
343  // Read byte, add NACK in I2C mode
344  switch (serial_mode)
345  {
346  case spi_mode:
347  rx_data = spi_read(0);
348  byte_to_hex(rx_data);
349  Serial.print(byte_to_hex_buffer);
350  break;
351  case i2c_mode:
352  rx_data = i2c_read(WITH_NACK);
353  byte_to_hex(rx_data);
354  Serial.print(byte_to_hex_buffer);
355  break;
356  case i2c_auxiliary_mode:
357  rx_data = i2c_read(WITH_NACK);
358  byte_to_hex(rx_data);
359  Serial.print(byte_to_hex_buffer);
360  break;
361  }
362  break;
363  case 's': // I2C start
364  if (serial_mode == i2c_mode) i2c_start();
365  if (serial_mode == i2c_auxiliary_mode) i2c_start();
366  break;
367  case 'S': // send byte
368  tx_data = read_hex();
369  switch (serial_mode)
370  {
371  case spi_mode:
372  spi_write(tx_data);
373  break;
374  case i2c_mode:
375  if (i2c_write(tx_data)==1) Serial.print('N');
376  break;
377  case i2c_auxiliary_mode:
378  if (i2c_write(tx_data)==1) Serial.print('N');
379  break;
380  }
381  break;
382  case 't': // recording loop
383  recording_index = 0;
384  do
385  {
386  command = get_char();
387  if (command == 'u') // stop recording
388  {
389  recording_buffer[recording_index]='\0';
390  recording_index = 0;
391  break;
392  }
393  else // add character to recording buffer
394  {
395  if (recording_index < RECORDING_SIZE) recording_buffer[recording_index++]=command;
396  }
397  }
398  while (1);
399  break;
400  case 'T': // transceive byte
401  tx_data = read_hex();
402  if (serial_mode == spi_mode)
403  {
404  rx_data = spi_read(tx_data);
405  byte_to_hex(rx_data);
406  Serial.print(byte_to_hex_buffer);
407  }
408  break;
409  case 'v': // echo recording loop
410  Serial.print(recording_buffer);
411  break;
412  case 'w':
413  recording_mode = playback;
414  break;
415  case 'x':
416  output_low(QUIKEVAL_CS);
417  break;
418  case 'X':
419  output_high(QUIKEVAL_CS);
420  break;
421  case 'Z': // line feed
422  Serial.print('\n');
423  Serial.flush();
424  break;
425  case 0x80: // Reset
426  if (pseudo_reset == 0)
427  {
428  delay(500); // The delay is needed for older GUI's
429  Serial.print("hello\n");
430  pseudo_reset = 1;
431  }
432  break;
433  }
