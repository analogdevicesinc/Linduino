73 int8_t LTC2666_write(uint8_t cs, uint8_t dac_command, uint8_t dac_address, uint16_t dac_code)
74 // Write the 16-bit dac_code to the LTC2666
75 {
76  static uint8_t last_data_array[4];
77  uint8_t data_array[4], rx_array[4];
78  int8_t ret;
79  LT_union_int16_2bytes data;
80 
81  data.LT_int16 = dac_code; // Copy DAC code to union
82  data_array[3] = 0; // Only required for 32 byte readback transaction
83  data_array[2] = dac_command | dac_address; // Build command / address byte
84  data_array[1] = data.LT_byte[1]; // MS Byte
85  data_array[0] = data.LT_byte[0]; // LS Byte
86 
87  spi_transfer_block(cs, data_array, rx_array, (uint8_t) 4);
88 
89  // Compare data read back to data that was sent the previous time this function was called
90  if ((rx_array[2] == last_data_array[2]) && (rx_array[1] == last_data_array[1]) && (rx_array[0] == last_data_array[0]))
91  {
92  ret = 0;
93  }
94  else
95  {
96  ret = 1;
97  }
98 
99  last_data_array[0] = data_array[0]; // Copy data array to a static array to compare
100  last_data_array[1] = data_array[1]; // the next time the function is called
101  last_data_array[2] = data_array[2];
102 
103  return(ret);
104 }

106 uint16_t LTC2666_voltage_to_code(float dac_voltage, float min_output, float max_output)
107 // Calculate a LTC2666 DAC code given the desired output voltage and the minimum / maximum
108 // outputs for a given softspan range.
109 {
110  uint16_t dac_code;
111  float float_code;
112  float_code = 65535.0 * (dac_voltage - min_output) / (max_output - min_output); // Calculate the DAC code
113  float_code = (float_code > (floor(float_code) + 0.5)) ? ceil(float_code) : floor(float_code); // Round
114  if (float_code < 0.0) float_code = 0.0;
115  if (float_code > 65535.0) float_code = 65535.0;
116  dac_code = (uint16_t) (float_code); // Convert to unsigned integer
117  return (dac_code);
118 }
