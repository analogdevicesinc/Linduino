1 /*!
2 Linear Technology DC1703A Demonstration Board.
3 LTC2655: Quad I2C 16-/12-Bit Rail-to-Rail DACs with 10ppm/C Max Reference.
4 
5 @verbatim
6 NOTES
7  Setup:
8  Set the terminal baud rate to 115200 and select the newline terminator.
9  The program displays calculated voltages which are based on the voltage
10  of the reference used, be it internal or external. A precision voltmeter
11  is needed to verify the actual measured voltages against the calculated
12  voltage displayed. If an external reference is used, a precision voltage
13  source is required to apply the external reference voltage. A precision
14  voltmeter is also required to measure the external reference voltage.
15  No external power supply is required. Any assembly option may be used:
16  DC1703A-A, DC1703A-B, DC1703A-C, DC1703A-D.
17 
18 
19  Explanation of Commands:
20  1- Select DAC: Select one of four DACs to test: A, B, C, D.
21 
22  2- Write to DAC input register: Value is stored in the DAC for updating
23  later, allowing multiple channels to be updated at once, either
24  through a software "Update All" command or by asserting the LDAC# pin.
25  User will be prompted to enter either a code in hex or decimal, or a
26  voltage. If a voltage is entered, a code will be calculated based on
27  the active scaling and reference parameters - ideal values if no
28  calibration was ever stored.
29 
30  3- Write and Update: Similar to item 1, but DAC is updated immediately.
31 
32  4- Update DAC: Copies the value from the input register into the DAC
33  Register. Note that a "write and update" command writes the code to
34  BOTH the input register and DAC register, so subsequent "update" commands
35  will simply re-copy the same data (no change in output).
36 
37  5- Power Down DAC: Disable DAC output. Power supply current is reduced. DAC
38  code present in DAC registers at time of shutdown are preserved.
39 
40  6- Set reference mode, either internal or external: Selecting external mode
41  prompts for the external refernce voltage, which is used directly if no
42  individual DAC calibration is stored. The selection and entered volgage
43  are stored to EEPROM so it is persisent across reset / power cycles.
44 
45  7- Calibrate DAC: Use a precision voltmeter to obtain and enter VOUT readings
46  taken with different DAC codes. Set reference mode FIRST, as values are stored
47  separately for internal and external reference mode. Entries are used to
48  calculate the closest code to send to the DAC to achieve an entered voltage.
49 
50  8- Enable / Disable calibration: Switch between stored calibration values and
51  defaults. Calibration parameters are stored separately for internal and
52  external reference modes. Ideal calibration will be used if the calibration
53  parameter valid key is not set.
54 
55 
56 USER INPUT DATA FORMAT:
57  decimal : 1024
58  hex : 0x400
59  octal : 02000 (leading 0 "zero")
60  binary : B10000000000
61  float : 1024.0

202 //! Repeats Linduino loop
203 void loop()
204 {
205  int8_t ack=0;
206  int16_t user_command;
207  static int16_t selected_dac = 0; // The selected DAC to be updated (0=A, 1=B ... 5=All). Initialized to "A".
208  // The main control loop
209  if (demo_board_connected) // Do nothing if the demo board is not connected
210  {
211  if (Serial.available()) // Check for user input
212  {
213  user_command = read_int(); // Read the user command
214  Serial.println(user_command);
215  Serial.flush();
216  ack = 0;
217  switch (user_command)
218  {
219  case 1:
220  ack |= menu_1_select_dac(&selected_dac);
221  break;
222  case 2:
223  ack |= menu_2_write_to_input_register(selected_dac);
224  break;
225  case 3:
226  ack |= menu_3_write_and_update_dac(selected_dac);
227  break;
228  case 4:
229  ack |= menu_4_update_power_up_dac(selected_dac);
230  break;
231  case 5:
232  ack |= menu_5_power_down_dac(selected_dac);
233  break;
234  case 6:
235  ack |= menu_6_set_reference_mode(); // int, ext, if ext, prompt for voltage
236  ack |= restore_calibration();
237  break;
238  case 7:
239  ack |= menu_7_calibrate_dacs();
240  restore_calibration();
241  break;
242  case 8:
243  menu_8_enable_calibration();
244  restore_calibration();
245  break;
246  default:
247  Serial.println("Incorrect Option");
248  break;
249  }
250  if (ack) Serial.println("I2C NACK received, check address\n");
251  Serial.println("\n*****************************************************************");
252  print_prompt(selected_dac);
253  }
254  }
255 }

395 //! Read stored calibration parameters from nonvolatile EEPROM on demo board
396 //! @return Return 1 if successful, 0 if not
397 int8_t restore_calibration()
398 // Read the DAC calibration from EEPROM
399 // Return 1 if successful, 0 if not
400 {
401  int8_t ack=0;
402  int16_t intvalid, extvalid;
403  uint8_t i, eeaddr;
404  float dac_count; // The number of codes, 4096 for 12 bits, 65536 for 16 bits
405 
406  Serial.println(F("\n\nReading Calibration parameters from EEPROM..."));
407  float full_scale; // To avoid confusion - in internal ref mode, FS=Vref, in ext mode, FS=2*Vref
408  // Read the cal keys from the EEPROM.
409  eeprom_read_int16(EEPROM_I2C_ADDRESS, &intvalid, INT_CAL_VALID_BASE);
410  eeprom_read_int16(EEPROM_I2C_ADDRESS, &extvalid, EXT_CAL_VALID_BASE);
411  // Read the stored reference state
412  eeprom_read_byte(EEPROM_I2C_ADDRESS, (char *) &reference_mode, STORED_REF_STATE_BASE);
413  // Read external ref V unconditionally, overwrite with defaults if no cal found
414  eeprom_read_float(EEPROM_I2C_ADDRESS, &reference_voltage, EXT_REF_V_BASE);
415 
416  if (reference_mode == REF_EXTERNAL)
417  {
418  Serial.println(F("Restored external ref. Voltage:"));
419  Serial.println(reference_voltage, 5);
420  }
421  else // EITHER reference is set to internal, OR not programmed in which case default to internal
422  {
423  reference_mode = REF_INTERNAL; // Redundant if already set
424  Serial.println("Internal reference mode set");
425  }
426 
427  // Write the reference mode to the DAC right away
428  ack |= LTC2655_write(LTC2655_I2C_ADDRESS, reference_mode, 0x0F, 0x0000);
429 
430  // Set up default values, shift count, DAC count
431  // Calibration parameters MAY be changed next, if match
432  // between reference mode and stored calibration
433  full_scale = reference_voltage * 2.0; // If external ref mode, this applies.
434  switch (demo_board.option)
435  {
436  case 'A':
437  // LTC2655CUF-L16, 16-bits, 2.5V full scale
438  shift_count = 0;
439  if (reference_mode == REF_INTERNAL) full_scale = 2.5;
440  dac_count = 65536;
441  break;
442  case 'B':
443  // LTC2655CUF-H16, 16-bits, 4.096V full scale
444  shift_count = 0;
445  if (reference_mode == REF_INTERNAL) full_scale = 4.096;
446  dac_count = 65536;
447  break;
448  case 'C':
449  // LTC2655CUF-L12, 12-bits, 2.5V full scale
450  shift_count = 4;
451  if (reference_mode == REF_INTERNAL) full_scale = 2.5;
452  dac_count = 4096;
453  break;
454  case 'D':
455  // LTC2655CUF-H12, 12-bits, 4.096V full scale
456  shift_count = 4;
457  if (reference_mode == REF_INTERNAL) full_scale = 4.096;
458  dac_count = 4096;
459  break;
460  }
461 
462  for (i = 0; i <= 4; i++)
463  {
464  LTC2655_offset[i] = 0;
465  LTC2655_lsb[i] = full_scale / dac_count;
466  }
467 
468  // Restore calibration IF reference mode matches stored calibraiton
469  eeaddr = 0; // Assume no calibration present or mismatch between cal and reference mode
470 
471  if ((intvalid == EEPROM_CAL_KEY) && (reference_mode == REF_INTERNAL))
472  {
473  eeaddr = INT_CAL_PARAMS_BASE;
474  Serial.println(F("Found internal calibration, restoring...)"));
475  }
476  else if ((extvalid == EEPROM_CAL_KEY) && (reference_mode == REF_EXTERNAL))
477  {
478  eeaddr = EXT_CAL_PARAMS_BASE;
479  Serial.println(F("Found external calibration, restoring...)"));
480  }
481  else Serial.println(F("Calibration not found for this\nreference mode, using ideal calibration"));
482 
483  if (eeaddr != 0) // If cal key was enabled and reference mode is correct, read offset and lsb
484  {
485  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2655_offset[0], eeaddr);
486  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2655_offset[1], eeaddr + 2);
487  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2655_offset[2], eeaddr + 4);
488  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2655_offset[3], eeaddr + 6);
489  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2655_lsb[0], eeaddr + 8);
490  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2655_lsb[1], eeaddr + 12);
491  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2655_lsb[2], eeaddr + 16);
492  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2655_lsb[3], eeaddr + 20);
493  LTC2655_offset[4] = LTC2655_offset[0]; // Copy cal value for DAC A to cal value for
494  LTC2655_lsb[4] = LTC2655_lsb[0]; // DAC ALL
495  Serial.println("Calibration Restored");
496  }
497  for (i=0; i<=4; ++i)
498  {
499  Serial.print("DAC ");
500  Serial.print((char) ('A' + i));
501  Serial.print(" offset: ");
502  Serial.print(LTC2655_offset[i]);
503  Serial.print(" , lsb: ");
504  Serial.print(LTC2655_lsb[i]*1000, 4);
505  Serial.println(" mv");
506  }
507  Serial.println("(DAC E applies to ALL DACs selections)");
508  if (eeaddr != 0) return (1);
509  return (ack);
510 }

512 //! Store measured calibration parameters to nonvolatile EEPROM on demo board
513 void store_calibration()
514 // Store the DAC calibration to the EEPROM
515 {
516  uint8_t eeaddr;
517  if (reference_mode == REF_INTERNAL)
518  {
519  eeprom_write_int16(EEPROM_I2C_ADDRESS, EEPROM_CAL_KEY, INT_CAL_VALID_BASE);
520  eeaddr = INT_CAL_PARAMS_BASE;
521  }
522  else
523  {
524  eeprom_write_int16(EEPROM_I2C_ADDRESS, EEPROM_CAL_KEY, EXT_CAL_VALID_BASE);
525  eeaddr = EXT_CAL_PARAMS_BASE;
526  }
527 
528  eeprom_write_int16(EEPROM_I2C_ADDRESS, LTC2655_offset[0], eeaddr); // Offset
529  eeprom_write_int16(EEPROM_I2C_ADDRESS, LTC2655_offset[1], eeaddr + 2);
530  eeprom_write_int16(EEPROM_I2C_ADDRESS, LTC2655_offset[2], eeaddr + 4);
531  eeprom_write_int16(EEPROM_I2C_ADDRESS, LTC2655_offset[3], eeaddr + 6);
532  eeprom_write_float(EEPROM_I2C_ADDRESS, LTC2655_lsb[0], eeaddr + 8); // lsb
533  eeprom_write_float(EEPROM_I2C_ADDRESS, LTC2655_lsb[1], eeaddr + 12);
534  eeprom_write_float(EEPROM_I2C_ADDRESS, LTC2655_lsb[2], eeaddr + 16);
535  eeprom_write_float(EEPROM_I2C_ADDRESS, LTC2655_lsb[3], eeaddr + 20);
536  Serial.println(F("Calibration Stored to EEPROM"));
537 }

636 //! Calibrate the selected DAC using a voltmeter. The routine
637 //! does a linear curve fit given two data points.
638 //! @return ACK bit (0=acknowledge, 1=no acknowledge)
639 int8_t calibrate_dac(uint8_t index)
640 {
641  int8_t ack=0;
642  uint16_t code1 = 0x0200; //! Calibration code 1
643  uint16_t code2 = 0xFFFF; //! Calibration code 2
644  float voltage1; //! Calibration voltage 1
645  float voltage2; //! Calibration voltage 2
646  Serial.println("");
647  Serial.print("Calibrating DAC ");
648  Serial.println((char) (0x41 + index));
649  // Left align 12-bit code1 to 16 bits & write to DAC
650  ack |= LTC2655_write(LTC2655_I2C_ADDRESS, LTC2655_CMD_WRITE_UPDATE, index, code1 << shift_count);
651  Serial.print("DAC code set to 0x");
652  Serial.println(code1, HEX);
653  Serial.print("Enter measured DAC voltage:");
654  voltage1 = read_float();
655  Serial.print(voltage1, 6);
656  Serial.println(" V");
657  // Left align 12-bit code2 to 16 bits & write to DAC
658  ack |= LTC2655_write(LTC2655_I2C_ADDRESS, LTC2655_CMD_WRITE_UPDATE, index, code2 << shift_count);
659  Serial.print("DAC code set to 0x");
660  Serial.println(code2, HEX);
661  Serial.print("Enter measured DAC voltage:");
662  voltage2 = read_float();
663  Serial.print(voltage2, 6);
664  Serial.println(" V");
665  LTC2655_calibrate(code1, code2, voltage1, voltage2, &LTC2655_lsb[index], &LTC2655_offset[index]);
666  return(ack);
667 }
