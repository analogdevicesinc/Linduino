282  }

286  }

289  digitalWrite(5, LOW);
290  }

293 //! Reads the LTC2400 and displays the results
294 void test_LTC2400(float vref, uint16_t eoc_timeout)
295 {
296  float voltage = 0.0;
297  int32_t adc_code = 0;
298  int i;
299 
300  quikeval_SPI_connect(); // Connect SPI to main data port
301  set_CS_decoder(LTC2400_CS_ADDR); // Set the decoder for the LTC2400 chip select
302 
303  Serial.println(F("\nTesting LTC2400 with a 32 bit read using LTC24XX_SPI_32bit_data..."));
304 
305  LTC24XX_EOC_timeout(CS, eoc_timeout); // First flush power-on conversion that probably happened before
306  LTC24XX_SPI_32bit_data(CS, &adc_code); // the DAC output was set
307 
308  LTC24XX_EOC_timeout(CS, eoc_timeout); // Wait for end of conversion
309  LTC24XX_SPI_32bit_data(CS, &adc_code); // Read the ADC
310  voltage = LTC24XX_SE_code_to_voltage(adc_code, vref); // Convert ADC code to voltage
311 
312  // Display results
313  Serial.print(F(" Formatted LTC2400 ADC code: 0x"));
314  Serial.println(adc_code, HEX);
315  Serial.print(F(" Voltage:"));
316  Serial.print(voltage, 8);
317  Serial.println(F(" V"));
318 }
