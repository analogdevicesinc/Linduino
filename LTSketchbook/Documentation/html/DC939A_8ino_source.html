2 /*!
3 Linear Technology DC939A Demonstration Board.
4 LTC2484: 24-Bit Delta Sigma ADC with Easy Drive Input Current Cancellation.
5 
6 @verbatim
7 
8 NOTES
9  Setup:
10  Set the terminal baud rate to 115200 and select the newline terminator. Equipment
11  required is a precision voltage source and a precision voltmeter. No external
12  power supply is required. Ensure JP1 is in the +5V position.
13 
14  To Read data:
15  The voltage source should be connected with positive lead to IN+ and negative
16  lead to IN-. The voltage source negative output must also be connected to the GND
17  pin in order to provide a ground-referenced voltage. Ensure voltage is within
18  analog input voltage range -0.3V to +2.5V. Swapping input voltages results in a
19  reversed polarity reading.
20 
21  How to calibrate:
22  Short the inputs to ground to calibrate the offset. Next, hit ENTER (this takes
23  the reading). Now apply approximately 2.49 volts to +IN, with -IN connected to
24  ground. Measure this voltage with a precise voltmeter and enter this value.
25  Calibration is now stored in EEPROM. Upon startup the calibration values will be
26  restored.
27 
28  Explanation of Commands:
29 
30  **** MAIN MENU ****
31 
32  0- Read- By entering this a voltage at the +IN, -IN terminals will be read.
33  1- Set Rejection- Select this to access the Filter rejection menu. Follow
34  command cues to enable desired rejection profile of on-chip digital filters.
35  2- Select 2X Rate- This selection allows disabling of the autocalibration
36  feature to achieve a benefit of twice the output rate.
37  3- Calibrate Voltage- Follow the calibration cues to calibrate the device
38  voltage.
39  4- Calibrate Temperature- Select this and follow the cues if it is desired to
40  calibrate the integrated
41  temperature sensor.
42 
43 USER INPUT DATA FORMAT:
44  decimal : 1024
45  hex : 0x400
46  octal : 02000 (leading 0 "zero")
47  binary : B10000000000
48  float : 1024.0
49 
50 @endverbatim

160 //! Repeats Linduino loop
161 void loop()
162 {
163  uint8_t user_command; // The user input command
164  uint8_t acknowledge = 0;
165  if (demo_board_connected)
166  {
167  if (Serial.available()) // Check for user input
168  {
169  user_command = read_int(); // Read the user command
170  if (user_command == 'm');
171  else
172  Serial.println(user_command);
173  delay(50); // Allow the print to finish
174  switch (user_command)
175  {
176  case 0:
177  acknowledge |= menu_0_read();
178  break;
179  case 1:
180  menu_1_set_rejection();
181  break;
182  case 2:
183  menu_2_set_1X_2X();
184  break;
185  case 3:
186  acknowledge |= menu_3_calibrate_voltage();
187  break;
188  case 4:
189  acknowledge |= menu_4_calibrate_temperature();
190  break;
191  default:
192  Serial.println("Incorrect Option");
193  break;
194  }
195  if (acknowledge)
196  Serial.println(F("***** SPI ERROR *****"));
197  Serial.println(F("*****************************************************************"));
198  print_prompt();
199  }
200  }
201 }

261 //! Read stored calibration parameters from nonvolatile EEPROM on demo board
262 //! @return 0 if successful, 1 if failure
263 int8_t restore_calibration()
264 // Read the DAC calibration from EEPROM
265 {
266  int16_t cal_key;
267  // Read the cal key from the EEPROM
268  eeprom_read_int16(EEPROM_I2C_ADDRESS, &cal_key, EEPROM_CAL_STATUS_ADDRESS);
269  if (cal_key == EEPROM_CAL_KEY)
270  {
271  // Calibration has been stored, read offset and lsb
272  eeprom_read_int32(EEPROM_I2C_ADDRESS, &LTC2484_offset_code, EEPROM_CAL_STATUS_ADDRESS + 2); // Offset
273  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2484_lsb, EEPROM_CAL_STATUS_ADDRESS + 6); // LSB
274  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2484_r0, EEPROM_CAL_STATUS_ADDRESS + 10); // Temp r0
275  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2484_t0, EEPROM_CAL_STATUS_ADDRESS + 14); // Temp t0
276  Serial.println(F("Calibration Restored"));
277  return (1);
278  }
279  else
280  {
281  Serial.println(F("Calibration not found"));
282  return (0);
283  }
284 }

298 //! Construct ADC command from rejection, input, and 2X parameters
299 //! @return ADC command
300 uint8_t build_adc_command(uint8_t temperature)
301 // Build the ADC command byte
302 {
303  uint8_t adc_command = 0; // The LTC2484 command byte
304  adc_command = LTC2484_ENABLE; // Set ENABLE
305  switch (adc_rejection)
306  {
307  case REJECTION50_60: // Set REJECTION
308  adc_command |= LTC2484_REJECTION_50HZ_60HZ;
309  break;
310  case REJECTION50:
311  adc_command |= LTC2484_REJECTION_50HZ;
312  break;
313  case REJECTION60:
314  adc_command |= LTC2484_REJECTION_60HZ;
315  break;
316  }
317  if (temperature) // Set Temperature Input
318  {
319  adc_command |= LTC2484_TEMPERATURE_INPUT;
320  adc_command |= LTC2484_AUTO_CALIBRATION;
321  }
322  else
323  {
324  adc_command |= LTC2484_EXTERNAL_INPUT; // Set Voltage Input
325  if (adc_2x)
326  adc_command |= LTC2484_SPEED_2X; // Set 2X Speed
327  else
328  adc_command |= LTC2484_AUTO_CALIBRATION;
329  }
330  return(adc_command);
331 }

333 //! Read ADC
334 //! @return 0 if successful, 1 if failure
335 uint8_t menu_0_read()
336 {
337  // Read values
338  int32_t adc_code = 0; // The LTC2484 code
339  float adc_voltage = 0.0; // The LTC2484 input voltage
340  float adc_temperature; // The LTC2484 temperature sensor value
341  uint8_t adc_command; // The LTC2484 command byte
342 
343  adc_command = build_adc_command(0); // Build ADC command byte for voltage input
344  if (LTC2484_EOC_timeout(LTC2484_CS, MISO_TIMEOUT)) // Check for EOC
345  return(1);
346  LTC2484_read(LTC2484_CS, adc_command, &adc_code); // Throw away last reading
347  adc_command = build_adc_command(1); // Build ADC command byte for temperature input
348  if (LTC2484_EOC_timeout(LTC2484_CS, MISO_TIMEOUT)) // Check for EOC
349  return(1);
350  LTC2484_read(LTC2484_CS, adc_command, &adc_code); // Read voltage
351  Serial.println(F("\nVoltage Measurement"));
352  Serial.print(F(" Received Code: 0x"));
353  Serial.println(adc_code, HEX);
354  Serial.print(F(" Voltage:"));
355  adc_voltage = LTC2484_code_to_voltage(adc_code, LTC2484_lsb, LTC2484_offset_code);
356  Serial.print(adc_voltage, 6);
357  Serial.println(F("V "));
358  adc_command = build_adc_command(1); // Build ADC command byte for voltage input
359  if (LTC2484_EOC_timeout(LTC2484_CS, MISO_TIMEOUT)) // Check for EOC
360  return(1);
361  LTC2484_read(LTC2484_CS, adc_command, &adc_code); // Read temperature
362  Serial.println(F("\nTemperature Measurement"));
363  Serial.print(F(" Received Code: 0x"));
364  Serial.println(adc_code, HEX);
365  Serial.print(F(" Sensor Voltage:"));
366  adc_voltage = LTC2484_code_to_voltage(adc_code, LTC2484_lsb, LTC2484_offset_code);
367  Serial.print(adc_voltage, 6);
368  Serial.println(F("V"));
369  Serial.print(F(" Temperature:"));
370  adc_temperature = LTC2484_temperature(adc_code, LTC2484_t0, LTC2484_r0);
371  Serial.print(adc_temperature, 1);
372  Serial.println(F("C\n"));
373  return(0);
374 }
