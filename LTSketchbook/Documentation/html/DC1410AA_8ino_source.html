5 @verbatim
6 
7 NOTES
8  Setup:
9  Set the terminal baud rate to 115200 and select the newline terminator. Equipment
10  required is a precision voltage source and a precision voltmeter. Additionally,
11  an external power supply is required to provide a negative voltage for Amp V-.
12  Set it to anywhere from -1V to -5V. Set Amp V+ to Vcc. Ensure the COM and REF-
13  pins are connected to ground. The REF+ pin should be connected to +5V.
14 
15  How Measure a Load Cell:
16  The LTC2498 allowas a wide common mode range of 0V to Vcc. The LT1678 OPAMP common
17  mode range is V- + 1.5 to V+ - 0.8V. Ensure the load cell sensor meets the requirments
18  of the ADC and amplifier. After, set the desired gain and read in differential mode.

179  }
180  if (ack_EOC)
181  Serial.println(F("\n******SPI ERROR******\n"));
182  Serial.print(F("\n*************************\n"));

184  }

213 //! Read channels in differential mode
214 //! @return Returns 0=successful, 1=unsuccessful (exceeded timeout)
215 uint8_t menu_1_read_differential()
216 {
217  int8_t y; // Offset into differential channel array to select polarity
218  uint8_t adc_command_high; // The LTC2498 command high byte
219  uint8_t adc_command_low; // The LTC2498 command low byte
220  int16_t user_command; // The user input command
221  int32_t adc_code[12]; // The LTC2498 code
222  float adc_voltage; // The LTC2498 voltage
223  static float offset = 0.0;
224  static float gain_unity = 0.0;
225  static float gain_8 = 0.0;
226  static float gain_16 = 0.0;
227  static float gain_32 = 0.0;
228  static float gain_value[4];
229 
230  adc_command_low = rejection_mode | two_x_mode;
231  do
232  {
233  for (int8_t i = 0; i <= 11; i++)
234  {
235  set_gain(A_PIN, B_PIN, read_gain_seq[i]); // Set the MUX to the desired gain
236  adc_command_high = read_command_seq[i]; // Set channel
237  // Ensure a conversions are complete
238  if (LTC2498_EOC_timeout(LTC2498_CS, eoc_timeout))
239  return 1;
240  // Read the LTC2498
241  LTC2498_read(LTC2498_CS, adc_command_high, adc_command_low, adc_code + i);
242  }
243  if (filt_flag)
244  {
245  // Reset the filter
246  filt_flag = 0;
247  offset = adc_code[1] - 536870912;
248  gain_unity = ((float)adc_code[2]) - 536870912;
249  gain_8 = ((float)adc_code[3] - 536870912) / gain_unity;
250  gain_16 = ((float)adc_code[4] - 536870912) / gain_unity;
251  gain_32 = ((float)adc_code[5] - 536870912) / gain_unity;
252  }
253  else
254  {
255  // IIR exponential filter
256  offset = (1.0 - smoothing_const) * ((float)adc_code[1] - 536870912) + smoothing_const * offset;
257 
258  if (two_x_mode)
259  {
260  gain_unity = ((float)adc_code[2]) - offset - 536870912;
261  gain_value[1] = ((float)adc_code[3] - offset - 536870912) / gain_unity;
262  gain_16 = ((float)adc_code[4] - offset - 536870912) / gain_unity;
263  gain_32 = ((float)adc_code[5] - offset - 536870912) / gain_unity;
264  }
265  else
266  {
267  gain_unity = ((float)adc_code[2]) - 536870912;
268  gain_8 = (1.0 - smoothing_const) * (((float)adc_code[3] - 536870912) / gain_unity) + smoothing_const * gain_8;
269  gain_16 = (1.0 - smoothing_const) * (((float)adc_code[4] - 536870912) / gain_unity) + smoothing_const * gain_16;
270  gain_32 = (1.0 - smoothing_const) * (((float)adc_code[5] - 536870912) / gain_unity) + smoothing_const * gain_32;
271  }
272  }
273 
274  gain_value[0] = 1; // Assume ideal unity gain
275  gain_value[1] = gain_8;
276  gain_value[2] = gain_16;
277  gain_value[3] = gain_32;
278 
279  Serial.println(F("\n**********"));
280  Serial.print(F(" Offset: "));
281  Serial.print(offset, 6);
282  Serial.print(F(" "));
283  Serial.print(LTC2498_code_to_voltage((int32_t)offset + 536870912, LTC2498_vref) * 1000, 4);
284  Serial.println(F(" mV"));
285  Serial.print(F(" G8: "));
286  Serial.println(gain_8, 6);
287  Serial.print(F(" G16: "));
288  Serial.println(gain_16, 6);
289  Serial.print(F(" G32: "));
290  Serial.println(gain_32, 6);
291  Serial.println();
292 
293  Serial.println(F("CH ADC Code Gain Comp Voltage (mV)"));
294  Serial.print(F("P0-N1: 0x"));
295  Serial.print(adc_code[6], HEX);
296  Serial.print(" ");
297  Serial.print(gain_value[channel_gain[0]], 6);
298  Serial.print(F(" "));
299  Serial.println(1000 * LTC24XX_diff_code_to_calibrated_voltage(adc_code[6], LTC2498_vref / 536870911, offset) / gain_value[channel_gain[0]], 6);
300 
301  Serial.print(F("P2-N3: 0x"));
302  Serial.print(adc_code[7], HEX);
303  Serial.print(F(" "));
304  Serial.print(gain_value[channel_gain[1]], 6);
305  Serial.print(" ");
306  Serial.println(1000 * LTC24XX_diff_code_to_calibrated_voltage(adc_code[7], LTC2498_vref / 536870911, offset) / gain_value[channel_gain[1]], 6);
307 
308  Serial.print(F("P4-N5: 0x"));
309  Serial.print(adc_code[8], HEX);
310  Serial.print(F(" "));
311  Serial.print(gain_value[channel_gain[2]], 6);
312  Serial.print(" ");
313  Serial.println(1000 * LTC24XX_diff_code_to_calibrated_voltage(adc_code[8], LTC2498_vref / 536870911, offset) / gain_value[channel_gain[2]], 6);
314 
315  Serial.print(F("P6-N7: 0x"));
316  Serial.print(adc_code[9], HEX);
317  Serial.print(F(" "));
318  Serial.print(gain_value[channel_gain[3]], 6);
319  Serial.print(" ");
320  Serial.println(1000 * LTC24XX_diff_code_to_calibrated_voltage(adc_code[9], LTC2498_vref / 536870911, offset) / gain_value[channel_gain[3]], 6);
321 
322  Serial.print(F("P8-N9: 0x"));
323  Serial.print(adc_code[10], HEX);
324  Serial.print(F(" "));
325  Serial.print(gain_value[channel_gain[4]], 6);
326  Serial.print(" ");
327  Serial.println(1000 * LTC24XX_diff_code_to_calibrated_voltage(adc_code[10], LTC2498_vref / 536870911, offset) / gain_value[channel_gain[4]], 6);
328 
329  Serial.print(F("P10-N11: 0x"));
330  Serial.print(adc_code[11], HEX);
331  Serial.print(F(" "));
332  Serial.print(gain_value[channel_gain[5]], 6);
333  Serial.print(" ");
334  Serial.println(1000 * LTC24XX_diff_code_to_calibrated_voltage(adc_code[11], LTC2498_vref / 536870911, offset) / gain_value[channel_gain[5]], 6);
335 
336  Serial.println();
337  Serial.println(F("Enter any character to exit"));
338 
339  }
340  while (Serial.available() == 0);
341  while (Serial.available())
342  Serial.read();
343  Serial.flush();
344  return 0;
345 }
