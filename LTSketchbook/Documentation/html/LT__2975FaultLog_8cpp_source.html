86  * Read LTC2975 fault log
87  *
88  * address: PMBUS address
89  */
90 void
91 LT_2975FaultLog::read(uint8_t address)
92 {
93  // Copy to RAM
94  pmbus_->smbus()->sendByte(address, MFR_FAULT_LOG_RESTORE);
95  // Monitor BUSY bit
96  while ((pmbus_->smbus()->readByte(address, MFR_COMMON) & (1 << 6)) == 0);
97 
98  uint16_t size = sizeof(struct LT_2975FaultLog::FaultLogLtc2975);
99  uint8_t *data = (uint8_t *) malloc(size);
100  if (data == 0)
101  Serial.print(F("bad malloc."));
102 #ifdef RAW_EEPROM
103  getNvmBlock(address, 0, 128, 0xC8, data);
104 #else
105  // Read block data with log
106  pmbus_->smbus()->readBlock(address, MFR_FAULT_LOG, data, 255);
107 #endif
108 
109  struct LT_2975FaultLog::FaultLogLtc2975 *log = (struct LT_2975FaultLog::FaultLogLtc2975 *)data;
110 
111  log->firstValidByte = 79; // first byte after preamble
112 
113  log->lastValidByte = 238; // last byte of loop data
114 
115  uint8_t cycle_start = log->preamble.position_last;
116 
117  //57 corresponds to datasheet p77, last line 2nd column
118  log->loops = (LT_2975FaultLog::FaultLogReadLoopLtc2975 *) (log->telemetryData - 57 + cycle_start);
119 
120  faultLog2975 = log;
121 }
