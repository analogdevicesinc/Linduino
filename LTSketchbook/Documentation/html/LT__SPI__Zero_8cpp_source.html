69 // Reads and sends a word
70 // Return 0 if successful, 1 if failed
71 void spi_transfer_word(uint8_t cs_pin, uint16_t tx, uint16_t *rx)
72 {
73  union
74  {
75  uint8_t b[2];
76  uint16_t w;
77  } data_tx;
78 
79  union
80  {
81  uint8_t b[2];
82  uint16_t w;
83  } data_rx;
84 
85  data_tx.w = tx;
86 
87  output_low(cs_pin); //! 1) Pull CS low
88 
89  data_rx.b[1] = SPI.transfer(data_tx.b[1]); //! 2) Read MSB and send MSB
90  data_rx.b[0] = SPI.transfer(data_tx.b[0]); //! 3) Read LSB and send LSB
91 
92  *rx = data_rx.w;
93 
94  output_high(cs_pin); //! 4) Pull CS high
95 }

97 // Reads and sends a byte array
98 void spi_transfer_block(uint8_t cs_pin, uint8_t *tx, uint8_t *rx, uint8_t length)
99 {
100  int8_t i;
101 
102  output_low(cs_pin); //! 1) Pull CS low
103 
104  for (i=(length-1); i >= 0; i--)
105  rx[i] = SPI.transfer(tx[i]); //! 2) Read and send byte array
106 
107  output_high(cs_pin); //! 3) Pull CS high
108 }
