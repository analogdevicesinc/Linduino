141 //! Repeats Linduino Loop
142 void loop()
143 {
144  int8_t user_command; // The user input command
145  uint8_t acknowledge = 0;
146  if (Serial.available()) // Check for user input
147  {
148  user_command = read_int(); // Read the user command
149  if (user_command != 'm')
150  Serial.println(user_command); // Prints the user command to com port
151  Serial.flush();
152  switch (user_command)
153  {
154  case 1:
155  menu1_display_adc_output();
156  break;
157  case 2:
158  menu2_change_softspan_range();
159  break;
160  default:
161  Serial.println(F("Incorrect Option"));
162  }
163  if (acknowledge)
164  Serial.println(F("***** I2C ERROR *****"));
165  Serial.print(F("\n****************************** Press Enter to Continue ******************************\n"));
166  read_int();
167  print_prompt();
168  }
169 }

171 //! Configure sneaker port to talk to the ADC
172 void sneaker_port_init()
173 {
174  uint8_t value = 0;
175  i2c_read_byte(SNEAKER_PORT_I2C_ADDRESS, &value);
176  delay(100);
177  value = value & 0x7F; // P7 = WRIN = 0
178  value = value | 0x04; // P2 = WRIN2 = 1
179  i2c_write_byte(SNEAKER_PORT_I2C_ADDRESS, value);
180  delay(100);
181 }

256 void decode_values(uint32_t channel_data)
257 {
258  uint8_t channel, config_number;
259  float voltage;
260  uint32_t code;
261 
262  channel = (channel_data & CHANNEL_NUMBER) >> 3;
263  config_number = channel_data & SOFTSPAN;
264  code = (channel_data & 0xFFFFC0) >> 6;
265  voltage = LTC23XX_voltage_calculator(code, config_number);
266 
267  Serial.print(F("\nChannel : "));
268  Serial.print(channel);
269  Serial.print(F("\t\tData : 0x"));
270  Serial.println(code, HEX);
271  Serial.print(F("Voltage : "));
272  Serial.print(voltage, 6);
273  Serial.print(F(" V"));
274  Serial.print(F("\tConfig Number: "));
275  Serial.println(config_number);
276 }

278 //! Displays the ADC output and calculated voltage for all channels
279 void menu1_display_adc_output()
280 {
281  uint8_t display_channels;
282  uint8_t Result[24];
283  uint8_t pos;
284  data.LT_uint32 = 0;
285 
286  Serial.print("\nEnter the number of channels to be displayed: ");
287  display_channels = read_int();
288  if (display_channels < 1)
289  display_channels = 1;
290  else if (display_channels > num_of_channels)
291  display_channels = num_of_channels;
292 
293  while (!Serial.available())
294  {
295  LTC23XX_read(QUIKEVAL_CS, configuration_bits.LT_uint32, Result);
296  Serial.println(F("\n********************************************"));
297 
298  pos = 23;
299  for (int i = 0; i < display_channels; ++i)
300  {
301  data.LT_byte[2] = Result[pos--];
302  data.LT_byte[1] = Result[pos--];
303  data.LT_byte[0] = Result[pos--];
304 
305  decode_values(data.LT_uint32);
306  }
307  }
308 }

327  }
