51 uint16_t httoi(char *value)
52 {
53  struct CHexMap
54  {
55  char chr;
56  int value;
57  };
58 
59  struct CHexMap HexMap[22] =
60  {
61  {'0', 0}, {'1', 1},
62  {'2', 2}, {'3', 3},
63  {'4', 4}, {'5', 5},
64  {'6', 6}, {'7', 7},
65  {'8', 8}, {'9', 9},
66  {'A', 10}, {'B', 11},
67  {'C', 12}, {'D', 13},
68  {'E', 14}, {'F', 15},
69  {'a', 10}, {'b', 11},
70  {'c', 12}, {'d', 13},
71  {'e', 14}, {'f', 15}
72  };
73  char *s = value;
74  uint16_t result = 0;
75  boolean firsttime = true;
76  boolean found;
77  int i;
78  if (*s == '0' && *(s + 1) == 'x') s += 2;
79  while (*s != '\0')
80  {
81  found = false;
82  for (i = 0; i < 22; i++)
83  {
84  if (*s == HexMap[i].chr)
85  {
86  if (!firsttime) result <<= 4;
87  //printf("add: %d\n", HexMap[i].value);
88  result |= HexMap[i].value;
89  found = true;
90  break;
91  }
92  }
93  if (!found) break;
94  s++;
95  firsttime = false;
96  }
97  //printf("finish: 0x%02x\n", result);
98  return result;
99 }
