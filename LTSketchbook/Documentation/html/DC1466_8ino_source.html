1 /*!
2 Linear Technology DC1466 Demonstration Board.
3 LTC2636: Octal 12-/10-/8-Bit SPI VOUT DACs with 10ppm/°C Reference.
4 
5 Linear Technology DC1488 Demonstration Board.
6 LTC2634: Quad 12-/10-/8-Bit SPI VOUT DACs with 10ppm/°C Reference.
7 
8 @verbatim
9 NOTES
10  Setup:
11  Set the terminal baud rate to 115200 and select the newline terminator.
12  The program displays calculated voltages which are based on the voltage
13  of the reference used, be it internal or external. A precision voltmeter
14  is needed to verify the actual measured voltages against the calculated
15  voltage displayed. If an external reference is used, a precision voltage
16  source is required to apply the external reference voltage. A
17  precision voltmeter is also required to measure the external reference
18  voltage. No external power supply is required. Any assembly option
19  may be used: DC1466A-A, DC1466A-B, DC1466A-C, DC1466A-D.
20 
21 
22  Explanation of Commands:
23  1- Select DAC: Select one of four DACs to test : A, B, C, D.
24 
25  2- Write to DAC input register: Value is stored in the DAC for updating
26  later, allowing multiple channels to be updated at once, either
27  through a software "Update All" command or by asserting the LDAC# pin.
28  User will be prompted to enter either a code in hex or decimal, or a
29  voltage. If a voltage is entered, a code will be calculated based on
30  the active scaling and reference parameters - ideal values if no
31  calibration was ever stored.
32 
33  3- Write and Update: Similar to item 1, but DAC is updated immediately.
34 
35  4- Update DAC: Copies the value from the input register into the DAC
36  Register. Note that a "write and update" command writes the code to
37  BOTH the input register and DAC register, so subsequent "update"
38  commands will simply re-copy the same data (no change in output.)
39 
40  5- Power Down DAC: Disable DAC output. Power supply current is reduced.
41  DAC code present in DAC registers at time of shutdown are preserved.
42 
43  6- Set reference mode, either internal or external: Selecting external
44  mode prompts for the external reference voltage, which is used directly
45  if no individual DAC calibration is stored. The selection and entered
46  voltage are stored to EEPROM so it is persistent across reset / power cycles.
47 
48 USER INPUT DATA FORMAT:
49  decimal : 1024
50  hex : 0x400
51  octal : 02000 (leading 0 "zero")
52  binary : B10000000000
53  float : 1024.0
54 
55 @endverbatim

177 //! Repeats Linduino loop
178 void loop()
179 {
180  int16_t user_command;
181  static int16_t selected_dac = 0; // The selected DAC to be updated (0=A, 1=B ... 8=All). Initialized to "A".
182  // The main control loop
183  if (demo_board_connected) // Do nothing if the demo board is not connected
184  {
185  if (Serial.available()) // Check for user input
186  {
187  user_command = read_int(); // Read the user command
188  Serial.println(user_command);
189  Serial.flush();
190  switch (user_command)
191  {
192  case 1:
193  menu_1_select_dac(&selected_dac);
194  break;
195  case 2:
196  menu_2_write_to_input_register(selected_dac);
197  break;
198  case 3:
199  menu_3_write_and_update_dac(selected_dac);
200  break;
201  case 4:
202  menu_4_update_power_up_dac(selected_dac);
203  break;
204  case 5:
205  menu_5_power_down_dac(selected_dac);
206  break;
207  case 6:
208  menu_6_set_reference_mode();
209  break;
210  case 7:
211  menu_7_calibrate_dacs();
212  restore_calibration();
213  break;
214  default:
215  Serial.println("Incorrect Option");
216  break;
217  }
218  Serial.println("\n*****************************************************************");
219  print_prompt(selected_dac);
220  }
221  }
222 }

322 //! Read stored calibration parameters from nonvolatile EEPROM on demo board
323 //! @return void
324 void restore_calibration()
325 {
326  uint8_t i;
327  float dac_count; // The number of codes, 4096 for 12 bits, 65536 for 16 bits
328 
329  if (reference_mode == REF_EXTERNAL)
330  {
331  Serial.println(F("External reference mode set"));
332  Serial.print("Enter external reference voltage: ");
333  reference_voltage = read_float();
334  Serial.print(reference_voltage, 3);
335  Serial.println("V");
336  }
337  else // EITHER reference is set to internal, OR not programmed in which case default to internal
338  {
339  reference_mode = REF_INTERNAL; // Redundant if already set
340  Serial.println("Internal reference mode set");
341  }
342  LTC2636_write(LTC2636_CS, reference_mode, LTC2636_DAC_ALL, 0x0000);
343  // The following two IF statements are used to allow the program to run with
344  // a QuikEval string that does not contain the demo board option.
345  // If the demo board option is found then these values are overwritten.
346  if (strcmp(demo_board.product_name, "LTC2636-LM") == 0)
347  {
348  // LTC2636-LM, 12-bits, 2.5V full scale
349  shift_count = 4;
350  if (reference_mode == REF_INTERNAL)
351  {
352  reference_voltage = 2.5;
353  Serial.println("Internal reference voltage = 2.5 V");
354  }
355  dac_count = 4096;
356  }
357  if (strcmp(demo_board.product_name, "LTC2636-HZ") == 0)
358  {
359  // LTC2636-HZ, 12-bits, 4.096V full scale
360  shift_count = 4;
361  if (reference_mode == REF_INTERNAL)
362  {
363  reference_voltage = 4.096;
364  Serial.println("Internal reference voltage = 4.096 V");
365  }
366  dac_count = 4096;
367  }
368  if (strcmp(demo_board.product_name, "LTC2636-HM") == 0)
369  {
370  // LTC2636-HM, 12-bits, 4.096V full scale
371  shift_count = 4;
372  if (reference_mode == REF_INTERNAL)
373  {
374  reference_voltage = 4.096;
375  Serial.println("Internal reference voltage = 4.096 V");
376  }
377  dac_count = 4096;
378  }
379  if (strcmp(demo_board.product_name, "LTC2636-LZ") == 0)
380  {
381  // LTC2636-LZ, 12-bits, 4.096V full scale
382  shift_count = 4;
383  if (reference_mode == REF_INTERNAL)
384  {
385  reference_voltage = 4.096;
386  Serial.println("Internal reference voltage = 4.096 V");
387  }
388  dac_count = 4096;
389  }
390 
391  for (i = 0; i <= num_of_channels; i++)
392  {
393  LTC2636_offset[i] = 0;
394  LTC2636_lsb[i] = reference_voltage / dac_count;
395  }
396 }

488 //! Calibrate the selected DAC using a voltmeter. The routine
489 //! does a linear curve fit given two data points.
490 //! @return ACK bit (0=acknowledge, 1=no acknowledge)
491 int8_t calibrate_dac(uint8_t index)
492 {
493  int8_t ack=0;
494  uint16_t code1 = 0x0200; //! Calibration code 1
495  uint16_t code2 = 0x0FFF; //! Calibration code 2
496  float voltage1; //! Calibration voltage 1
497  float voltage2; //! Calibration voltage 2
498  Serial.println("");
499  Serial.print("Calibrating DAC ");
500  Serial.println((char) (0x41 + index));
501  // Left align 12-bit code1 to 16 bits & write to DAC
502  LTC2636_write(LTC2636_CS, LTC2636_CMD_WRITE_UPDATE, index, code1 << shift_count);
503  Serial.print("DAC code set to 0x");
504  Serial.println(code1, HEX);
505  Serial.print("Enter measured DAC voltage:");
506  voltage1 = read_float();
507  Serial.print(voltage1, 6);
508  Serial.println(" V");
509  // Left align 12-bit code2 to 16 bits & write to DAC
510  LTC2636_write(LTC2636_CS, LTC2636_CMD_WRITE_UPDATE, index, code2 << shift_count);
511  Serial.print("DAC code set to 0x");
512  Serial.println(code2, HEX);
513  Serial.print("Enter measured DAC voltage:");
514  voltage2 = read_float();
515  Serial.print(voltage2, 6);
516  Serial.println(" V");
517  LTC2636_calibrate(code1, code2, voltage1, voltage2, &LTC2636_lsb[index], &LTC2636_offset[index]);
518  return(ack);
519 }
