73 void LTC5599_write_read(uint8_t cs, uint8_t address, uint8_t rw, uint8_t tx, uint8_t *rx)
74 {
75  address = address << 1;
76  union
77  {
78  uint8_t b[2];
79  uint16_t w;
80  } command, data;
81 
82  if (rw == 0) //write
83  {
84  command.b[1] = address;
85  command.b[0] = tx; //write this command byte
86 
87  spi_transfer_word(cs,command.w,&data.w); //write an address byte and a command byte to the device
88  }
89  else
90  {
91  address = address + 0x01;
92  command.b[1] = address;
93  command.b[0] = 0x00; //read into this data byte
94 
95  spi_transfer_word(cs,command.w,&data.w); //write an address byte to the device and read back a byte
96  *rx = data.b[0];
97  }
