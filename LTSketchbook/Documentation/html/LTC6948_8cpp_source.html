93 /* -------------------------------------------------------------------------
94  FUNCTION: LTC6948_read
95  - reads 8 bit Data field to LTC6948.
96  - has to shift data by one bit to account for RW bit
97  -------------------------------------------------------------------------- */
98 uint8_t LTC6948_read(uint8_t cs, int8_t address)
99 {
100  int8_t address_shift;
101  LT_union_int16_2bytes rx;
102 
103  address_shift =(address << 1) | 0x01; // shift to left to account for R/W bit, set bit high for read
104  spi_transfer_word(cs, address_shift<<8 , &rx.LT_uint16);
105 
106  LTC6948_reg[address]=rx.LT_byte[0];
107  return(rx.LT_byte[0]);
108 }

111 /* -------------------------------------------------------------------------
112  FUNCTION: LTC6948_read_field
113  For SPI FIELDS located in 1 or multiple address location
114  - reads specific address locations
115  - identifies and returns specific field in question
116  - can handle SPI fields in multiple addresses, if MSB bit is in the lower number address
117 --------------------------------------------------------------------------- */
118 long LTC6948_read_field(uint8_t cs, uint8_t address, uint8_t MSB_loc, uint8_t numbits)
119 {
120  int bit_shift, i, num_reg;
121  long field_val, maskbits, pow2;
122 
123  num_reg=0;
124  field_val=0;
125 // determines how many register are used
126  do
127  {
128  bit_shift = (MSB_loc+1)- (numbits-num_reg*8); // determines bit_shift for last register location
129  field_val=LTC6948_read(cs, (address+num_reg))+(field_val<<8); // reads current address locations, shifts previous address location 8 bits
130  num_reg++;
131  }
132  while ((bit_shift<0) && (num_reg<4));
133 
134 // creates a bit mask for complete word,
135  maskbits = 1;
136  pow2=1;
137  for (i=1, maskbits=1; i<numbits; i++)
138  {
139  pow2=pow2*2;
140  maskbits = maskbits+pow2;
141  }
142 
143  field_val=(field_val >>bit_shift) &maskbits;
144  return field_val;
145 }

174 /* -------------------------------------------------------------------------
175  FUNCTION: LTC6948_write_field
176  For SPI FIELDS
177  - reads specific address location
178  - identifies and returns specific field in question
179  - can handle SPI fields in multiple addresses, if MSB bit is in the lower number address
180 ---------------------------------------------------------------------------- */
181 uint8_t LTC6948_write_field(uint8_t cs, long field_data, uint8_t address, uint8_t MSB_loc, uint8_t numbits)
182 {
183  long current_content, desired_content, reg_val;
184  int LSB_loc, i, j, num_reg, bit_shift;
185  long temp_arr[32];
186 
187  for (i=0; i<32 ; i++) temp_arr[i]=0; // init temp_arr
188 
189 // read data in current address location and put in a bit array
190  num_reg=0;
191  current_content=0;
192  do
193  {
194  bit_shift=(MSB_loc+1)-(numbits-num_reg*8);
195  current_content=LTC6948_read(cs, (address+num_reg)) + (current_content<<8);
196 
197  num_reg++;
198  }
199  while ((bit_shift<0) && (num_reg<4));
200  for (i=0; i<(8*num_reg); i++)
201  {
202  temp_arr[i]=(current_content>>i) & 1;
203  }
204 
205 // exchange current bits with desired bits
206  LSB_loc = 8*(num_reg-1)+MSB_loc-numbits+1;
207  for (i=LSB_loc, j=0; i<=(MSB_loc+(num_reg-1)*8); i++, j++)
208  {
209  temp_arr[i] = (field_data>>j) &1;
210  } // end of for loop
211 
212 // reconstruct bits into an integer
213  desired_content = 0;
214  for (i=0; i<(8*num_reg); i++)
215  {
216  desired_content = desired_content | (temp_arr[i]<<i);
217  } // end of for loop
218 
219 // write new field value to part
220  for (i=0; i<num_reg; i++)
221  {
222  reg_val = (desired_content >> 8*(num_reg-1-i)) & 0xff;
223  LTC6948_write(cs, (address+i), reg_val);
224  } // end of for loop
225 } // end of LTC6948_write_field

544  FUNCTION: calc_odiv
545  - calculates the output divider setting based on the frf and version of LTC6948
546  - @return odiv = 1-6 for valid setting, 999 as invalid frequency
547 ---------------------------------------------------------------------------- */
548 unsigned long LTC6948_calc_odiv(char part_version[], unsigned long frf[2])
549 {
550  unsigned long odiv, max_fout, min_fout, temp_val,i, temp_out;
551  unsigned long max_fout64[2];
552  unsigned long min_fout64[2];
553  unsigned long temp_fout[2];
554  unsigned long temp_i[2];
555  boolean valid_input=false;
556 
557 // Determine which frequency range to verify too based on part version
558  if (part_version[8]=='1') // if this is a LTC6948-1
559  {
560  max_fout=LTC6948_1_MAXFREQ;
561  min_fout=LTC6948_1_MINFREQ;
562  }
563  else if (part_version[8]=='2') // if this is a LTC6948-2
564  {
565  max_fout=LTC6948_2_MAXFREQ;
566  min_fout=LTC6948_2_MINFREQ;
567  }
568  else if (part_version[8]=='3') // if this is a LTC6948-3
569  {
570  max_fout=LTC6948_3_MAXFREQ;
571  min_fout=LTC6948_3_MINFREQ;
572  }
573  else if (part_version[8]=='4') // if this is a LTC6948-4
574  {
575  max_fout=LTC6948_4_MAXFREQ;
576  min_fout=LTC6948_4_MINFREQ;
577  }
578  else
579  {
580  Serial.print("No default Frequencies for this board: ");
581  Serial.println(part_version);
582  } // end if-then-else statement
583  HZto64(max_fout64,max_fout,0L);
584  HZto64(min_fout64,min_fout,0L);
585

595  {

597  {
598  valid_input=true;

600  }
601  }
602  } // end of for loop
603 
604  if (valid_input == false) odiv= 999L;
605  return odiv;

609 /* -------------------------------------------------------------------------
610  FUNCTION: LTC6948_set_frf
611  Calculates the integer (N), fractional (NUM) and output divider (OD) SPI values
612  using self created 64bit math functions.
613 
614  Datasheet equations
615  fvco = fpfd*(N + F)
616  frf = fvco/O
617  fpfd = fref/R
618  F=NUM/2^18 = NUM/262144
619 
620  can be modified to the following equations
621  N = (int) (fvco/fpfd) = (int) frf*O*R/fref
622  NUM = (int) [262144 * (frf*O*R/fref -N)]
623 
624  where
625  N = ndiv, O= odiv, NUM=fracN_num in the code below
626 
627  Linduino One (Arduino Uno) is limited to 32 bit floats/double/long.
628  32 bit math functions will create rounding errors with the above equations,
629  that can result in frequency errors.
630  Therefore, the following code uses self created 64bit functions for 64bit integer math.
631 
632  frf (33 bits) LTC6948-4 max frf/fvco = 6.4GHZ, which is 23 bit number (2^33 = 8.59G)
633  fref (23 bits) LTC6948 min fref = 10MHz, which is a 23 bit number (2^23 = 8.3M)
634  O (3 bits)
635  R (5 bits)
636 
637  step 1: create 64 bit frf and fref numbers
638 
639  step 2: calculate O (output divider)
640 
641  step 3: get current R-divider setting
642 
643  step 4: calculate frf*O*R
644  max bit count/resolution: 33b+3b+5b= 41b
645  step 5: calculate N(10b), using value from step 1
646  N = (int) frf*O*R/fref
647  max bit count/resolution: 41b-23b = 18b
648  step 6: calculate NUM (18b)
649  NUM = (int) [262144 * (frf*O*R/fref -N)]
650  = (int) [2^18 * (frf*O*R/fref -N)]
651  to get the correct value the above calculation are performed like this
652  = ((int)((frf*O*R/fref)<<23) - N<<23 + rounding) >> 5, where << & >> are bit shift left, right
653  * note << 23b was chosen, because step one use 41b, 41b +23b = 64b (max number for 64 bit math)
654  ** note >> 5 was chosen because 23-5 = 2^18, which is the same as multiplying by 262144
655  *** the rounding variable is equivalent to adding 0.5 to the NUM number. The integer operation always
656  rounds down, so adding a 0.5 to the number before rounding down ensures the closest value is
657  selected.
658  In this function the lower 5 bits of (frf*O*R/fref)<<23) - N<<23 (possible variable 0-31)
659  are check to see if they are >=16.
660  If they are then 16 is added to this number before performing the int and >>5 functions.
661 ---------------------------------------------------------------------------- */
662 void LTC6948_set_frf(char part_version[])
663 {
664  unsigned long frf_MHz, frf_Hz, fref_MHz, fref_Hz, odiv, rdiv, ndiv, fracN_num, frac_num_remainder;
665  unsigned long N64[2], NUM64[2], R64[2], O64[2], temp_long[2];
666  char buffer[100];
667  unsigned long frf[2], frf_act[2];
668  unsigned long fref[2];
669  unsigned long temp_math[2];
670  unsigned long frf_rdiv_odiv[2];
671  unsigned long frf_rdiv_odiv_int[2];
672  unsigned long roundup[2];
673 
674  /* step 1: create 64 bit frf and fref numbers
675  32 bit xxxx MHz number and 32 bit yyy yyy Hz number. */
676  frf_MHz=LTC6948_Frf_MHz;
677  frf_Hz=LTC6948_Frf_Hz;
678  HZto64(frf, frf_MHz, frf_Hz);
679 
680  fref_MHz=LTC6948_Fref_MHz;
681  fref_Hz=LTC6948_Fref_Hz;
682  HZto64(fref,fref_MHz,fref_Hz);
683 
684 // step 2: calculate O (output divider)
685  odiv=LTC6948_calc_odiv(part_version,frf);
686 
687 // step 3: get current R-divider setting
688  rdiv=get_LTC6948_SPI_FIELD(LTC6948_CS,LTC6948_RD); // reads selected field
689 
690 // step 3: calculate frf*O*R
691  frf_rdiv_odiv[0]=0;
692  frf_rdiv_odiv[1]=odiv*rdiv;
693  mul64(frf_rdiv_odiv, frf); // odiv*rdiv*frf
694  frf_rdiv_odiv_int[0]=frf_rdiv_odiv[0]; // copy odiv*rdiv*frf to another variable
695  frf_rdiv_odiv_int[1]=frf_rdiv_odiv[1];
696 
697 // step 4: calculate N(10b), using value from step 3; N = (int) frf*O*R/fref
698  temp_math[0]=fref[0]; // copy fref to temp variable for math operation
699  temp_math[1]=fref[1];
700  div64(frf_rdiv_odiv_int, temp_math); // frf_rdiv_odiv_int= [(frf*odiv*rdiv)]/fref --> int(fvco/fpfd)
701  ndiv=frf_rdiv_odiv_int[1];
702 
703 // step 5: calculate NUM (18b), NUM = (int) [262144 * (frf*O*R/fref -N)]
704 // SEE notes in function header for step 5
705 // ((int)((frf*O*R/fref)<<23) - N<<23 + rounding) >> 5, where << & >> are bit shift left, right
706  shl64by(frf_rdiv_odiv,23);
707  div64(frf_rdiv_odiv, fref); // frf_rdiv_odiv --> [(frf*odiv*rdiv)<<23]/fref
708 
709  shl64by(frf_rdiv_odiv_int,23); // frf_rdiv_odiv_int --> [(int)((frf*odiv*rdiv)/fref)] <<23
710 
711  sub64(frf_rdiv_odiv,frf_rdiv_odiv_int);
712 
713 // NUM is a 18 bit number in the spi register
714 // At this point in the program it is represented by 23 bits
715 // if the lower 5 bits are >=16 (16/31),
716 // then add 16 to the frac_num_remainder. After the bits are
717 // shifted down 5 bits this will be close to added 0.5 bits.
718  frac_num_remainder=(frf_rdiv_odiv[1] & 31L);
719  if (frac_num_remainder >= 16L)
720  {
721  roundup[0]=0L;
722  roundup[1]=16L;
723  add64(frf_rdiv_odiv,roundup);
724  }
725 
726 // shift NUM to get desired 18 bit number for SPI map.
727  shr64by(frf_rdiv_odiv,5);
728  fracN_num=frf_rdiv_odiv[1];
729 
730 // program part and print out results to screen
731  set_LTC6948_SPI_FIELD(LTC6948_CS,LTC6948_OD,odiv); // programs output divider
732  Serial.print("OD = ");
733  Serial.println(odiv);
734 
735  Serial.print("RD = ");
736  Serial.println(rdiv);
737 
738  set_LTC6948_SPI_FIELD(LTC6948_CS,LTC6948_ND,ndiv); // programs N-divider
739  Serial.print(F("ND = "));
740  Serial.println(ndiv);
741 
742  if (fracN_num > 0L)
743  {
744  set_LTC6948_SPI_FIELD(LTC6948_CS,LTC6948_INTN,0L); // programs fractional mode
745  set_LTC6948_SPI_FIELD(LTC6948_CS,LTC6948_NUM,fracN_num); // programs NUM
746  if ( (ndiv<35) || (ndiv>1019) ) Serial.print(F("N DIV set to invalid setting - REFER TO DATASHEET\n"));
747  }
748  else
749  {
750  set_LTC6948_SPI_FIELD(LTC6948_CS,LTC6948_INTN,1L); // programs INTN
751  Serial.print(F("When NUM=0, changes to Integer Mode (NUM can not be 0)\n"));
752  if ( (ndiv<32) || (ndiv>1023) ) Serial.print(F("N DIV set to invalid setting - REFER TO DATASHEET\n"));
753  }
754  Serial.print(F("NUM = "));
755  Serial.println(fracN_num);
756 
757  set_LTC6948_SPI_FIELD(LTC6948_CS,LTC6948_CAL,1); // calibrates part
758 
759 
760 ///// calculate actual fout here and print out
761  frf_act[0]=fref[0];
762  frf_act[1]=fref[1];
763  N64[0]=0;
764  NUM64[0]=0;
765  R64[0]=0;
766  O64[0]=0;
767  N64[1]=ndiv;
768  NUM64[1]=fracN_num;
769  R64[1]=rdiv;
770  O64[1]=odiv;
771 
772  shl64by(N64,18);
773  add64(N64,NUM64); // (N*262144 +NUM)
774  div64(frf_act,R64); // fref/rdiv (or fpfd)
775  mul64(frf_act,N64); // (N*262144 +NUM)*fpdf
776  div64(frf_act,O64); // ((N*262144 +NUM)*fpdf)/odiv
777  shr64by(frf_act,17);
778  if (frf_act[0]==0) // no 32 bit overflow case
779  {
780  Serial.print(F("Actual Output Frequency (closest available) is "));
781  Serial.print(frf_act[1]);
782  Serial.println("Hz");
783  }
784  else // overflow case
785  {
786  temp_math[0]=frf_act[0];
787  temp_math[1]=frf_act[1];
788  temp_long[0]=0;
789  temp_long[1]=OneMHz;
790  div64(temp_math,temp_long); // frf/1e6
791  Serial.print(F("Actual Output Frequency (closest available) is "));
792  Serial.print(temp_math[1]);
793  Serial.print("MHz + ");
794  temp_long[0]=0;
795  temp_long[1]=OneMHz;

797  sub64(frf_act,temp_math); // frf-int(frf/1e6)*1e6
798  Serial.print(frf_act[1]);
799  Serial.println("Hz");
800  }
