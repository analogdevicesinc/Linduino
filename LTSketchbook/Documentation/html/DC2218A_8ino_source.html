243  int switch_pos; // New switch possition

247  do

250  An1 = analogRead(A1); // Read 8 position switch pin 2
251  An2 = analogRead(A2); // Read 8 position switch pin 3
252

254 
255  //Interpret 3 bits from 8 position switch as decimal integer
256  if (An0 > 120 && An1 < 120 && An2 < 120)
257  switch_pos = 1;
258  else if (An0 < 120 && An1 > 120 && An2 < 120)
259  switch_pos = 2;
260  else if (An0 > 120 && An1 > 120 && An2 < 120)
261  switch_pos = 3;
262  else if (An0 < 120 && An1 < 120 && An2 > 120)
263  switch_pos = 4;
264  else if (An0 > 120 && An1 < 120 && An2 > 120)
265  switch_pos = 5;
266  else if (An0 < 120 && An1 > 120 && An2 > 120)

276 
277  // Call function associated with corresponding switch position value
278  switch (switch_pos)

280  case 0:
281  orbit();
282  break;
283  case 1:

337 //! Function to read I2C bytes depending on the bytes received from GUI.
338 int8_t read_bytes(uint8_t size, uint8_t byte1, uint8_t byte2, uint8_t byte3)
339 {
340  int8_t ret = 0 ;
341  if (size == 250 && byte1 == 1 && byte2 == 1 && byte3 == 1) // ALERT - ACMODE READ
342  {
343  uint8_t fault;
344  fault = 1;
345  fault = digitalRead(2); // digital pin2 is connected to ALERT/
346  if (!fault) // fault condition
347  Serial.write(0);
348  else
349  Serial.write(1);
350  }
351  else
352  {
353  uint8_t x;
354  if (i2c_start() != 0) //I2C START
355  return (1); //Stop and return 0 if START fail
356 
357  // write byte1 and read size number of bytes
358  ret |= i2c_write(byte1);
359  if (ret == 0) // If response = ACK (slave responded)
360  {
361  for (x = 0; x < (size - 1); ++x)
362  {
363  Serial.write(i2c_read(WITH_ACK));
364  }
365  Serial.write(i2c_read(WITH_NACK));
366  }
367  else // If response = NACK (no slave responded
368  {
369  for (x = 0; x < size; ++x)
370  {
371  Serial.write(0); // The GUI is waiting to read back bytes, sending dummy bytes
372  }
373  }
374  i2c_stop(); //I2C STOP
375  }
376  if (ret != 0) //Returns 1 if failed
377  return (1);
378  return (0); //Returns 0 if success
379 }
