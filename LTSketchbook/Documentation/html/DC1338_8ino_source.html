1 /*!
2 Linear Technology DC1338 Demonstration Board.
3 LTC2990: 14-bit ADC Quad I2C Voltage, Current, and Temperature monitor.
4 
5 @verbatim
6 
7  Setup:
8  Set the terminal baud rate to 115200 and select the newline terminator.
9  A precision voltage source (preferably low-noise) may be used to apply a voltage
10  to input terminals V1-V8. A precision voltmeter may be used to verify applied
11  voltages. An oscilloscope may be used to view the PWM output. Ensure JP5, JP6
12  and JP7 are in the LOW position. Refer to Demo Manual DC1338A
13 
14  Explanation of Commands:
15 
16  1 - Single-Ended Voltage - Selects the Single-Ended Voltage Menu.
17  1-4: Displays the measured single-ended voltage at one of the V1-V4
18  inputs. When measuring V1 and V8, ensure jumpers are set to VOLT
19  position.
20  5: Vcc - Displays the measured Vcc voltage.
21  6: ALL - Displays the measured voltages at all of the V1-V8 inputs
22  and Vcc.
23 
24  2 - Differential Voltage - Selects the Differential Voltage Menu.
25  Maximum full scale differential voltage is 0.300V.
26 
27  1-2: Displays the measured differential voltage across one of the V1-V4
28  input pairs. The input common-mode range is 0V to Vcc. It is
29  easiest to ground the lower input. When measuring V1 and V4,
30  ensure jumpers are set to VOLT position.
31  3: ALL - Displays the measured differential voltages at all terminals.
32 
33  3 - Temperature - Selects the Temperature Menu
34  To measure temperature using on-board transistors, set JP1, JP2, JP3 and JP4
35  to TEMP position.
36  1: V1-V2 - Measure temperature of Q1 (mounted to demo board) when JP1
37  and JP2 are in TEMP position.
38  2: V3-V4 - Measure temperature of external transistor connected to V3
39  and V4 terminals.
40  3: Internal - Measure temperature using the internal temperature
41  sensor.
42  4: All - Displays temperatures at all connections as well as the
43  internal temperature sensor.
44 
45  4 - Settings - Selects the Settings Menu
46  Toggles temperature units between degrees Celsius or degrees Kelvin.
47  1: Toggle temperature units between degrees Celsius and degrees
48  Kelvin.
49 
50 USER INPUT DATA FORMAT:
51  decimal : 1024
52  hex : 0x400
53  octal : 02000 (leading 0 "zero")
54  binary : B10000000000
55  float : 1024.0
56 
57 @endverbatim

155 //! Repeats Linduino loop
156 void loop()
157 {
158  int8_t ack=0;
159 
160  uint8_t user_command;
161  if (demo_board_connected) // Does nothing if the demo board is not connected
162  {
163  if (Serial.available()) // Checks for user input
164  {
165  user_command = read_int(); // Reads the user command
166  if (user_command != 'm')
167  Serial.println(user_command);
168  ack = 0;
169  switch (user_command) // Prints the appropriate submenu
170  {
171  case 1:
172  ack |= menu_1_single_ended_voltage(); // Print single-ended voltage menu
173  break;
174  case 2:
175  ack |= menu_2_read_differential_voltage(); // Differential voltage menu
176  break;
177  case 3:
178  ack |= menu_3_read_temperature(); // Temperature menu
179  break;
180  case 4:
181  ack |= menu_4_settings(); // Settings menu
182  break;
183  default:
184  Serial.println("Incorrect Option");
185  break;
186  }
187  if (ack != 0)
188  {
189  Serial.print(F("Error: No Acknowledge. Check I2C Address.\n"));
190  }
191  print_prompt();
192  }
193  }
194 }

433 //! Read temperatures
434 //! @return Returns the state of the acknowledge bit after the I2C address write. 0=acknowledge, 1=no acknowledge.
435 int8_t menu_3_read_temperature()
436 {
437  int8_t ack=0;
438  boolean isKelvin = false; // Keeps track of the unit of measurement
439  uint8_t user_command;
440  do
441  {
442  // Displays temperature menu
443  Serial.print(F("\nTemperature\n\n"));
444  Serial.print(F(" 1-V1-V2\n"));
445  Serial.print(F(" 2-V3-V4\n"));
446  Serial.print(F(" 3-Internal\n"));
447  Serial.print(F(" 4-ALL\n"));
448  Serial.print(F(" m-Main Menu\n"));
449  Serial.print(F("\n\nEnter a command: "));
450  user_command = read_int(); // Reads the user command
451  if (user_command == 'm') // Print m if it is entered
452  {
453  Serial.print(F("m\n"));
454  }
455  else
456  Serial.println(user_command); // Print user command
457 
458  // Read Temperature
459  int8_t data_valid = 0;
460  int16_t adc_code;
461  uint8_t reg_data;
462  float temperature;
463 
464  // Enables temperature mode
465  ack |= LTC2990_register_set_clear_bits(LTC2990_I2C_ADDRESS, LTC2990_CONTROL_REG, LTC2990_TR1_TR2, LTC2990_VOLTAGE_MODE_MASK);
466  // Triggers a conversion by writing any value to the trigger register
467  ack |= LTC2990_register_write(LTC2990_I2C_ADDRESS, LTC2990_TRIGGER_REG, 0x01);
468  // Flushes one reading following mode change.
469  ack |= LTC2990_adc_read_new_data(LTC2990_I2C_ADDRESS, LTC2990_V1_MSB_REG, &adc_code, &data_valid, LTC2990_TIMEOUT);
470  //ack |= LTC2990_register_write(LTC2990_I2C_ADDRESS, LTC2990_TRIGGER_REG, 0x01);
471  //delay(165); //Wait max conversion time
472 
473  // Reads temperature from ADC and prints it.
474  switch (user_command)
475  {
476  case 1:
477  // Flush one ADC reading in case it is stale. Then, take a new fresh reading.
478  ack |= LTC2990_adc_read_new_data(LTC2990_I2C_ADDRESS, LTC2990_V1_MSB_REG, &adc_code, &data_valid, LTC2990_TIMEOUT);
479  ack |= LTC2990_register_read(LTC2990_I2C_ADDRESS, LTC2990_CONTROL_REG, &reg_data);
480  if (reg_data & LTC2990_KELVIN_ENABLE) isKelvin= true;
481  else isKelvin=false;
482  temperature = LTC2990_temperature(adc_code, LTC2990_TEMPERATURE_lsb, isKelvin);
483  Serial.print(F("\n V1-V2: "));
484  Serial.print(temperature, 2);
485  if (isKelvin) Serial.print(F(" K\n"));
486  else Serial.print(F(" C\n"));
487  break;
488  case 2:
489  // Flush one ADC reading in case it is stale. Then, take a new fresh reading.
490  ack |= LTC2990_adc_read_new_data(LTC2990_I2C_ADDRESS, LTC2990_V3_MSB_REG, &adc_code, &data_valid, LTC2990_TIMEOUT);
491  ack |= LTC2990_register_read(LTC2990_I2C_ADDRESS, LTC2990_CONTROL_REG, &reg_data);
492  if (reg_data & LTC2990_KELVIN_ENABLE) isKelvin=true;
493  else isKelvin = false;
494  temperature = LTC2990_temperature(adc_code, LTC2990_TEMPERATURE_lsb, isKelvin);
495  Serial.print(F("\n V3-V4: "));
496  Serial.print(temperature, 2);
497  if (isKelvin) Serial.print(F(" K\n"));
498  else Serial.print(F(" C\n"));
499  break;
500  case 3:
501  // Flush one ADC reading in case it is stale. Then, take a new fresh reading.
502  ack |= LTC2990_adc_read_new_data(LTC2990_I2C_ADDRESS, LTC2990_TINT_MSB_REG, &adc_code, &data_valid, LTC2990_TIMEOUT);
