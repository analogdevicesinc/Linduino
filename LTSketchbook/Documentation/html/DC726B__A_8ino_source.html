158 //! Sets the output frequency.

164 
165  Serial.print(F("Enter the desired clock freq (KHz):"));
166 
167  freq = read_float();
168 
169  Serial.println(freq, 4);

173 
174

178 //! Manually Sets OCT and DAC Code
179 void manually_set_reg()
180 {
181  uint8_t oct;
182  uint8_t ack;

184 
185  Serial.print(F("Enter the OCT:"));
186

197  Serial.println(dac_code);
198  Serial.println();
199 
200  // Manually set registers
201  LTC6903_write(LTC6903_CS, (uint16_t)((oct<<12)|(dac_code<<2)|output_config));
202 }
