86 //! Repeats Linduino loop
87 //! @return void
88 void loop()
89 {
90  uint8_t user_command;
91  uint8_t res;
92  uint8_t model[7];
93  uint8_t *addresses = NULL;
94 
95  if (Serial.available()) //! Checks for user input
96  {
97  user_command = read_int(); //! Reads the user command
98  if (user_command != 'm')
99  Serial.println(user_command);
100 
101  switch (user_command) //! Prints the appropriate submenu
102  {
103  case 1:
104  menu_1_basic_commands(); // Print single-ended voltage menu
105  break;
106  case 2:
107  pmbus->readModel(ltc3887_i2c_address, model);
108  Serial.print(F("MODEL "));
109  Serial.print((char *)model);
110  Serial.println();
111  break;
112  case 3:
113  res = pmbus->readPmbusRevision(ltc3887_i2c_address);
114  Serial.print(F("REVISION "));
115  Serial.println(res, HEX);
116  break;
117  case 4:
118  pmbus->enablePec(ltc3887_i2c_address);
119  delete smbus;
120  delete pmbus;
121  smbus = new LT_SMBusPec();
122  pmbus = new LT_PMBus(smbus);
123  break;
124  case 5:
125  pmbus->disablePec(ltc3887_i2c_address);
126  delete smbus;
127  delete pmbus;
128  smbus = new LT_SMBusNoPec();
129  pmbus = new LT_PMBus(smbus);
130  break;
131  case 6:
132  addresses = smbus->probe(0);
133  while (*addresses != 0)
134  {
135  Serial.print(F("ADDR 0x"));
136  Serial.println(*addresses++, HEX);
137  }
138  break;
139  case 7 :
140  pmbus->reset(ltc3887_i2c_address);
141  break;
142  default:
143  Serial.println(F("Incorrect Option"));
144  break;
145  }
146  print_prompt();
147  }
