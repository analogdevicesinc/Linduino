82 // Read the specified ADC value (SENSE, ADIN, ADIN2) and output in human readable format to the serial console.
83 int8_t LTM9100_adc_read(uint8_t i2c_address, uint8_t base_address, float *register_data)
84 {
85  uint8_t data_LSBs, data_MSBs;
86  uint16_t data_ten_bit;
87  int8_t ack = 0;
88  ack |= LTM9100_register_read(i2c_address, base_address, &data_MSBs);
89  ack |= LTM9100_register_read(i2c_address, base_address+1, &data_LSBs);
90  data_ten_bit = ((uint16_t)data_MSBs<<2) | ((uint16_t)data_LSBs>>6);
91 
92  switch (base_address)
93  {
94  case LTM_9100_SENSE_E_REG:
95  *register_data = ((float)data_ten_bit * LTM_9100_SENSE_mV_PER_TICK)/sense_resistor;
96  Serial.print(F("ADC Current Sense Voltage(mA): "));
97  break;
98  case LTM_9100_ADIN_I_REG:
99  *register_data = ((float)data_ten_bit * LTM_9100_ADIN_V_PER_TICK)/adin_gain;
100  Serial.print(F("ADIN Voltage(V): "));
101  break;
102  case LTM_9100_ADIN2_G_REG:
103  *register_data = ((float)data_ten_bit * LTM_9100_ADIN2_V_PER_TICK)/adin2_gain;
104  Serial.print(F("ADIN2 Voltage(V or K): "));
105  break;
106  default:
107  return ack;
108  }
109 
110  Serial.println(*register_data, 4);
111  return ack;
112 }

122 // Sets any bit inside the LTM9100 using the standard I2C repeated start format.
123 int8_t LTM9100_bit_set(uint8_t i2c_address, uint8_t register_address, uint8_t bit_number)
124 {
125  uint8_t register_data;
126  uint8_t bit_mask;
127  int8_t ack = 0;
128  bit_mask = 0x01 << bit_number;
129  ack |= LTM9100_register_read(i2c_address, register_address, &register_data); //Read register
130  register_data = register_data | bit_mask; //Set the bit at bit_address
131  ack |= LTM9100_register_write(i2c_address, register_address, register_data); //Write new data to register
132  return(ack);
133 }

135 // Clears any bit inside the LTM9100 using the standard I2C repeated start format.
136 int8_t LTM9100_bit_clear(uint8_t i2c_address, uint8_t register_address, uint8_t bit_number)
137 {
138  uint8_t register_data;
139  uint8_t bit_mask;
140  int8_t ack = 0;
141  bit_mask = 0x01 << bit_number;
142  ack |= LTM9100_register_read(i2c_address, register_address, &register_data); //Read register
143  register_data = register_data & (~bit_mask); //Clears the bit at bit_address
144  ack |= LTM9100_register_write(i2c_address, register_address, register_data); //Write new data to register
145  return(ack);
146 }

148 // Read the bit specified by bit_number from the LTM9100.
149 int8_t LTM9100_bit_read(uint8_t i2c_address, uint8_t register_address, uint8_t bit_number, uint8_t *register_data)

165 // Read all LTM9100 registers and output to the serial console.
166 int8_t LTM9100_print_all_registers(uint8_t i2c_address)
167 {
168  int8_t ack = 0;
169  uint8_t data;
170  uint8_t i;
171  for (i=0; i<sizeof(reg_read_list); i++)
172  {
173  ack |= LTM9100_register_read(i2c_address, reg_read_list[i], &data); //! Read register
174  Serial.print("Register 0x");
175  Serial.print(reg_read_list[i], HEX);
176  Serial.print(":\t");
177  if (!ack)
178  {
179  Serial.print("0x");
180  Serial.println(data, HEX);
181  }
182  else
183  {
184  Serial.println("No Acknowledge");
185  }
186  }
187  return ack;
188 }

190 // Read all LTM9100 registers and output to the serial console every second until a key press is detected.
191 int8_t LTM9100_continuous_read_all_registers(uint8_t i2c_address)
192 {
193  int8_t ack = 0;
194  uint8_t count = 0;
195  float data = 0;
196  char anykey[1];
197 
198  //Note: Arduino serial monitor doesn't support all VT100 commands,
199  //thus the continuous_read can't scrollback and rewrite the latest register values.
200  while (count < 1)
201  {
202  ack |= LTM9100_print_all_registers(i2c_address);
203  ack |= LTM9100_adc_read(i2c_address, LTM_9100_SENSE_E_REG, &data);
204  ack |= LTM9100_adc_read(i2c_address, LTM_9100_ADIN_I_REG, &data);
205  ack |= LTM9100_adc_read(i2c_address, LTM_9100_ADIN2_G_REG, &data);
206  Serial.println("");
207 
208  count = Serial.readBytes(anykey, 1); //default timeout is 1000 milliseconds. Change via the setTimeout method of Serial.
209  }
210 
211  return ack;
212 }
