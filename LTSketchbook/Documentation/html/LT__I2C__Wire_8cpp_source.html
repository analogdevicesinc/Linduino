142 {
143  int8_t ret = 0;
144

156  return(1);

164  i--;

169  Wire.endTransmission();

171  return(0);

279 int8_t i2c_two_byte_command_read_block(uint8_t address, uint16_t command, uint8_t length, uint8_t *values)
280 {

290 
291 #if defined(ARDUINO_ARCH_SAM)
292  Wire.beginTransmission(address);
293  readBack = Wire.requestFrom((uint8_t)address, (uint8_t)length, (uint32_t)command, (uint8_t)2, (uint8_t)false);
294 #elif defined(ARDUINO_ARCH_SAMD) || defined(ARDUINO_ARCH_AVR)
295  Wire.beginTransmission(address);
296  Wire.write(byte(comm.b[1]));
297  Wire.write(byte(comm.b[0]));
298 
299  if (Wire.endTransmission(false)) // endTransmission(false) is a repeated start
300  {
301  // endTransmission returns zero on success
302  Wire.endTransmission();
303  return(1);
304  }
305  readBack = Wire.requestFrom((uint8_t)address, (uint8_t)length, (uint8_t)true);
306 #endif
307 
308  if (readBack == length)
309  {
310  while (Wire.available())
311  {
312  values[i] = Wire.read();
313  if (i == 0)
314  break;
315  i--;
316  }
317  return (0);
318  }
319  else
320  {
321  return (1);
322  }
