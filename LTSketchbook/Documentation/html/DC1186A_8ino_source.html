1 /*!
2 Linear Technology DC1186A Demonstration Board.
3 LTC2308: 12-Bit, 8-Channel 500ksps SAR ADC with SPI Interface.
4 
5 @verbatim
6 
7 NOTES
8  Setup:
9  Set the terminal baud rate to 115200 and select the newline terminator. Equipment
10  required is a voltage source (preferably low-noise) and a precision voltmeter. No
11  external power supply is required. Ensure all jumpers on the demo board are
12  installed in their default positions from the factory. Refer to Demo Manual
13  DC1186A.
14 
15  How to test Single-Ended mode:
16  Bipolar Mode:
17  The voltage source should be connected to one of the inputs CH0,...CH7. Ensure the input is
18  within its specified absolute input voltage range. (It is easiest
19  to tie the voltage source negative terminal to COM.) Ensure the voltage
20  source is set within the range of 0V to +4.096V.
21  (Swapping input voltages results in a reversed polarity reading.)
22 
23  Unipolar Mode:
24  The voltage source should be connected to one of the inputs CH0,...CH7. Ensure the input is
25  within its specified absolute input voltage range. (It is easiest
26  to tie the voltage source negative terminal to COM.) Ensure the voltage
27  source is set within the range of 0V to +4.096V.
28 
29  How to test Differential Mode:
30  Bipolar Mode:
31  The voltage source should be connected between paired channels. Ensure both
32  inputs are within their specified absolute input voltage range. (It is easiest
33  to tie the voltage source negative terminal to COM or GND.) Ensure the voltage
34  source is set within the range of 0 to +4.096V (differential voltage range).
35  (Swapping input voltages results in a reversed polarity reading.)
36 
37  Unipolar Mode:
38  The voltage source should be connected between paired channels. (It is easiest
39  to tie the voltage source negative terminal to COM or GND.) Ensure both inputs
40  are within their specified absolute input voltage range. (It is easiest to tie
41  the voltage source negative terminal to COM or GND.) Ensure the voltage source
42  is set within the range of 0 to +4.096V (differential voltage range).
43 
44 USER INPUT DATA FORMAT:
45  decimal : 1024
46  hex : 0x400
47  octal : 02000 (leading 0 "zero")
48  binary : B10000000000
49  float : 1024.0
50 
51 @endverbatim

189 // Function Definitions
190 //! Read channels
191 void menu_1_read_input()
192 {
193  uint8_t user_command;
194  uint16_t adc_command; // The LTC2308 command byte
195  uint16_t adc_code = 0; // The LTC2308 code
196  uint16_t display_code;
197  uint8_t x, y, startcount, endcount;
198  float adc_voltage;
199 
200  if (uni_bipolar == LTC2308_UNIPOLAR_MODE)
201  {
202  if (single_ended_differential == LTC2308_SINGLE_ENDED_MODE)
203  {
204  Serial.println(F(" Read Input in Single-Ended, Unipolar mode:"));
205  Serial.println(F(" Note that in Unipolar mode, input voltages less than zero are reported as 0.0V"));
206  user_command = print_user_command_single_ended();
207  }
208  else
209  {
210  Serial.println(F(" Read Input in Differential, Unipolar mode:"));
211  Serial.println(F(" Note that in Unipolar mode, input voltages less than zero are reported as 0.0V"));
212  user_command = print_user_command_differential();
213  }
214  }
215  else
216  {
217  if (single_ended_differential == LTC2308_SINGLE_ENDED_MODE)
218  {
219  Serial.println(F(" Read Input in Single-Ended, Bipolar mode:"));
220  user_command = print_user_command_single_ended();
221  }
222  else
223  {
224  Serial.println(F(" Read Input in Differential, Bipolar mode:"));
225  user_command = print_user_command_differential();
226  }
227  }
228 
229  if (user_command == 8) //read all channels
230  {
231  startcount = 0;
232  endcount = 7;
233  }
234  else
235  {
236  startcount = user_command;
237  endcount = user_command;
238  }
239 
240  if (single_ended_differential == LTC2308_SINGLE_ENDED_MODE) //read single-ended
241  {
242  adc_command = BUILD_COMMAND_SINGLE_ENDED[0] | uni_bipolar;
243  LTC2308_read(LTC2308_CS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
244  for (x = startcount; x <= endcount; x++)
245  {
246  adc_command = BUILD_COMMAND_SINGLE_ENDED[x % 8] | uni_bipolar; // Send channel config for the NEXT conversion to take place
247  LTC2308_read(LTC2308_CS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
248  LTC2308_read(LTC2308_CS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
249 
250  display_code = adc_code >> (16 - LTC2308_bits);
251  display_code = display_code & 0xFFF;
252 
253  Serial.print(F(" Received Code: b"));
254  Serial.println(display_code, BIN);
255 
256  adc_voltage = LTC2308_code_to_voltage(adc_code, LTC2308_vref, uni_bipolar);
257  Serial.print(F(" Voltage read on "));
258  Serial.print(F("Ch"));
259  Serial.print(x);
260  Serial.print(F(": "));
261  Serial.print(adc_voltage, 4);
262  Serial.println(F("V"));
263  Serial.println();
264  }
265  }
266 
267  else //read differential
268  {
269  adc_command = BUILD_COMMAND_SINGLE_ENDED[0] | uni_bipolar;
270  LTC2308_read(LTC2308_CS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
271  for (x = startcount; x <= endcount; x++)
272  {
273  adc_command = BUILD_COMMAND_DIFF[x % 8] | uni_bipolar; // Send channel config for the NEXT conversion to take place
274  LTC2308_read(LTC2308_CS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
275  LTC2308_read(LTC2308_CS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
276 
277  display_code = adc_code >> (16 - LTC2308_bits);
278  display_code = display_code & 0xFFF;
279 
280  Serial.print(F(" Received Code: b"));
281  Serial.println(display_code, BIN);
282 
283  adc_voltage = LTC2308_code_to_voltage(adc_code, LTC2308_vref, uni_bipolar);
284  Serial.print(F(" Voltage read between Chs "));
285  print_channel_selection(x + y);
286  Serial.print(adc_voltage, 4);
287  Serial.println(F("V"));
288  Serial.println();
289  }
290  }
291 }
