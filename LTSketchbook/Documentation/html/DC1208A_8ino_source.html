188 //! Build and send the snapshot command
189 //! @return Returns the state of the acknowledge bit after the I2C address write. 0=acknowledge, 1=no acknowledge.
190 int8_t LTC4151_snapshot(uint8_t i2c_address, //!< I2C address of the LTC4151.
191  uint8_t channel) //!< Desired channel from the header
192 {
193  int8_t ack = 0;
194  ack |= LTC4151_write(i2c_address, LTC4151_CONTROL_REG, LTC4151_CONTINUOUS_MODE); //! Disable previous snapshot mode to allow a new snapshot
195  ack |= LTC4151_write(i2c_address, LTC4151_CONTROL_REG, (channel | LTC4151_SNAPSHOT_MODE)); //! Re-enable snapshot mode on selected channel
196  return(ack);
197 }

199 //! Reads all inputs in continuous mode
200 //! @return Returns the state of the acknowledge bit after the I2C address write. 0=acknowledge, 1=no acknowledge.
201 int8_t menu_1_continuous_mode()
202 {
203  int8_t ack = 0;
204 
205  // Read Continuous Mode
206  ack |= LTC4151_write(LTC4151_I2C_ADDRESS, LTC4151_CONTROL_REG, LTC4151_CONTINUOUS_MODE); //! Set LTC4151 to continuous mode
207  do
208  {
209  uint16_t current_sense_adc_code, vin_adc_code, adin_adc_code;
210 
211  ack |= LTC4151_read_12_bits(LTC4151_I2C_ADDRESS, LTC4151_SENSE_MSB_REG, &current_sense_adc_code); //! Reads two bytes for SENSE, MSB register followed by LSB register
212  ack |= LTC4151_read_12_bits(LTC4151_I2C_ADDRESS, LTC4151_VIN_MSB_REG, &vin_adc_code); //! Reads two bytes for VIN, MSB register followed by LSB register
213  ack |= LTC4151_read_12_bits(LTC4151_I2C_ADDRESS, LTC4151_ADIN_MSB_REG, &adin_adc_code); //! Reads two bytes for ADIN, MSB register followed by LSB register
214 
215  float current, VIN_voltage, ADIN_voltage;
216  current = LTC4151_code_to_sense_current(current_sense_adc_code, resistor, LTC4151_sense_lsb);
217  VIN_voltage = LTC4151_code_to_vin_voltage(vin_adc_code, LTC4151_vin_lsb);
218  ADIN_voltage = LTC4151_code_to_ADIN_voltage(adin_adc_code, LTC4151_adin_lsb);
219 
220  //! Display current, VIN voltage, and ADIN Voltage
221 
222  Serial.println();
223  Serial.println("***************************");
224  Serial.print("Load Current: ");
225  Serial.print(current);
226  Serial.println(" A\n");
227 
228  Serial.print("VIN Voltage: ");
229  Serial.print(VIN_voltage);
230  Serial.println(" V\n");
231 
232  Serial.print("ADIN Voltage: ");
233  Serial.print(ADIN_voltage);
234  Serial.println(" mV\n");
235 
236  Serial.println("m-Main Menu");
237  delay(2000);
238  }
239  while (Serial.available() == false); //! Repeats until user enters a character
240  read_int(); // Clears Serial.available
241  return(ack);
242 }

244 //! Reads inputs in snapshot mode
245 //! @return Returns the state of the acknowledge bit after the I2C address write. 0=acknowledge, 1=no acknowledge.
246 int8_t menu_2_snapshot_mode()
247 {
248  int8_t ack = 0;
249  uint8_t user_command;
250 
251  // Snapshot Mode
252  do
253  {
254  //! Displays the Snapshot Mode menu
255  Serial.println("***************************");
256  Serial.println("Snapshot Mode");
257  Serial.println("1-Load Current:");
258  Serial.println("2-V_IN Voltage:");
259  Serial.println("3-ADIN Voltage:");
260  Serial.println("m-Main Menu");
261  Serial.print("Enter a command: ");
262 
263  user_command = read_int(); //! Reads the user command
264 
265  if (user_command == 'm') // Print m if it is entered
266  Serial.println("m");
267  else
268  Serial.println(user_command); // Print user command
269  Serial.println();
270 
271  uint16_t current_sense_adc_code, vin_adc_code, adin_adc_code;
272  float current, VIN_voltage, ADIN_voltage;
273 
274  //! Reads sense resistor current, Vin voltage, or ADIN voltage
275  switch (user_command)
276  {
277  case 1:
278  ack |= LTC4151_snapshot(LTC4151_I2C_ADDRESS, LTC4151_SENSE_CHANNEL_REG);
279  delay(200); //Wait for conversion to finish. Could watch busy bit instead to reduce wait time.
280  ack |= LTC4151_read_12_bits(LTC4151_I2C_ADDRESS, LTC4151_SENSE_MSB_REG, &current_sense_adc_code);
281  current = LTC4151_code_to_sense_current(current_sense_adc_code, resistor, LTC4151_sense_lsb);
282  Serial.print("Load Current: ");
283  Serial.print(current);
284  Serial.println(" A\n");
285  break;
286  case 2:
287  ack |=LTC4151_snapshot(LTC4151_I2C_ADDRESS, LTC4151_VIN_CHANNEL_REG);
288  delay(100); //Wait for conversion to finish. Could watch busy bit instead to reduce wait time.
289  ack |= LTC4151_read_12_bits(LTC4151_I2C_ADDRESS, LTC4151_VIN_MSB_REG, &vin_adc_code);
290  VIN_voltage = LTC4151_code_to_vin_voltage(vin_adc_code, LTC4151_vin_lsb);
291  Serial.print("VIN Voltage: ");
292  Serial.print(VIN_voltage);
293  Serial.println(" V\n");
294  break;
295  case 3:
296  ack |= LTC4151_snapshot(LTC4151_I2C_ADDRESS, LTC4151_ADIN_CHANNEL_REG);
297  delay(100); //Wait for conversion to finish. Could watch busy bit instead to reduce wait time.
298  ack |= LTC4151_read_12_bits(LTC4151_I2C_ADDRESS, LTC4151_ADIN_MSB_REG, &adin_adc_code);
299  ADIN_voltage = LTC4151_code_to_ADIN_voltage(adin_adc_code, LTC4151_adin_lsb);
300  Serial.print("ADIN Voltage: ");
301  Serial.print(ADIN_voltage);
302  Serial.println(" mV\n");
303  break;
304  default:
305  if (user_command != 'm')
306  Serial.println("Incorrect Option");
307  break;
308  }
309  ack = LTC4151_write(LTC4151_I2C_ADDRESS, LTC4151_CONTROL_REG, LTC4151_CONTINUOUS_MODE); // Disable snapshot mode
310  }
311  while ((user_command != 'm') && (ack == 0));
312  return(ack);
313 }
