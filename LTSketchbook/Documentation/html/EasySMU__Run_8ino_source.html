25 EasySMU is a single-channel ±12V/40mA programmable-voltage/programmable-current
26 source with accurate voltage/current measurement capability. The LTC4316 I2C
27 Address Translator enables up to eight independent EasySMUs to be controlled
28 by a single I2C master.
29 
30 In this demonstration, each EasySMU board contains four I2C slaves and the
31 associated components to implement a single-channel ±12V/40mA programmable-
32 voltage/programmable-current source. The LTC4316 translates the I2C addresses
33 of each EasySMU to a unique set of addresses, enabling up to eight EasySMU
34 boards to be stacked on a single Linduino (I2C master). In this form, it
35 resembles a multi-channel automated test system. Alternatively, an optional
36 touchscreen allows the user to interactively control up to four channels,
37 forming a compact multi-channel programmable-voltage/programmable-current
38 bench source for lab testing, powered from a single 12V AC wall adapter.
39 
40 The primary purpose of the EasySMU is to demonstrate the LTC4316 I2C Address
41 Translator. The programmable-voltage/programmable-current source and meter
42 also provide a convenient demonstration of the associated components: LT1970A,
43 LT5400-3, LTC2655-H, LTC3265, LTC2051, LT3010, LT1991, LTC6655, and LTC2485.
44 While the EasySMU is not designed to demonstrate the ultimate performance that
45 can be obtained from each of those components, the EasySMU does provide
46 impressive results from a reasonably simple circuit.

165 //! Check if first three characters match a three character command in chrCommands.
166 //! @return Index of location in chrCommands of the match. -1 if no match is found.
167 int16_t getToken(char strData[], //!< string to be checked to see if a match is found in chrCommands
168  const char chrCommands[][3], //!< list of valid three character commands
169  int16_t sizeSerialCommands //!< the number of commands in chrCommands
170  )
171 {
172  int16_t i;
173  char *strchr1;
174 
175  for (i=0; i<sizeSerialCommands; i++)
176  {
177  if (chrCommands[i][0]==toupper(strData[0]) && chrCommands[i][1]==toupper(strData[1]) && chrCommands[i][2]==toupper(strData[2]))
178  {
179  if ((strchr1=strchr(strData,':'))==NULL) strchr1=strchr(strData,' '); //returns location of next ':' or ' ' else it returns NULL
180  if (strchr1==NULL) strData[0]='\0';
181  else strcpy(strData,strchr1+1); //Removes token from strData string
182  return i;
183  }
184  }
185  return -1;
186 }

220 Checks for the following valid commands from the serial interface and performs the corresponding actions.
221 
222 Where indicated, 'x' is substituted with 0-7, a number corresponding to the appropriate channel.
223 |Command | Description |
224 |:--------------|:-------------------------|
225 |*RST | Reset to power-up state |
226 |*IDN or *IDN? | Return ID string that includes revision, channels that are present, serial number of each channel, etc. |
227 |LCD:ENA | Enable the LCD/TFT display (default) |
228 |LCD:DIS | Disable the LCD/TFT display. This might be useful to reduce noise at the output caused by SPI communication with the screen. |
229 |CHx:DIS | Disable the output to put it in a relatively high-impedance state. |
230 |CHx:ENA | Enable the channel. (Default state) |
231 |CHx:VOL value | Set the output voltage to 'value' in volts |
232 |CHx:CUR value | Set the output current to 'current' in amps. If current is preceeded by '+' or '-' it will force source or sink only operation. |
233 |CHx:CAL:SET | Calibration routine. (Factory use only. Disabled by default.) |
234 |CHx:CAL:RES | Restore factory calibration. (Only useful if the user has overwritten the calibration, which is not possible without editing the firmware.) |
235 |CHx:MEA:VOL | Returns the measured output voltage in volts. |
236 |CHx:MEA:CUR | Returns the measured output current in amps. |
237 */

488 //! Check if the user is pressing the V+, V++, V-, or V-- buttons. If so, adjust voltage.
489 void SetVoltage()
490 {
491  int16_t step_size;
492  int16_t down_up;
493  int16_t SMUnum;
494 
495  SMUnum=IOpanel.SMUselected_;
496  IOpanel.CheckButton();
497  while ((IOpanel.button_pressed_>=_VPLUS) && (IOpanel.button_pressed_<=_VMINUSMINUS))
498  {
499  if ((IOpanel.button_pressed_==_VPLUS) || (IOpanel.button_pressed_==_VPLUSPLUS)) down_up=-1;
500  else down_up=1;
501  //else return;
502 
503  //Could also just set value based on duration button is pressed.
504  step_size=(IOpanel.duration_button_pressed_ > _BUTTON_TIME_1) ? _VBUTTON_STEP_1 : _VBUTTON_STEP_2;
505  if ((IOpanel.button_pressed_==_VPLUSPLUS) || (IOpanel.button_pressed_==_VMINUSMINUS))
506  {
507  step_size=step_size*16;
508  }
509  step_size*=down_up;
510  delay(10);
511  float flt_old=SMU[SMUnum].fltReadVoltageSourceSetting();
512  SMU[SMUnum].codeStepVoltageSourceSetting(step_size); //Changes Set Voltage by step_size LSB's (positive or negative) while preventing rollover
513  float flt_new=SMU[SMUnum].fltReadVoltageSourceSetting();
514  IOpanel.DisplayVoltageSourceSetting(SMUnum,flt_old,flt_new);
515  while (IOpanel.duration_button_pressed_<_BUTTON_WAIT_TIME && ((IOpanel.button_pressed_>=_VPLUS) && (IOpanel.button_pressed_<=_VMINUSMINUS))) //Allow single button press for LSB increment.
516  {
517  IOpanel.CheckButton();
518  }
519  IOpanel.CheckButton();
520 
521  //Give user _BUTTON_WAIT_TIME to push other button to reverse before changing output voltage.
522  //Use of cast to (int32_t) in comparison should avoid rollover issues according to http://playground.arduino.cc/Code/TimingRollover
523  uint32_t end_time=millis()+_BUTTON_WAIT_TIME;
524  while (((int32_t)(millis()-end_time)<0) && (IOpanel.button_pressed_==_NONE)) //Give user _BUTTON_WAIT_TIME to push other button to reverse before changing output voltage.
525  {
526  IOpanel.CheckButton();
527  }
528  }
529 
530  SMU[SMUnum].CommitVoltageSourceSetting();
531 }
