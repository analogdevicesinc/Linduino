87 // Calculate a LTC26505 DAC code given the desired output voltage and DAC address (0-3)
88 uint16_t LTC2605_voltage_to_code(float dac_voltage, float LTC2605_lsb, int16_t LTC2605_offset)
89 {
90  int32_t dac_code;
91  float float_code;
92  float_code = dac_voltage / LTC2605_lsb; //! 1) Calculate the DAC code
93  float_code = (float_code > (floor(float_code) + 0.5)) ? ceil(float_code) : floor(float_code); //! 2) Round
94  dac_code = (int32_t)float_code - LTC2605_offset; //! 3) Subtract offset
95  if (dac_code < 0) //! 4) If DAC code < 0, Then DAC code = 0
96  dac_code = 0;
97  return ((uint16_t)dac_code); //! 5) Cast DAC code as uint16_t
98 }

100 // Calculate the LTC2605 DAC output voltage given the DAC code and DAC address (0-3)

102 {
103  float dac_voltage;
104  dac_voltage = ((float)(dac_code + LTC2605_offset)* LTC2605_lsb); //! 1) Calculates the dac_voltage
