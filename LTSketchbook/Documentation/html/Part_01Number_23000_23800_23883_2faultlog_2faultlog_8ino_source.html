89 //! Repeats Linduino loop
90 //! @return void
91 void loop()
92 {
93  uint8_t user_command;
94  uint8_t *addresses = NULL;
95  struct LT_3883FaultLog::FaultLogLtc3883 *ltc3883_fault_log;
96 
97 
98  if (Serial.available()) //! Checks for user input
99  {
100  while ((user_command = read_int()) == 0); //! Reads the user command
101  if (user_command != 'm')
102  Serial.println(user_command);
103 
104  uint8_t status;
105 
106  switch (user_command) //! Prints the appropriate submenu
107  {
108  case 1:
109  status = pmbus->readMfrStatusByte(ltc3883_i2c_address);
110  if (status & LTC3883_SMFR_FAULT_LOG)
111  {
112  faultLog3883->read(ltc3883_i2c_address);
113 
114  faultLog3883->dumpBinary(&Serial);
115  Serial.println();
116  faultLog3883->print(&Serial);
117 
118  faultLog3883->release();
119  }
120  else
121  Serial.println(F("No LTC3883 Fault Log"));
122 
123  break;
124  case 2:
125  faultLog3883->clearFaultLog(ltc3883_i2c_address);
126 
127  break;
128  case 3:
129  pmbus->clearAllFaults(ltc3883_i2c_address);
130 
131  break;
132  case 4:
133  pmbus->enablePec(ltc3883_i2c_address);
134  delete faultLog3883;
135  delete smbus;
136  delete pmbus;
137  smbus = new LT_SMBusPec();
138  pmbus = new LT_PMBus(smbus);
139  faultLog3883 = new LT_3883FaultLog(pmbus);
140  break;
141  case 5:
142  pmbus->disablePec(ltc3883_i2c_address);
143  delete faultLog3883;
144  delete smbus;
145  delete pmbus;
146  smbus = new LT_SMBusNoPec();
147  pmbus = new LT_PMBus(smbus);
148  faultLog3883 = new LT_3883FaultLog(pmbus);
149  break;
150  case 6:
151  addresses = smbus->probe(0);
152  while (*addresses != 0)
153  {
154  Serial.print(F("ADDR 0x"));
155  Serial.println(*addresses++, HEX);
156  }
157  break;
158  case 7:
159  pmbus->restoreFromNvmGlobal();
160  delay(2000);
161  pmbus->resetGlobal();
162  break;
163  case 8:
164  smbus->sendByte(ltc3883_i2c_address, MFR_FAULT_LOG_STORE);
165  break;
166  default:
167  Serial.println(F("Incorrect Option"));
168  break;
169  }
170  print_prompt();
171  }
172 }
