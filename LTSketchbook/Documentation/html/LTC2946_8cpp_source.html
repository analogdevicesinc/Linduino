1 /*!
2 LTC2946: 12-Bit Wide Range Power, Charge and Energy Monitor
3 
4 @verbatim
5 
6 The LTC®2946 is a rail-to-rail system monitor that measures
7 current, voltage, power, charge and energy. It features an
8 operating range of 2.7V to 100V and includes a shunt regulator
9 for supplies above 100V. The current measurement common mode
10 range of 0V to 100V is independent of the input supply.
11 A 12-bit ADC measures load current, input voltage and an
12 auxiliary external voltage. Load current and internally
13 calculated power are integrated over an external clock or
14 crystal or internal oscillator time base for charge and energy.
15 An accurate time base allows the LTC2946 to provide measurement
16 accuracy of better than ±0.6% for charge and ±1% for power and
17 energy. Minimum and maximum values are stored and an overrange
18 alert with programmable thresholds minimizes the need for software
19 polling. Data is reported via a standard I2C interface.
20 Shutdown mode reduces power consumption to 15uA.
21 
22 @endverbatim

143 // Reads a 12-bit adc_code from LTC2946
144 int8_t LTC2946_read_12_bits(uint8_t i2c_address, uint8_t adc_command, uint16_t *adc_code)
145 // The function returns the state of the acknowledge bit after the I2C address write. 0=acknowledge, 1=no acknowledge.
146 {
147  // Use union type defined in Linduino.h to combine two uint8_t's (8-bit unsigned integers) into one uint16_t (unsigned 16-bit integer)
148  // Then, shift by 4 bits and return in *adc_code
149  int32_t ack;
150 
151  ack = i2c_read_word_data(i2c_address, adc_command, adc_code);
152 
153  *adc_code >>= 4;
154  return ack;
155 }

235 // Calculate the LTC2946 energy
236 float LTC2946_code_to_energy(int32_t adc_code,float resistor, float LTC2946_Power_lsb, float LTC2946_TIME_lsb)
237 // Returns the LTC2946 energy in Joules
238 {
239  float energy_lsb, energy;
240  energy_lsb=(float)(LTC2946_Power_lsb/resistor)*65536*LTC2946_TIME_lsb; //! 1) Calculate Energy lsb from Power lsb and Time lsb
241  energy = adc_code*energy_lsb; //! 2) Calculate Energy using Energy lsb and adc code
242  return(energy);
