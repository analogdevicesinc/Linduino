2 LTC2442: 24-Bit High Speed 4-Channel Delta Sigma ADCs with Integrated Amplifier
3 
4 @verbatim
5 
6 The LTC2442 is a 4-channel (2-differential) high speed 24-bit No Latency Delta Sigma ADCs.
7 They use a proprietary delta-sigma architecture enabling variable speed/resolution. Through
8 a simple 4-wire serial interface, ten speed/resolution combinations 6.9Hz/280nVRMS to
9 3.5kHz/25uVRMS (4kHz with external oscillator) can be selected with no latency between
10 conversion results or shift in DC accuracy (offset, full-scale, linearity, drift).
11 Additionally, a 2X speed mode can be selected enabling output rates up to 7kHz (8kHz if an
12 external oscillator is used) with one cycle latency.

150 //! Repeats Linduino loop
151 void loop()
152 {
153  static char user_command; // The user input command
154  user_command = read_char(); // Reads the users input
155 // Serial.println(user_command); // Displays the user input
156 
157  switch (user_command)
158  {
159  case 'S':
160  case 's':
161  // Single loop
162  read_LTC2442(LTC2442_vref, EOC_timout, channel_read_delay); // Read all the channels
163  break;
164  case 'C':
165  case 'c':
166  // Continous loop
167  do
168  {
169  read_LTC2442(LTC2442_vref, EOC_timout, channel_read_delay); // Read all the channels
170  delay(loop_delay); // Delay before repeating the loop
171  }
172  while (Serial.available() == false); // Check to see is anything was entered
173  read_int(); // Clears the Serial.available
174  break;
175  default:
176  Serial.println(F("Incorrect Option"));
177  break;
178  }
179 }

192 //! Reads the LTC2442 and displays the results
193 void read_LTC2442(float vref, uint16_t eoc_timeout, uint16_t channel_delay)
194 {
195  float floatresults[16];
196  int32_t rawresults[16];
197  int32_t adc_code = 0;
198  uint8_t i;
199 
200 #ifdef PRIME_BEFORE_SCAN // If defined, prime first channel so that data is "less stale"
201  LTC24XX_EOC_timeout(CS, eoc_timeout); // Wait for end of conversion
202  LTC24XX_SPI_16bit_command_32bit_data(CS, CHANNEL_ARRAY[0], OSR, &adc_code); // Prime with first channel
203 #endif
204 
205  for (i=0; i<array_size; i++)
206  {
207  delay(channel_delay);
208  LTC24XX_EOC_timeout(CS, eoc_timeout); // Wait for end of conversion
209  LTC24XX_SPI_16bit_command_32bit_data(CS, CHANNEL_ARRAY[(i+1)%array_size], OSR, &adc_code); // Read the ADC
210  rawresults[i%array_size] = adc_code;
211  floatresults[i%array_size] = LTC24XX_diff_code_to_voltage(adc_code, vref); // Convert ADC code to voltage
212  }
213 #ifdef FLOATING
214  print_all(floatresults); // Display results
215 #else
216  print_raw(rawresults);
217 #endif
218 }
