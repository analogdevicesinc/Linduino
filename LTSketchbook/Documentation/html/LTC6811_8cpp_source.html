1 /*!
2 LTC6811-1: Multicell Battery Monitor
3 @verbatim
4  The LTC6811 is a 3rd generation multicell battery stack
5  monitor that measures up to 12 series connected battery
6  cells with a total measurement error of less than 1.2mV. The
7  cell measurement range of 0V to 5V makes the LTC6811
8  suitable for most battery chemistries. All 12 cell voltages
9  can be captured in 290uS, and lower data acquisition rates
10  can be selected for high noise reduction.
11 
12  Using the LTC6811-1, multiple devices are connected in
13  a daisy-chain with one host processor connection for all
14  devices.
15 @endverbatim

383 //Helper function to set discharge bit in CFG register
384 void LTC6811_set_discharge(int Cell, uint8_t total_ic, cell_asic ic[])
385 {
386  for (int i=0; i<total_ic; i++)
387  {
388  if (Cell<9)
389  {
390  ic[i].config.tx_data[4] = ic[i].config.tx_data[4] | (1<<(Cell-1));
391  }
392  else if (Cell < 13)
393  {
394  ic[i].config.tx_data[5] = ic[i].config.tx_data[5] | (1<<(Cell-9));
395  }
396  }

427 void LTC6811_max_min(uint8_t total_ic, cell_asic ic_cells[],
428  cell_asic ic_min[],
429  cell_asic ic_max[],
430  cell_asic ic_delta[])
431 {
432  for (int j=0; j < total_ic; j++)
433  {
434  for (int i = 0; i< 12; i++)
435  {
436  if (ic_cells[j].cells.c_codes[i]>ic_max[j].cells.c_codes[i])ic_max[j].cells.c_codes[i]=ic_cells[j].cells.c_codes[i];
437  else if (ic_cells[j].cells.c_codes[i]<ic_min[j].cells.c_codes[i])ic_min[j].cells.c_codes[i]=ic_cells[j].cells.c_codes[i];
438  ic_delta[j].cells.c_codes[i] = ic_max[j].cells.c_codes[i] - ic_min[j].cells.c_codes[i];
439  }
440  }
