1 /*!
2 Linear Technology DC1684AB Demonstration Board.
3 LTC2752: Dual Serial 16-Bit SoftSpan IOUT DAC
4 
5 @verbatim
6 NOTES
7  Setup:
8  Set the terminal baud rate to 115200 and select the newline terminator.
9 
10  An external +/- 15V power supply is required to power the circuit.
11 
12  Explanation of Commands:
13  1- Select DAC
14  Select between DAC A, DAC B, or both.
15 
16  2- Change Span of Selected DAC
17  | Command | Range Selected |
18  | C3 C2 C1 C0 | |
19  |-------------|----------------|
20  | 1 0 0 0 | 0V - 5V |
21  | 1 0 0 1 | 0V - 10V |
22  | 1 0 1 0 | -5V - +5V |
23  | 1 0 1 1 | -10V - +10V |
24  | 1 1 0 0 | -2.5V - +2.5V |
25  | 1 1 0 1 | -2.5V - +7V |
26 
27  3- Voltage Output
28  Displays the calculated voltage depending on the code input from user and
29  voltage range selected.
30 
31  4- Square wave output
32  Generates a square wave on the output pin. This function helps to measure
33  settling time and glitch impulse.
34 
35 USER INPUT DATA FORMAT:
36  decimal : 1024
37  hex : 0x400
38  octal : 02000 (leading 0 "zero")
39  binary : B10000000000
40  float : 1024.0

457 //! Function to generate a square wave of desired frequency and voltage ranges
458 uint8_t menu4_square_wave_output()
459 {
460  uint16_t freq;
461  float time;
462  float voltage_high, voltage_low;
463  uint16_t code_high, code_low;
464  uint8_t receive_enter; // To receive enter key pressed
465 
466  Serial.print("\nEnter voltage_high: ");
467  while (!Serial.available());
468  voltage_high = read_float();
469  Serial.print(voltage_high);
470  Serial.println(" V");
471 
472  Serial.print("\nEnter voltage_low: ");
473  while (!Serial.available());
474  voltage_low = read_float();
475  Serial.print(voltage_low);
476  Serial.println(" V");
477 
478  Serial.print("\nEnter the required frequency in Hz: ");
479  freq = read_int();
480  Serial.print(freq);
481  Serial.println(" Hz");
482 
483  time = (float)1000/freq;
484  Serial.print("\nT = ");
485  Serial.print(time);
486  Serial.println(" ms");
487 
488  //! Converting voltage into data
489  if (DAC_SELECTED == ADDRESS_DACA || DAC_SELECTED == ADDRESS_DAC_ALL)
490  {
491  code_high = LTC2752_voltage_to_code(voltage_high, DACA_RANGE_LOW, DACA_RANGE_HIGH);
492  code_low = LTC2752_voltage_to_code(voltage_low, DACA_RANGE_LOW, DACA_RANGE_HIGH);
493  }
494  if (DAC_SELECTED == ADDRESS_DACB || DAC_SELECTED == ADDRESS_DAC_ALL)
495  {
496  code_high = LTC2752_voltage_to_code(voltage_high, DACB_RANGE_LOW, DACB_RANGE_HIGH);
497  code_low = LTC2752_voltage_to_code(voltage_low, DACB_RANGE_LOW, DACB_RANGE_HIGH);
498  }
499 
500  while (!Serial.available()) //! Generate square wave until a key is pressed
501  {
502  LTC2752_write(LTC2752_CS, LTC2752_WRITE_CODE_UPDATE_DAC, DAC_SELECTED, code_high);
503  delayMicroseconds(time * 500);
504  LTC2752_write(LTC2752_CS, LTC2752_WRITE_CODE_UPDATE_DAC, DAC_SELECTED, code_low);
505  delayMicroseconds(time * 500);
506  }
507  receive_enter = read_int();
508  return 0;
509 }
