85 int8_t LTC24XX_EOC_timeout(uint8_t cs, uint16_t miso_timeout)
86 // Checks for EOC with a specified timeout (ms)
87 {
88  uint16_t timer_count = 0; // Timer count for MISO
89  output_low(cs); //! 1) Pull CS low
90  while (1) //! 2) Wait for SDO (MISO) to go low
91  {
92  if (input(MISO) == 0) break; //! 3) If SDO is low, break loop
93  if (timer_count++>miso_timeout) // If timeout, return 1 (failure)
94  {
95  output_high(cs); // Pull CS high
96  return(1);
97  }
98  else
99  delay(1);
100  }
101  output_high(cs); // Pull CS high
102  return(0);
103 }

271 int8_t LTC24XX_I2C_8bit_command_24bit_data(uint8_t i2c_address, uint8_t adc_command, int32_t *adc_code, uint16_t eoc_timeout)
272 {
273  int8_t ack;
274  uint16_t timer_count = 0; // Timer count to wait for ACK
275  int8_t buf[4];
276  LT_union_int32_4bytes data; // LTC24XX data
277  while (1)
278  {
279  ack = i2c_read_block_data(i2c_address, adc_command, 3, data.LT_byte);
280  if (!ack) break; // !ack indicates success
281  if (timer_count++>eoc_timeout) // If timeout, return 1 (failure)
282  return(1);
283  else
284  delay(1);
285  }
286  data.LT_byte[3] = data.LT_byte[2]; // Shift bytes up by one. We read out 24 bits,
287  data.LT_byte[2] = data.LT_byte[1]; // which are loaded into bytes 2,1,0. Need to left-
288  data.LT_byte[1] = data.LT_byte[0]; // justify.
289  data.LT_byte[0] = 0x00;
290  data.LT_uint32 >>= 2; // Shifts data 2 bits to the right; operating on unsigned member shifts in zeros.
291  data.LT_byte[3] = data.LT_byte[3] & 0x3F; // Clear upper 2 bits JUST IN CASE. Now the data format matches the SPI parts.
292  *adc_code = data.LT_int32;

377 // Calculates the voltage corresponding to an adc code, given the reference voltage (in volts)
378 float LTC24XX_SE_code_to_voltage(int32_t adc_code, float vref)
379 {
380  float voltage;
381  adc_code -= 0x20000000; //! 1) Subtract offset
382  voltage=(float) adc_code;
383  voltage = voltage / 268435456.0; //! 2) This calculates the input as a fraction of the reference voltage (dimensionless)
384  voltage = voltage * vref; //! 3) Multiply fraction by Vref to get the actual voltage at the input (in volts)
385  return(voltage);

389 // This function handles all differential input parts, including the "single-ended" mode on multichannel
390 // differential parts. Data from I2C parts must be right-shifted by two bit positions such that the MSB
391 // is in bit 28 (the same as the SPI parts.)

403  adc_code -= 0x20000000; //! 1) Converts offset binary to binary

405  voltage = voltage / 536870912.0; //! 2) This calculates the input as a fraction of the reference voltage (dimensionless)

407  return(voltage);

429 // Calculate the lsb weight and offset code given a full-scale code and a measured zero-code.
430 void LTC24XX_calibrate_voltage(int32_t zero_code, int32_t fs_code, float zero_voltage, float fs_voltage, float *LTC24XX_lsb, int32_t *LTC24XX_offset_code)
431 {
432  zero_code -= 536870912; //! 1) Converts zero code from offset binary to binary
433  fs_code -= 536870912; //! 2) Converts full scale code from offset binary to binary
434 
435  float temp_offset;
436  *LTC24XX_lsb = (fs_voltage-zero_voltage)/((float)(fs_code - zero_code)); //! 3) Calculate the LSB
437 
438  temp_offset = (zero_voltage/ *LTC24XX_lsb) - zero_code; //! 4) Calculate Unipolar offset
439  temp_offset = (temp_offset > (floor(temp_offset) + 0.5)) ? ceil(temp_offset) : floor(temp_offset); //! 5) Round
440  *LTC24XX_offset_code = (int32_t)temp_offset; //! 6) Cast as int32_t
441 }
