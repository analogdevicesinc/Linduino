148 //! Set DAC's From -10V to 10V, Reads DAC Channels Through MUX, and Displays The Results
149 void demo_board_test()
150 {
151  int32_t adc_code[16];
152  float voltage[16];
153  float ideal_voltage;
154  float full_scale = 10.24;
155 
156  LTC2668_write(LTC2668_CS,LTC2668_CMD_SPAN_ALL, 0, LTC2668_SPAN_PLUS_MINUS_10V); // Set DAC's To +-10V
157 
158  // Set Channels From -10v to 10V Incrementing By Channel.
159  for (uint8_t i = 0; i <= 15; ++i)
160  {
161  LTC2668_write(LTC2668_CS,LTC2668_CMD_WRITE_N_UPDATE_N, i, i * 4369);
162  }
163 
164  LTC2668_write(LTC2668_CS,LTC2668_CMD_MUX, 0, LTC2668_MUX_ENABLE | 0); // Set To Channel 0
165  LTC2378_read(LTC2378_CS, &adc_code[0]); // Discard reading
166 
167  // Cycle Through the Channels of the MUX and Reads the MUX Voltage
168  for (uint8_t i = 1; i <= 15; ++i)
169  {
170  LTC2668_write(LTC2668_CS,LTC2668_CMD_MUX, 0, LTC2668_MUX_ENABLE | i);
171  LTC2378_read(LTC2378_CS, &adc_code[i-1]);
172  }
173  LTC2378_read(LTC2378_CS, &adc_code[15]);
174 
175  // Display The Results
176  for (uint8_t i = 0; i<=15; ++i)
177  {
178  voltage[i] = LTC2378_code_to_voltage(adc_code[i], full_scale);
179  ideal_voltage = LTC2668_code_to_voltage(i * 4369, -10.0, 10.0);
180  Serial.print(F("Channel "));
181  Serial.print(i);
182  if (i < 10)
183  Serial.print(" ");
184  Serial.print(F(" : Ideal Voltage - "));
185  if (ideal_voltage >= 0)
186  Serial.print(" ");
187  Serial.print(ideal_voltage, 6);
188  if ((ideal_voltage < 10.0) && (ideal_voltage > -10.0))
189  Serial.print(" ");
190  Serial.print(F(" V Measured Voltage - "));
191  if (voltage[i] >= 0)
192  Serial.print(" ");
193  Serial.print(voltage[i],6);
194  if ((voltage[i] < 10.0) && (voltage[i] > -10.0))
195  Serial.print(" ");
196  Serial.print(F(" V difference - "));
197  if ((ideal_voltage-voltage[i]) >= 0)
198  Serial.print(" ");
199  Serial.print((ideal_voltage-voltage[i]),6);
200  Serial.println(" V");
201  }
202 
203  Serial.println();
204  Serial.println("Test Complete\n");
205 }
