205 //! LTC2947's operation control menu
206 void mm1_operation_control()
207 {
208  Serial.println(F("\n\nOperation control"));
209  Serial.println(F(" 1. Continuous measurement"));
210  Serial.println(F(" 2. Single measurement"));
211  Serial.println(F(" 3. Clear measurement results"));
212  Serial.println(F(" 4. Stop measurement"));
213  Serial.println(F(" 5. Shutdown"));
214  Serial.println(F(" 6. Wakeup"));
215  Serial.println(F(" 7. Read current operation mode"));
216  Serial.println(F(" 8. RESET"));
217  Serial.println(F(" Any other number to cancel"));
218  Serial.print(F("Enter: "));
219 
220  while (!Serial.available()); //! wait for user input
221  int32_t opt = read_int(); //! Read user input command
222  Serial.println(opt); //! echo user input
223  switch (opt)
224  {
225  case 1: //! Continuous measurement
226  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, LTC2947_BM_OPCTL_CONT);
227  break;
228  case 2: //! Single measurement
229  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, LTC2947_BM_OPCTL_SSHOT);
230  break;
231  case 3: //! Clear measurement results
232  {
233  byte data[1];
234  LTC2947_RD_BYTE(LTC2947_REG_OPCTL, data); //! store current operation mode
235  LTC2947_WR_BYTE(LTC2947_REG_OPCTL,
236  LTC2947_BM_OPCTL_CLR | //! set clear bit
237  (data[0] & LTC2947_BM_OPCTL_CONT) //! restore last operation mode (idle or cont. meas.)
238  );
239  }
240  break;
241  case 4: //! Stop measurement
242  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, 0);
243  break;
244  case 5: //! Shutdown
245  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, LTC2947_BM_OPCTL_SHDN);
246  break;
247  case 6: //! Wakeup
248  {
249  int16_t wakeupTime = LTC2947_wake_up();
250  if (wakeupTime < 0)
251  Serial.println(F("Failed to wakeup LTC2947 due to timeout"));
252  else
253  {
254  Serial.print(F("LTC2947 woke up after "));
255  Serial.print(wakeupTime);
256  Serial.println(F(" milliseconds"));
257  }
258  }
259  break;
260  case 7: //! Read current operation mode
261  {
262  byte data[1];
263  LTC2947_RD_BYTE(LTC2947_REG_OPCTL, data);
264  if (bitMaskSetChk(data[0], LTC2947_BM_OPCTL_CONT))
265  Serial.println(F("Continuous measurement enabled"));
266  else if (bitMaskSetChk(data[0], LTC2947_BM_OPCTL_SHDN))
267  Serial.println(F("LTC2947 was in shutdown mode and should\nhave been woken up by this read operation"));
268  else
269  Serial.println(F("Idle mode"));
270  }
271  break;
272  case 8: //! make a reset of LTC2947
273  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, LTC2947_BM_OPCTL_RST);
274  delay(120); //! give device some time to restart
275  go_idle_cfg_tbctl(); //! re-configure TBCTL
276  break;
277  default:
278  Serial.println(F("Unsupported command, canceled!"));
279  break;
280  }
281 }

349 //! Initialize accumulators with user defined values
350 void mm5_accu_init()
351 {
352  boolean accuSet1;
353  byte bytes[16]; //! 16 bytes to store charge, energy and time
354  uint8_t sel;
355  boolean lsbs;
356  double cetb;
357  uint8_t accu_base_addr;
358 
359  //! Which ACCUs? (C1,E1,TB1 or C2,E2,TB2)
360  Serial.println(F("\nInit Accumulator set 1 or 2?"));
361  Serial.print(F("Enter: "));
362  while (!Serial.available());
363  sel = read_int();
364  accuSet1 = (sel == 1);
365  if (accuSet1)
366  Serial.println(F("1 (C1,E1,TB1)"));
367  else if (sel == 2)
368  Serial.println(F("2 (C2,E2,TB2)"));
369  else
370  Serial.println(F("Unknown selection, will use C2,E2,TB2"));
371 
372  //! assign base register address of the accu set
373  accu_base_addr = accuSet1 ? LTC2947_VAL_C1 : LTC2947_VAL_C2;
374 
375  //! Set accus in LSBs or in real values?
376  Serial.println(F("\nInput LSBs or real values in As/Ws/s?"));
377  Serial.println(F(" 1. LSBs"));
378  Serial.println(F(" 2. Real Values in As/Ws/s"));
379  Serial.print(F("Enter: "));
380  while (!Serial.available());
381  sel = read_int();
382  lsbs = (sel == 1);
383  if (lsbs)
384  Serial.println(F("1 (LSBs)"));
385  else if (sel == 2)
386  Serial.println(F("2 (As/Ws/s)"));
387  else
388  Serial.println(F("Unknown selection, will use 1 (LSBs)"));
389 
390  //! Input new charge value
391  Serial.println();
392  Serial.print(accuSet1 ? F("C1: ") : F("C2: "));
393  while (!Serial.available());
394  if (lsbs)
395  //! convert integer input to floating point value
396  //! Note: actually this is not necessary as we have to
397  //! write integer values to the device anyway. So
398  //! we could convert this int value directly to
399  //! a bytes array and write it to the device instead of first going to float
400  //! and then going back to an integer representation
401  //! in form of a byte array. Still the focus of the code below
402  //! is to show the usage of the function LTC2947_DoubleToBytes
403  cetb = read_int() * LTC2947_LSB_C1;
404  else
405  cetb = read_float();
406  Serial.print(cetb, 8);
407  Serial.println(F(" As"));
408  LTC2947_DoubleToBytes(cetb, LTC2947_LSB_C1, bytes, 6); //! convert to bytes
409 
410  //! Input new energy value
411  Serial.println();
412  Serial.print(accuSet1 ? F("E1: ") : F("E2: "));
413  while (!Serial.available());
414  if (lsbs)
415  cetb = read_int() * LTC2947_LSB_E1;
416  else
417  cetb = read_float();
418  Serial.print(cetb, 8);
419  Serial.println(F(" Ws"));
420  LTC2947_DoubleToBytes(cetb, LTC2947_LSB_E1, bytes + 6, 6); //! convert to bytes
421 
422  //! Input new time value
423  Serial.println();
424  Serial.print(accuSet1 ? F("TB1: ") : F("TB2: "));
425  while (!Serial.available());
426  if (lsbs)
427  cetb = read_int() * LTC2947_LSB_TB1;
428  else
429  cetb = read_float();
430  Serial.print(cetb, 8);
431  Serial.println(F(" s"));
432  LTC2947_DoubleToBytes(cetb, LTC2947_LSB_TB1, bytes + 6 + 6, 4); //! convert to bytes
433 
434  Serial.print(F("\nC,E,TB will be written with following values starting at address 0x"));
435  LTC2947_SerialPrint8hex(accu_base_addr);
436  Serial.println();
437 
438  //! report the whole byte array
439  for (int8_t i = 0; i < 16; i++)
440  {
441  Serial.print(i);
442  Serial.print(F(":0x"));
443  LTC2947_SerialPrint8hex(bytes[i]);
444  if (i == 0)
445  Serial.print(F(" (MSB)"));
446  else if (i == 15)
447  Serial.print(F(" (LSB)"));
448  Serial.println();
449  }
450 
451  LTC2947_SetPageSelect(false); //! make sure we are on page 0
452  LTC2947_WR_BYTES(accu_base_addr, 16, bytes); //! write new values to the device
453 
454  //! readback values from the device
455  Serial.println(F("\nRead back of both accu sets from device:"));
456  mm4_accu_measurements();
457 }

459 //! Read tracking (min/max) measurement values
460 void mm6_min_max()
461 {
462  float min, max;
463  byte bytes[8];
464  Serial.println(F("\nMIN/MAX measurements"));
465 
466  //! read current/power max/min from device
467  LTC2947_RD_BYTES(LTC2947_VAL_IMAX, 8, bytes);
468 
469  //! current max/min
470  //! LSB is given in mA, *1e-3 gives A. Note also the different LSB size of
471  //! LTC2947_LSB_IMAX (or LTC2947_LSB_IMIN which is the same) compared to LTC2947_LSB_I!
472  Serial.print(F("Current(A) "));
473  max = LTC2947_2BytesToInt16(bytes + 0) * LTC2947_LSB_IMAX * 1e-3;
474  min = LTC2947_2BytesToInt16(bytes + 2) * LTC2947_LSB_IMIN * 1e-3;
475  print_min_max(min, max, 3, bytes + 2, bytes + 0);
476 
477  Serial.print(F("Power(W) "));
478  max = LTC2947_2BytesToInt16(bytes + 4) * LTC2947_LSB_PMAX * 1e-3;
479  min = LTC2947_2BytesToInt16(bytes + 6) * LTC2947_LSB_PMIN * 1e-3;
480  print_min_max(min, max, 3, bytes + 6, bytes + 4);
481 
482  //! read voltage/temperature max/min from device
483  LTC2947_RD_BYTES(LTC2947_VAL_VMAX, 8, bytes);
484 
485  Serial.print(F("Voltage(V) "));
486  max = LTC2947_2BytesToInt16(bytes + 0) * LTC2947_LSB_VMAX * 1e-3;
487  min = LTC2947_2BytesToInt16(bytes + 2) * LTC2947_LSB_VMIN * 1e-3;
488  print_min_max(min, max, 3, bytes + 2, bytes + 0);
489 
490  Serial.print(F("Temp.(C) "));
491  max = LTC2947_2BytesToInt16(bytes + 4) * LTC2947_LSB_TEMPMAX + LTC2947_OFFS_TEMPMAX;
492  min = LTC2947_2BytesToInt16(bytes + 6) * LTC2947_LSB_TEMPMIN + LTC2947_OFFS_TEMPMIN;
493  print_min_max(min, max, 2, bytes + 6, bytes + 4);
494 }

598 //! put LTC2947 into idle mode and configure TBCTL
599 void go_idle_cfg_tbctl()
600 {
601  byte data[1];
602  byte tbctl;
603  LTC2947_RD_BYTE(LTC2947_REG_OPCTL, data); //! check if we are in idle mode
604  if (data[0] != 0) //! not in idle mode
605  {
606  Serial.println(F("Will set LTC2947 to IDLE mode now!"));
607  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, 0);//! go to idle mode
608  delay(200); //! after 2 system tickes (2x100ms) LTC2947 must be in idle mode
609  LTC2947_RD_BYTE(LTC2947_REG_OPCTL, data);
610  }
611  if (data[0] == 0)
612  {
613  Serial.println(F("LTC2947 in IDLE mode now!"));
614  }
615  else
616  {
617  Serial.println(F("Unable to set LTC2947 to IDLE mode!"));
618  }
619 
620  LTC2947_SetPageSelect(false); //! make sure we are on page 0
621 
622  tbctl = //! generate TBCTL configuration value
