112 //! Repeats Linduino loop
113 void loop()
114 {
115  int8_t ack=0;
116  uint8_t user_command;
117  if (demo_board_connected) //! Does nothing if the demo board is not connected
118  {
119  if (Serial.available()) //! Checks for user input
120  {
121  user_command = read_int(); //! Reads the user command
122  if (user_command != 'm')
123  Serial.println(user_command);
124  ack = 0;
125  switch (user_command) //! Prints the appropriate submenu
126  {
127  case 1:
128  ack |= menu_1_read_write_registers(); // Print single-ended voltage menu
129  break;
130  case 2:
131  ack |= menu_2_regulator_settings(); // Differential voltage menu
132  break;
133  case 3:
134  ack |= menu_3_control_settings(); // Control Settings menu
135  break;
136  case 4:
137  ack |= menu_4_sequencing(reg_phase); // Sequencing menu
138  break;
139  default:
140  Serial.println("Incorrect Option");
141  break;
142  }
143  if (ack != 0)
144  {
145  Serial.print(F("Error: No Acknowledge. \n"));
146  }
147  print_prompt();
148  }
149  }
150 }

185 int8_t LTC3676_print_all_registers(uint8_t i2c_address)
186 {
187  int8_t ack = 0;
188  uint8_t user_register = 0x01;
189  uint8_t data;
190  do
191  {
192  ack |= LTC3676_register_read(i2c_address, user_register, &data); //! Read register
193  Serial.print("Register 0x");
194  Serial.print(user_register, HEX);
195  Serial.print(":\t0x");
196  Serial.println(data, HEX);
197  user_register++;
198  }
199  while ((user_register <= 0x17) && (ack != 1));
200  return ack;
201 }

215 int8_t menu_1_read_write_registers()
216 {
217  int8_t ack=0;
218  uint8_t user_command;
219  uint8_t data;
220  uint8_t user_register;
221  uint8_t user_bit;
222  do
223  {
224  //! Displays the Read/Write Registers menu
225  Serial.print(F("\nRead/Write Registers\n\n"));
226  Serial.print(F(" 1-Read All Registers\n"));
227  Serial.print(F(" 2-Read Single Register\n"));
228  Serial.print(F(" 3-Write Single Register\n"));
229  Serial.print(F(" 4-Set Bit\n"));
230  Serial.print(F(" 5-Clear Bit\n"));
231  Serial.print(F(" 6-Clear IRQ\n"));
232  Serial.print(F(" 7-Send Hard Reset Command\n"));
233  Serial.print(F(" m-Main Menu\n"));
234  Serial.print(F("\nEnter a command: "));
235 
236  user_command = read_int(); //! Reads the user command
237  if (user_command == 'm') // Print m if it is entered
238  {
239  Serial.print(F("m\n"));
240  }
241  else
242  Serial.println(user_command); // Print user command
243 
244  //! Reads or writes to a LTC3676 and prints result.
245  switch (user_command)
246  {
247  case 1:
248  // Read every register and print it's data.
249  ack |= LTC3676_print_all_registers(i2c_address);
250  break;
251  case 2:
252  // Read a single register and print it's data.
253  Serial.print(F("\nAddress (in hex with '0x' prefix) of register to read: "));
254  user_register = read_int();
255  Serial.print("0x");
256  Serial.println(user_register, HEX);
257  if (user_register < 0x01 || user_register > 0x17)
258  {
259  Serial.println(F(" Invalid input."));
260  break;
261  }
262  ack |= LTC3676_register_read(i2c_address, user_register, &data);
263  Serial.print("Register data: 0x");
264  Serial.println(data, HEX);
265  break;
266  case 3:
267  // Write a byte to a chosen register.
268  Serial.print(F("\nAddress (in hex with '0x' prefix) of register to write: "));
269  user_register = read_int();
270  Serial.print("0x");
271  Serial.println(user_register, HEX);
272  if (user_register == 0x1E)
273  {
274  Serial.println(F(" Hard Reset command sent."));
275  ack |= LTC3676_register_write(i2c_address, user_register, 0x01);
276  break;
277  }
278  else if (user_register == 0x1F)
279  {
280  Serial.println(F(" Clear IRQ command sent."));
281  ack |= LTC3676_register_write(i2c_address, user_register, 0x01);
282  break;
283  }
284  else if (user_register < 0x01 || user_register > 0x17)
285  {
286  Serial.println(F(" Invalid input."));
287  break;
288  }
289  Serial.print(F("Data (in hex with '0x' prefix) to write: "));
290  data = read_int();
291  Serial.println(data, HEX);
292  ack |= LTC3676_register_write(i2c_address, user_register, data);
293  Serial.print("0x");
294  Serial.print(data, HEX);
295  Serial.print(" written to register 0x");
296  Serial.println(user_register, HEX);
297  break;
298  case 4:
299  // Set a single bit within a chosen register.
300  Serial.print(F("\nAddress (in hex with '0x' prefix) of register: "));
301  user_register = read_int();
302  Serial.print("0x");
303  Serial.println(user_register, HEX);
304  if (user_register < 0x01 || user_register > 0x17)
305  {
306  Serial.println(F(" Invalid input."));
307  break;
308  }
309  Serial.print(F("Bit position (0-7) to set: "));
310  data = read_int();
311  if (data < 0 || data > 7)
312  {
313  Serial.println(F(" Invalid input."));
314  break;
315  }
316  Serial.println(data, DEC);
317  ack |= LTC3676_bit_set(i2c_address, user_register, data);
318  Serial.print("Bit set. Register data is now 0x");
319  ack |= LTC3676_register_read(i2c_address, user_register, &data);
320  Serial.println(data, HEX);
321  break;
322  case 5:
323  // Clear a single bit within a chosen register.
324  Serial.print(F("\nAddress (in hex with '0x' prefix) of register: "));
325  user_register = read_int();
326  Serial.print("0x");
327  Serial.println(user_register, HEX);
328  if (user_register < 0x01 || user_register > 0x17)
329  {
330  Serial.println(F(" Invalid input."));
331  break;
332  }
333  Serial.print(F("Bit position (0-7) to clear: "));
334  data = read_int();
335  if (data < 0 || data > 7)
336  {
337  Serial.println(F("Invalid input."));
338  break;
339  }
340  Serial.println(data, DEC);
341  ack |= LTC3676_bit_clear(i2c_address, user_register, data);
342  Serial.print("Bit cleared. Register data is now 0x");
343  ack |= LTC3676_register_read(i2c_address, user_register, &data);
344  Serial.println(data, HEX);
345  break;
346  case 6:
347  // Clear IRQ register
348  Serial.println(F("\n Clear IRQ command sent."));
349  ack |= LTC3676_register_write(i2c_address, LTC3676_REG_CLIRQ, 0x01);
350  break;
351  case 7:
352  // Initiate a hard reset
353  Serial.println(F("\n Hard Reset command sent."));
354  ack |= LTC3676_register_write(i2c_address, LTC3676_REG_HRST, 0x01);
355  break;
356  default:
357  if (user_command != 'm')
358  Serial.println("Invalid Selection");
359  break;
360  }
361  }
362  while ((user_command != 'm') && (ack != 1));
363  return(ack);
364 }

366 int8_t menu_2_regulator_settings()
367 {
368  int8_t ack=0;
369  uint8_t user_command;
370  uint8_t data;
371  uint8_t user_register;
372  uint8_t user_int;
373  uint8_t user_buck;
374  uint8_t user_ldo;
375  int8_t user_char;
376  do
377  {
378  //! Displays the Regulator Settings menu
379  Serial.print(F("\nRegulator Settings\n\n"));
380  Serial.print(F(" 1-Enable/Disable Bucks\n"));
381  Serial.print(F(" 2-Enable/Disable LDOs\n"));
382  Serial.print(F(" 3-Set Buck Output Voltage\n"));
383  Serial.print(F(" 4-Set Buck Feedback Reference\n"));
384  Serial.print(F(" 5-Select Buck Reference\n"));
385  Serial.print(F(" 6-Set Buck Switching Mode\n"));
386  Serial.print(F(" 7-Set Start-Up Mode (300mV check)\n"));
387  Serial.print(F(" 8-Set PGOOD Slewing Mask\n"));
388  Serial.print(F(" 9-Exit Software Control Mode\n"));
389  if (board_option == 'B')
390  Serial.print(F(" 10-Set LDO4 Output Voltage\n"));
391  Serial.print(F(" m-Main Menu\n"));
392  Serial.print(F("\nEnter a command: "));
393 
394  user_command = read_int(); //! Reads the user command
395  if (user_command == 'm') // Print m if it is entered
396  Serial.print(F("m\n"));
397  else
398  Serial.println(user_command); // Print user command
399 
400  //! Reads or writes to a LTC3676 and prints result.
401  switch (user_command)
402  {
403  case 1:
404  //Enter software control mode
405  if (!LTC3676_bit_is_set(i2c_address, LTC3676_REG_CNTRL, LTC3676_SOFTWARE_CNTRL))
406  {
407  Serial.print(F("\n********** Note: LTC3676 is now in Sofware Control Mode **********\n"));
408  Serial.print(F("***************Select Option 9 to resume Pin Control *************\n"));
409  ack |= LTC3676_bit_set(i2c_address, LTC3676_REG_CNTRL, LTC3676_SOFTWARE_CNTRL);
410  }
411  // Enable/Disable Bucks
412  Serial.print(F("\nSelect Buck(1-4, 5=all): "));
413  user_buck = read_int();
414  Serial.println(user_buck, DEC);
415  if (user_buck < 1 || user_buck > 5)
416  {
417  Serial.println(F(" Invalid input."));
418  break;
419  }
420  Serial.println(F("0=Disable, 1=Enable"));
421  Serial.print(F("Enter selection: "));
422  user_int = read_int();
423  Serial.println(user_int, DEC);
424  if (user_int > 1)
425  {
426  Serial.println(F(" Invalid input."));
427  break;
428  }
429  if (user_int == 1)
430  ack |= LTC3676_bit_set(i2c_address, LTC3676_REG_CNTRL, LTC3676_PWR_ON);
431  if (user_buck == 5)
432  {
433  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_BUCK1, 7, user_int);
434  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_BUCK2, 7, user_int);
435  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_BUCK3, 7, user_int);
436  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_BUCK4, 7, user_int);
437  }
438  else
439  ack |= LTC3676_bit_write(i2c_address, user_buck, 7, user_int);
440  Serial.println(F("Done."));
441  break;
442  case 2:
443  //Enter software control mode
444  if (!LTC3676_bit_is_set(i2c_address, LTC3676_REG_CNTRL, LTC3676_SOFTWARE_CNTRL))
445  {
446  Serial.print(F("\n********** Note: LTC3676 is now in Sofware Control Mode **********\n"));
447  Serial.print(F("***************Select Option 8 to resume Pin Control *************\n"));
448  ack |= LTC3676_bit_set(i2c_address, LTC3676_REG_CNTRL, LTC3676_SOFTWARE_CNTRL);
449  }
450  // Enable/Disable LDOs
451  Serial.print(F("\nSelect LDO(2-4, 5=all): "));
452  user_ldo = read_int();
453  Serial.println(user_ldo, DEC);
454  if (user_ldo < 2 || user_ldo > 5)
455  {
456  Serial.println(F(" Invalid input."));
457  break;
458  }
459  Serial.println(F("0=Disable, 1=Enable"));
460  Serial.print(F("Enter selection: "));
461  user_int = read_int();
462  Serial.println(user_int, DEC);
463  if (user_int > 1)
464  {
465  Serial.println(F(" Invalid input."));
466  break;
467  }
468  if (user_int == 1)
469  ack |= LTC3676_bit_set(i2c_address, LTC3676_REG_CNTRL, LTC3676_PWR_ON);
470  if (user_ldo == 5)
471  {
472  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_LDOA, 2, user_int);
473  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_LDOA, 5, user_int);
474  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_LDOB, 2, user_int);
475  }
476  else if (user_ldo == 2)
477  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_LDOA, 2, user_int);
478  else if (user_ldo == 3)
479  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_LDOA, 5, user_int);
480  else if (user_ldo == 4)
481  ack |= LTC3676_bit_write(i2c_address, LTC3676_REG_LDOB, 2, user_int);
482  Serial.println(F("Done."));
483  break;
484  case 3:
485  // Set buck output voltage in mV
486  float user_output;
487  float new_output;
488  Serial.print(F("\nSelect Buck(1-4): "));
489  user_int = read_int();
490  Serial.println(user_int, DEC);
491  if (user_int < 1 || user_int > 4)
492  {
493  Serial.println(F("Invalid input."));
494  break;
495  }
496  Serial.print(F("Select Reference (A=DVB"));
497  Serial.print(user_int, DEC);
498  Serial.print(F("A, B=DVB"));
499  Serial.print(user_int, DEC);
500  Serial.print(F("B): "));
501  user_char = read_char();
502  Serial.write(user_char);
503  if (user_char == 'A' || user_char == 'a')
504  user_register = (user_int*2) + 8; //Converts selection to appropriate register address.
505  else if (user_char == 'B' || user_char == 'b')
506  user_register = (user_int*2) + 9; //Converts selection of appropriate register address.
507  else
508  {
509  Serial.println(F("\nInvalid input."));
510  break;
511  }
512  Serial.print(F("\nPotential output voltage range: "));
513  Serial.print(LTC3676_buck_vout_min(user_int), 0);

530  ack |= LTC3676_register_read(i2c_address, user_register, &data);
531  Serial.println(data & 0x1F, HEX);
532  break;

552  if (user_char == 'A' || user_char == 'a')
