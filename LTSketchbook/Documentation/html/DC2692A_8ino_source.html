1 /*!
2 Linear Technology DC2692A Demonstration Board.
3 LTC2662: 5-Channel 16-/12-Bit 300mA SoftSpan Current Source DACs
4 
5 @verbatim
6 NOTES
7  Setup:
8  Set the terminal baud rate to 115200 and select the newline terminator.
9 
10  An external +7V - 5V power supply is required to power the circuit.
11 
12  The program displays calculated currents which are based on the voltage
13  of the reference used and Radj resistor, be it internal or external. A
14  precision voltmeter is needed to verify the actual measured voltages
15  against the calculated current displayed.
16 
17  If an external reference is used, a precision voltage
18  source is required to apply the external reference voltage. A
19  precision voltmeter is also required to measure the external reference
20  voltage.
21 
22  Explanation of Commands:
23  1- Select DAC: Select one of five DACs to test : 0 to 4
24 
25  2- Write to DAC input register: Value is stored in the DAC for updating
26  later, allowing multiple channels to be updated at once, either
27  through a software "Update All" command or by asserting the LDAC# pin.
28  User will be prompted to enter either a code in hex or decimal, or a
29  current. If a current is entered, a code will be calculated based on
30  the active scaling and reference parameters - ideal values if no
31  calibration was ever stored.
32 
33  3- Write and Update: Similar to item 1, but DAC is updated immediately.
34 
35  4- Update DAC: Copies the value from the input register into the DAC
36  Register. Note that a "write and update" command writes the code to
37  BOTH the input register and DAC register, so subsequent "update"
38  commands will simply re-copy the same data (no change in output.)
39 
40  5- Power Down DAC: Disable DAC output. Power supply current is reduced.
41  DAC code present in DAC registers at time of shutdown are preserved.
42 
43  6- Set reference mode, either internal or external: Selecting external
44  mode powers down the internal reference voltage. It is the users
45  responsibility to apply a 1.25v reference.
46 
47  7- Set SoftSpan: There are ten options.
48 
49  8- Toggle Selected Word: Switch between register A or B for DAC code.
50 
51  9- Set MUX: Sets the channel.
52 
53  10- Global Toggle Bit Settings - Enabling this feature sets the DAC to
54  toggle from register A(when TGL pin is LOW) and register b(when TGL pin
55  is HIGH). TGL pin is set HIGH with an internal pull up when the global toggle bit
56  is set, and TGL pin is set LOW with an internal pull down when the global toggle bit
57  is not set.
58 
59  11- Enable, disable, or store to EEPROM: To store to EEROM ensure all
60  registers are set to their desired settings. Next, go to th store
61  setting EEPROM menu and select it. Upon Linduinos power up, the
62  previously stored settings will be restored.
63 
64 USER INPUT DATA FORMAT:
65  decimal : 1024
66  hex : 0x400
67  octal : 02000 (leading 0 "zero")
68  binary : B10000000000
69  float : 1024.0
70 
71 @endverbatim

212 {

214  uint8_t fault_reg = 0;

239  break;
240  case 6:
241  fault_reg |= menu_6_set_reference_mode();
242  break;
243  case 7:
244  fault_reg |= menu_7_set_softspan_range(selected_dac);
245  break;
246  case 8:
247  fault_reg |= menu_8_toggle_select_word();
248  break;
249  case 9:
250  fault_reg |= menu_9_set_mux();
251  break;
252  case 10:
253  fault_reg |= menu_10_global_toggle_settings();
254  break;
255  case 11:
256  menu_11_enable_disable_eeprom_restore();
257  break;
258  case 14:
259  menu_14_demo_board_test();

286 //! Write data to input register, but do not update DAC output
287 //! @return fault code
288 uint8_t menu_2_write_to_input_register(uint8_t selected_dac)
289 {
290  uint16_t dac_code;
291  uint8_t fault_reg = 0;
292 
293  if (prompt_current_or_code(selected_dac) == PROMPT_CURRENT)
294  dac_code = get_current(selected_dac);
295  else
296  dac_code = get_code();
297 
298  if (selected_dac <= 4)
299  {
300  // Store dac_code to register variable a or b
301  if (eeprom.data_struct.toggle_word & (0x01 << selected_dac))
302  {
303  eeprom.data_struct.dac_code_b[selected_dac] = dac_code;
304  }
305  else
306  eeprom.data_struct.dac_code_a[selected_dac] = dac_code;
307  fault_reg |= LTC2662_write(LTC2662_CS, LTC2662_CMD_WRITE_N, selected_dac, dac_code);
308  }
309  else
310  {
311  // Store dac_code to register variable a or b
312  for (uint8_t i = 0; i <= 4 ; i++)
313  {
314  if (eeprom.data_struct.toggle_word & (0x01 << i))
315  {
316  eeprom.data_struct.dac_code_b[i] = dac_code;
317  }
318  else
319  eeprom.data_struct.dac_code_a[i] = dac_code;
320  }
321  fault_reg |= LTC2662_write(LTC2662_CS, LTC2662_CMD_WRITE_ALL, selected_dac, dac_code);
322  }
323  return fault_reg;
324 }

361  }
362  fault_reg |= LTC2662_write(LTC2662_CS,LTC2662_CMD_WRITE_ALL_UPDATE_ALL, 0, dac_code); // Send dac_code

550 //! Enable / Disable restoration of DAC values from EEPROM Use with caution - behaviour is undefined if you
551 //! enable restoration and data has NOT been previously stored from a known state.
552 void menu_11_enable_disable_eeprom_restore()
553 {
554  int16_t user_input;
555  Serial.println(F("\n0- Enable Restoration"));
556  Serial.println(F("1- Disable Restoration"));
557  Serial.println(F("2- Store DAC Settings"));
558  user_input = read_int();
559  switch (user_input)
560  {
561  case 0:
562  Serial.println(F("Enabling EEPROM restoration"));
563  eeprom.data_struct.cal_key = EEPROM_CAL_KEY;
564  eeprom_write_int16(EEPROM_I2C_ADDRESS, eeprom.data_struct.cal_key, EEPROM_CAL_STATUS_ADDRESS);
565  break;
566  case 1:
567  Serial.println(F("Disabling EEPROM restoration"));
568  eeprom.data_struct.cal_key = 0;
569  eeprom_write_int16(EEPROM_I2C_ADDRESS, eeprom.data_struct.cal_key, EEPROM_CAL_STATUS_ADDRESS);
570  break;
571  case 2:
572  eeprom.data_struct.cal_key = EEPROM_CAL_KEY;
573  eeprom_write_byte_array(EEPROM_I2C_ADDRESS, eeprom.byte_array, EEPROM_CAL_STATUS_ADDRESS, sizeof(eeprom_data_union));
574  Serial.println(F("Stored Settings to EEPROM"));
575  }
576 }

813 //! Read the alert settings from EEPROM
814 void restore_dac_from_eeprom()
815 {
816  int16_t cal_key;
817  int8_t i;
818 
819  // Read the cal key from the EEPROM
820  eeprom_read_int16(EEPROM_I2C_ADDRESS, &cal_key, EEPROM_CAL_STATUS_ADDRESS);
821 
822  if (cal_key == EEPROM_CAL_KEY)
823  {
824  // Calibration has been stored, read thresholds
825  eeprom_read_byte_array(EEPROM_I2C_ADDRESS, eeprom.byte_array, EEPROM_CAL_STATUS_ADDRESS, sizeof(eeprom_data_union));
826 
827  // Write SoftSpan ranges
828  for (uint8_t i = 0; i <= 15; i++)
829  LTC2662_write(LTC2662_CS, LTC2662_CMD_SPAN, i, eeprom.data_struct.soft_span_range[i]);
830 
831  // Write stored data to DAC A registers
832  for (uint8_t i = 0; i <= 15 ; i++)
833  LTC2662_write(LTC2662_CS,LTC2662_CMD_WRITE_N, i, eeprom.data_struct.dac_code_a[i]);
834 
835  LTC2662_write(LTC2662_CS, LTC2662_CMD_TOGGLE_SEL, 0, 0xFFFF); // Set to B registers
836 
837  // Write stored data to DAC B registers
838  for (uint8_t i = 0; i <= 15 ; i++)
839  LTC2662_write(LTC2662_CS,LTC2662_CMD_WRITE_N, i, eeprom.data_struct.dac_code_b[i]);
840 
841  LTC2662_write(LTC2662_CS, LTC2662_CMD_TOGGLE_SEL, 0, eeprom.data_struct.toggle_word); // Set toggle word
842 
843  LTC2662_write(LTC2662_CS, LTC2662_CMD_GLOBAL_TOGGLE, 0, eeprom.data_struct.global_toggle_bit); // Set global toggle Bit
844 
845  LTC2662_write(LTC2662_CS,LTC2662_CMD_MUX, 0, eeprom.data_struct.mux_state); // Set MUX
846 
847  LTC2662_write(LTC2662_CS, LTC2662_CMD_CONFIG, 0, eeprom.data_struct.reference_mode); // Set reference
848 
849  LTC2662_write(LTC2662_CS,LTC2662_CMD_UPDATE_ALL, 0, 0); // Update ALL

851  Serial.println(F("DAC settings loaded from EEPROM"));
