103 /* -------------------------------------------------------------------------
104  FUNCTION: LTC6950_read
105  - reads 8 bit Data field to LTC6950.
106  - has to shift data by one bit to account for RW bit
107  -------------------------------------------------------------------------- */
108 uint8_t LTC6950_read(uint8_t cs, int8_t address)
109 {
110  int8_t address_shift;
111  LT_union_int16_2bytes rx;
112 
113  address_shift =(address << 1) | 0x01; // shift to left to account for R/W bit, set bit high for read
114  spi_transfer_word(cs, address_shift<<8 , &rx.LT_uint16);
115 
116  LTC6950_reg[address]=rx.LT_byte[0];
117  return(rx.LT_byte[0]);
118 }

121 /* -------------------------------------------------------------------------
122  FUNCTION: LTC6950_read_field
123  For SPI FIELDS located in 1 or multiple address location
124  - reads specific address locations
125  - identifies and returns specific field in question
126  - can handle SPI fields in multiple addresses, if MSB bit is in the lower number address
127 --------------------------------------------------------------------------- */
128 long LTC6950_read_field(uint8_t cs, uint8_t address, uint8_t MSB_loc, uint8_t numbits)
129 {
130  int bit_shift, i, num_reg;
131  long field_val, maskbits, pow2;
132 
133  num_reg=0;
134  field_val=0;
135 // determines how many register are used
136  do
137  {
138  bit_shift = (MSB_loc+1)- (numbits-num_reg*8); // determines bit_shift for last register location
139  field_val=LTC6950_read(cs, (address+num_reg))+(field_val<<8); // reads current address locations, shifts previous address location 8 bits
140  num_reg++;
141  }
142  while ((bit_shift<0) && (num_reg<4));
143 
144 // creates a bit mask for complete word,
145  maskbits = 1;
146  pow2=1;
147  for (i=1, maskbits=1; i<numbits; i++)
148  {
149  pow2=pow2*2;
150  maskbits = maskbits+pow2;
151  }
152 
153  field_val=(field_val >>bit_shift) &maskbits;
154  return field_val;
155 }

184 /* -------------------------------------------------------------------------
185  FUNCTION: LTC6950_write_field
186  For SPI FIELDS
187  - reads specific address location
188  - identifies and returns specific field in question
189  - can handle SPI fields in multiple addresses, if MSB bit is in the lower number address
190 ---------------------------------------------------------------------------- */
191 uint8_t LTC6950_write_field(uint8_t cs, long field_data, uint8_t address, uint8_t MSB_loc, uint8_t numbits)
192 {
193  long current_content, desired_content, reg_val;
194  int LSB_loc, i, j, num_reg, bit_shift;
195  long temp_arr[32];
196 
197  for (i=0; i<32 ; i++) temp_arr[i]=0; // init temp_arr
198 
199 // read data in current address location and put in a bit array
200  num_reg=0;
201  current_content=0;
202  do
203  {
204  bit_shift=(MSB_loc+1)-(numbits-num_reg*8);
205  current_content=LTC6950_read(cs, (address+num_reg)) + (current_content<<8);
206 
207  num_reg++;
208  }
209  while ((bit_shift<0) && (num_reg<4));
210  for (i=0; i<(8*num_reg); i++)
211  {
212  temp_arr[i]=(current_content>>i) & 1;
213  }
214 
215 // exchange current bits with desired bits
216  LSB_loc = 8*(num_reg-1)+MSB_loc-numbits+1;
217  for (i=LSB_loc, j=0; i<=(MSB_loc+(num_reg-1)*8); i++, j++)
218  {
219  temp_arr[i] = (field_data>>j) &1;
220  } // end of for loop
221 
222 // reconstruct bits into an integer
223  desired_content = 0;
224  for (i=0; i<(8*num_reg); i++)
225  {
226  desired_content = desired_content | (temp_arr[i]<<i);
227  } // end of for loop
228 
229 // write new field value to part
230  for (i=0; i<num_reg; i++)
231  {
232  reg_val = (desired_content >> 8*(num_reg-1-i)) & 0xff;
233  LTC6950_write(cs, (address+i), reg_val);
234  } // end of for loop
235 } // end of LTC6950_write_field

684 /* -------------------------------------------------------------------------
685  FUNCTION: calc_odiv
686  - calculates the output divider setting based on the frf and on board
687  VCO frequencies of LTC6950
688  - @return odiv = 1-63 for valid setting, 999 as invalid odiv
689 ---------------------------------------------------------------------------- */
690 unsigned long LTC6950_calc_odiv(unsigned long frf[2])
691 {
692  unsigned long odiv, i;
693  unsigned long max_fout64[2];
694  unsigned long min_fout64[2];
695  unsigned long temp_fout[2];
696  unsigned long temp_i[2];
697  boolean valid_input=false;
698 
699 
700  HZto64(max_fout64,LTC6950_VCO_Max_Freq_MHz,LTC6950_VCO_Max_Freq_Hz);
701  HZto64(min_fout64,LTC6950_VCO_Min_Freq_MHz,LTC6950_VCO_Min_Freq_Hz);
702 
703 // verify desired frequency falls within a divider range (1-6)
704  valid_input=false;
705  for (i=1; i<=63; i++)
706  {
707  init64(temp_i,0L,i);
708  temp_fout[0] = frf[0];
709  temp_fout[1] = frf[1];
710  mul64(temp_fout,temp_i);
711  if (lt64(temp_fout,max_fout64) || eq64(temp_fout, max_fout64)) // same as frf*i <= max_fout
712  {
713  if (lt64(min_fout64,temp_fout) || eq64(temp_fout, min_fout64)) // same as frf*i >= min_fout
714  {
715  valid_input=true;
716  odiv=i;
717  }
718  }
719  } // end of for loop
720 
721  if (valid_input == false) odiv= 999L;
722  return odiv;
723 
724 } // end of LTC6950_calc_odiv

726 /* -------------------------------------------------------------------------
727  FUNCTION: LTC6950_set_frf
728  Calculates the integer (N) and output divider (OD) SPI values
729  using self created 64bit math functions. For Out0 only. Use menu 3 to set dividers to
730  achieve desired frequencies on other outputs.
731 
732  Datasheet equations
733  - fvco = fpfd*(N + F)
734  - frf = fvco/O
735  - fpfd = fref/R
736 
737  can be modified to the following equations
738  - N = (int) (fvco/fpfd) = (int) frf*O*R/fref
739 
740  where
741  - N = ndiv, O= M0div in the code below
742 
743  Linduino One (Arduino Uno) is limited to 32 bit floats/double/long.
744  32 bit math functions will create rounding errors with the above equations,
745  tha can result in frequency errors.
746  Therefore, the following code uses self created 64bit functions for 64bit integer math.
747 
748  - frf (33 bits) LTC6950 max frf/fvco = 1.4GHZ, which is 23 bit number (2^31 = 2.147G)
749  - fref (23 bits) LTC6950 min fref = 2MHz, which is a 20 bit number (2^20 = 1.05M)
750  - O (6 bits)
751  - R (10 bits)
752 
753  step 1: create 64 bit frf and fref numbers
754 
755  step 2: calculate O (output divider)
756 
757  step 3: get current R-divider setting
758 
759  step 4: calculate frf*O*R
760  - max bit count/resolution: 31b+6b+10b= 47b
761 
762  step 5: calculate N(11b), using value from step 1
763  - N = (int) frf*O*R/fref
764  - max bit count/resolution: 47b-20b = 27b
765 
766  step 6: find N for closest frequency - accounts for rounding with integer math
767  - 64b-47b=17b (using 16b instead of 17b)
768 ---------------------------------------------------------------------------- */
769 void LTC6950_set_frf()
770 {
771  unsigned long frf_MHz, frf_Hz, fref_MHz, fref_Hz, M0div, rdiv, ndiv, N_remainder;
772  unsigned long N64[2], R64[2], MO64[2], temp_long[2];
773  char buffer[100];
774  unsigned long frf[2], frf_act[2];
775  unsigned long fref[2];
776  unsigned long temp_math[2];
777  unsigned long frf_rdiv_M0div[2];
778  unsigned long frf_rdiv_M0div_int[2];
779  unsigned long roundup[2];
780 
781  /* step 1: create 64 bit frf and fref numbers
782  32 bit xxxx MHz number and 32 bit yyy yyy Hz number. */
783  frf_MHz=LTC6950_Frf_MHz;
784  frf_Hz=LTC6950_Frf_Hz;
785  HZto64(frf, frf_MHz, frf_Hz);
786 
787  fref_MHz=LTC6950_Fref_MHz;
788  fref_Hz=LTC6950_Fref_Hz;
789  HZto64(fref,fref_MHz,fref_Hz);
790 
791 // step 2: calculate O (output divider)
792  M0div=LTC6950_calc_odiv(frf);
793 
794 // step 3: get current R-divider setting
795  rdiv=get_LTC6950_SPI_FIELD(LTC6950_CS,LTC6950_R); // reads selected field
796 
797 // step 3: calculate frf*O*R
798  frf_rdiv_M0div[0]=0;
799  frf_rdiv_M0div[1]=M0div*rdiv;
800  mul64(frf_rdiv_M0div, frf); // odiv*rdiv*frf
801  frf_rdiv_M0div_int[0]=frf_rdiv_M0div[0]; // copy odiv*rdiv*frf to another variable
802  frf_rdiv_M0div_int[1]=frf_rdiv_M0div[1];
803 
804 // step 4: calculate N(10b), using value from step 3; N = (int) frf*O*R/fref
805  temp_math[0]=fref[0]; // copy fref to temp variable for math operation
806  temp_math[1]=fref[1];
807  div64(frf_rdiv_M0div_int, temp_math); // frf_rdiv_odiv_int= [(frf*odiv*rdiv)]/fref --> int(fvco/fpfd)
808  ndiv=frf_rdiv_M0div_int[1];
809 
810 // step 6: find N for closest frequency - accounts for rounding with integer math
811 // 64b-47b=17b (use 16b)
812  temp_math[0]=fref[0]; // copy fref to temp variable for math operation
813  temp_math[1]=fref[1];
814  shl64by(frf_rdiv_M0div,16); // frf_rdiv_M0div --> [(frf*M0div*rdiv)<<16]/fref = N(double) <<16
815  div64(frf_rdiv_M0div,temp_math);
816  shl64by(frf_rdiv_M0div_int,16); // frf_rdiv_M0div_int --> [(int)((frf*M0div*rdiv)/fref)] <<13 = N(int) <<16
817  sub64(frf_rdiv_M0div,frf_rdiv_M0div_int); // N(double) <<16 - N(int)<<16
818 
819 // at this point frf_rdiv_Modiv is the delta between N(double) and N(int) shifted by 16 bits.
820 // if this remainder is < 32768, N(int) will give the closest frequency
821 // if this remainder is >=32768, N(int)+1 will give the closest frequency
822 // 32768 is mid point of the 16 bit number range
823  N_remainder = (frf_rdiv_M0div[1] & 65535L); // 16 bits max code = 65535
824  if (N_remainder >= 32768)
825  {
826  ndiv=ndiv+1;
827  }
