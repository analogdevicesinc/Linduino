113 //! Repeats Linduino loop
114 void loop()
115 {
116  uint8_t user_select; // The user's choice of the three menu options
117  float user_value; // data value user inputs, either for gm or Rth
118  if (Serial.available())
119  {
120  user_select = read_int();
121  switch (user_select)
122  {
123  case 1: // Change gm
124  Serial.print(F("\nEnter desired gm (1-5.73), this will be rounded to nearest legal value:\n"));
125  user_value = read_float();
126 
127  gm_dec = gm_nearest_legal_value(user_value); // Snap given decimal gm value to nearest actual gm decimal value
128  gm = gm_decimal2hex(gm_dec); // Convert decimal gm to hex
129 
130  comp_config = (gm << 5) | ith; // Update comp_config with new gm value. Must bitshift gm left by 5. See page 74 of 3886 datasheet for details.
131  smbus->writeByte(ltc3886_i2c_address, MFR_PWM_COMP, comp_config);
132 
133  print_comp_config();
134  print_prompt();
135  break;
136  case 2: // Change ith
137  Serial.print(F("\nEnter desired ith (0-62), this will be rounded to nearest legal value:\n"));
138  user_value = read_float();
139 
140  ith_dec = ith_nearest_legal_value(user_value); // Snap given decimal ith value to nearest actual ith decimal value
141  ith = ith_decimal2hex(ith_dec); // Convert decimal ith to hex
142 
143  comp_config = (gm << 5) | ith; // Update comp_config with new ith value. Must bitshift gm left by 5. See page 74 of 3886 datasheet for details.
144  smbus->writeByte(ltc3886_i2c_address, MFR_PWM_COMP, comp_config);
145 
146  print_comp_config();
147  print_prompt();
148  break;
149  case 3: // Toggle Channel
150  if (channel == 0)
151  {
152  smbus->writeByte(ltc3886_i2c_address, PAGE, CH_1);
153  channel = CH_1;
154  }
155  else
156  {
157  smbus->writeByte(ltc3886_i2c_address, PAGE, CH_0);
158  channel = CH_0;
159  }
160 
161  Serial.println();
162  print_comp_config();
163  print_prompt();
164  break;
165  }
166  }
167 }

193 //! Prints current compensation configuration and saves ith, gm, and configuration hex values.
194 //! @return void
195 void print_comp_config()
196 {
197  comp_config = smbus->readByte(ltc3886_i2c_address, MFR_PWM_COMP);
198 
199  ith = 0x1F & comp_config; // Extract ith via bit-wise AND with 5 LSD
200  gm = (0xE0 & comp_config) >> 5; // Extract ith via bit-wise AND with 3 MSD. Bit shift right by 5.
201  ith_dec = ith_hex2decimal(ith); // Update ith hex value
202  gm_dec = gm_hex2decimal(gm); // Update gm hex value
203 
204  Serial.print(F("\nCH"));
205  Serial.print(channel, DEC);
206  Serial.print(F(" "));
207  Serial.print(F("Configuration, (gm, ith): ("));
208  Serial.print(gm_dec);
209  Serial.print(F(", "));
210  Serial.print(ith_dec);
211  Serial.print(F(")"));
212 }

215 //! Rounds decimal input ith to nearest legal decimal value.
216 //! @return double
217 float ith_nearest_legal_value(float ith)
218 {
219  for (int i=0; i<(sizeof(ith_vals)/4)-1; i++)
220  {
221  float cutoff = ith_vals[i] + (ith_vals[i+1] - ith_vals[i])/2;
222  if (ith < cutoff)
223  {
224  return ith_vals[i];
225  }
226  }
227  return ITH_MAX;
228 }
