98 /*
99  * Check if there is a fault log
100  *
101  * address: PMBUS address
102  */
103 bool
104 LT_FaultLog::hasFaultLog(uint8_t address)
105 {
106  uint8_t status;
107  PsmDeviceType t = pmbus_->deviceType(address);
108 
109  if (t == LTC3880 || t == LTC3886 || t == LTC3887 || t == LTM4675 || t == LTM4676|| t == LTM4676A || t == LTM4677)
110  {
111  status = readMfrStatusByte(address);
112  return (status & LTC3880_SMFR_FAULT_LOG) > 0;
113  }
114  else if (t == LTC3882 || t == LTC3882_1)
115  {
116  status = readMfrStatusByte(address);
117  return (status & LTC3882_SMFR_FAULT_LOG) > 0;
118  }
119  else if (t == LTC3883)
120  {
121  status = readMfrStatusByte(address);
122  return (status & LTC3883_SMFR_FAULT_LOG) > 0;
123  }
124  else if (t == LTC2974 || t == LTC2975)
125  {
126  status = readMfrFaultLogStatusByte(address);
127  return (status & LTC2974_SFL_EEPROM) > 0;
128  }
129  else if (t == LTC2977 || t == LTC2978 || t == LTC2980 || t == LTM2987)
130  {
131  status = readMfrFaultLogStatusByte(address);
132  return (status & LTC2978_SFL_EEPROM) > 0;
133  }
134  else
135  return false;
136 }

138 /*
139  * Enable fault log
140  *
141  * address: PMBUS address
142  */
143 void
144 LT_FaultLog::enableFaultLog(uint8_t address)
145 {
146  uint8_t config8;
147  uint16_t config16;
148  PsmDeviceType t = pmbus_->deviceType(address);
149 
150  if (
151  (t == LTC3880)
152  || (t == LTC3882)
153  || (t == LTC3882_1)
154  || (t == LTC3883)
155  || (t == LTC3886)
156  || (t == LTM4675)
157  || (t == LTM4676)
158  || (t == LTM4676A)
159  || (t == LTM4677)
160  || (t == LTC2978)
161  )
162  {
163  config8 = pmbus_->smbus()->readByte(address, MFR_CONFIG_ALL);
164  pmbus_->smbus()->writeByte(address, MFR_CONFIG_ALL, config8 | CFGALL_EFL);
165  }
166  else if (
167  (t == LTC2974)
168  || (t == LTC2975)
169  || (t == LTC2977)
170  || (t == LTC2980)
171  || (t == LTM2987)
172  )
173  {
174  config16 = pmbus_->smbus()->readWord(address, MFR_CONFIG_ALL);
175  pmbus_->smbus()->writeWord(address, MFR_CONFIG_ALL, config16 | LTC2974_CFGALL_EFL);
176  }
177 }

220 void LT_FaultLog::dumpBin(Print *printer, uint8_t *log, uint8_t size)
221 {
222  if (printer == 0)
223  printer = &Serial;
224  uint8_t *temp = log;
225  for (uint8_t i = 0; i < size; i++)
226  {
227  if (!(i % 16))
228  printer->println();
229  if (temp[i] < 0x10)
230  printer->write('0');
231  printer->print(temp[i], HEX);
232  }
233  printer->println();

249 uint64_t
250 LT_FaultLog::getSharedTime200us(FaultLogTimeStamp time_stamp)
251 {
252  uint64_t num200us = ((uint64_t) time_stamp.shared_time_byte5 << 40);
253  num200us = num200us | ((uint64_t) time_stamp.shared_time_byte4 << 32);
254  num200us = num200us | ((uint32_t) time_stamp.shared_time_byte3 << 24);
255  num200us = num200us | ((uint32_t) time_stamp.shared_time_byte2 << 16);
256  num200us = num200us | ((uint32_t) time_stamp.shared_time_byte1 << 8);
257  num200us = num200us | (time_stamp.shared_time_byte0);
258  return num200us;
259 
260 }
