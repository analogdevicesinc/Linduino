78 //! @name Assigned Pin Numbers

81 #define DI 11 // soft UART with interrupt pin (SSerial TX)

85 #define TE485b 6 // RS485 termination enable BAR

96 //! Initialize Linduino
97 void setup()
98 {
99  // Set pin directions
100  pinMode(RO, INPUT_PULLUP);
101  pinMode(DI, OUTPUT);
102  pinMode(DE485_F232, OUTPUT);
103  pinMode(RE485b, OUTPUT);
104  pinMode(MODE485,OUTPUT);
105  pinMode(TE485b,OUTPUT);
106  pinMode(LB,OUTPUT);
107  pinMode(SHDNb,OUTPUT);
108 
109  // Initialize pins
110  digitalWrite(DI,LOW);
111  digitalWrite(DE485_F232,LOW);
112  digitalWrite(RE485b,HIGH);
113  digitalWrite(MODE485,HIGH);
114  digitalWrite(TE485b,HIGH);
115  digitalWrite(LB,LOW);
116  digitalWrite(SHDNb,LOW);
117 
118  TCCR2B = TCCR2B & 0b11111000 | 0x01; // set PWM freq to 31372.55Hz (and scales timers!)
119  //TCCR2B = TCCR2B & 0b11111000 | 0x04; // set PWM freq on pins 2, 11 to 490.20Hz (default)
120 
121  // Open communications to serial monitor
122  Serial.begin(115200); // DON'T FORGET TO SET SERIAL MONITOR TO THIS SPEED!!!
123  // RS485Serial.begin(4800); // Start the software serial port, to another device
124  print_title(); // Menu on serial monitor
125  print_status();
126 }

128 //! Repeats Linduino loop
129 void loop()
130 {
131  uint8_t user_command = read_int();
132  Serial.print(F(" "));
133 
134  if (user_command > 0 && user_command < 27)
135  {
136  update(user_command);
137  Serial.print(user_command);
138  print_status(); // Print status after operation
139  }
140  else if (user_command == 'm')
141  {
142  update(user_command);
143  Serial.print(F("m"));
144  print_status(); // Print status after operation
145  }
146  else
147  {
148  Serial.println(F("Wrong choice. TRY AGAIN!"));
149  }
150 }
