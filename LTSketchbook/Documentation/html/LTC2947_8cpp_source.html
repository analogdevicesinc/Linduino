99 boolean LTC2947_Abs(uint8_t *bytes, uint8_t length)
100 {
101  if (bitMaskClrChk(*bytes, 0x80))
102  return false;// value is already positive
103 
104  // two's complement is generated by inverting all bits and add 1
105 
106  length--;
107  bytes += length; // seek to LSB
108  uint16_t cHelp = (~(*bytes)) & 0xFF; // invert LSB
109  cHelp++; // add 1
110  *bytes = cHelp; // store back to buffer
111 
112  while (length != 0)
113  {
114  // seek next byte (towards MSB)
115  length--;
116  bytes--;
117  cHelp = cHelp >> 8; // restore carry from previous sum
118  cHelp += (~(*bytes)) & 0xFF; // add inverted byte
119  *bytes = cHelp; // store back
120  }
121  return true;// value inverted
122 }

124 double LTC2947_BytesToDouble(uint8_t *bytes, uint8_t length, boolean sig, double lsb)
125 {
126  if (length == 0)
127  return 0.0;
128  else if (length == 1)
129  return sig ? (int8_t)(bytes[0])*lsb : bytes[0] * lsb;
130  else if (length == 2)
131  return sig ? LTC2947_2BytesToInt16(bytes)*lsb : LTC2947_2BytesToUInt16(bytes)*lsb;
132  else if (length == 3)
133  return sig ? LTC2947_3BytesToInt32(bytes)*lsb : LTC2947_3BytesToUInt32(bytes)*lsb;
134  else if (length == 4)
135  return sig ? LTC2947_4BytesToInt32(bytes)*lsb : LTC2947_4BytesToUInt32(bytes)*lsb;
136  else
137  return sig ? LTC2947_SignedBytesToDouble(bytes, length, lsb) : LTC2947_UnsignedBytesToDouble(bytes, length, lsb);
138 }

140 double LTC2947_UnsignedBytesToDouble(uint8_t *unsignedBytes, uint8_t length, double lsb)
141 {
142  // NOTE: On Arduino double is 32-bit and NOT 64-bit, thus the returned value
143  // will not reflect the full precission of e.g. C1, C2... which are 48-bit values
144  double ret = (*unsignedBytes); // MSB!
145 
146  while (length > 1)
147  {
148  unsignedBytes++; // go to next byte
149  length--;
150  ret = ret * 256.0 + (*unsignedBytes);
151  }
152  return ret*lsb;
153 }

250 double LTC2947_SignedBytesToDouble(uint8_t *signedBytes, uint8_t length, double lsb)
251 {
252  // reserve memory for unsigned bytes
253  uint8_t *unsignedBytes = (uint8_t *)malloc(length);
254  // copy signed bytes to unsigned bytes
255  memcpy(unsignedBytes, signedBytes, length);
256  // calculate absolute value of the signed bytes and store sign
257  // this function will change the unsigned bytes, for this reason
258  // we copied the original unsigned bytes to a new array
259  boolean sign = LTC2947_Abs(unsignedBytes, length);
260  // convert the unsigned bytes to a double value
261  double absDouble = LTC2947_UnsignedBytesToDouble(unsignedBytes, length, lsb);
262  // free the allocated memory of the copied array
263  free(unsignedBytes);
264  // recover the previously stored sign to return a signed value
265  return sign ? -absDouble : absDouble;
266 }

439 int8_t LTC2947_I2CRdBlock(uint8_t slvAddr, uint8_t regAddr, uint8_t length, uint8_t *values)
440 {
441  int8_t ret = 0;
442 
443  if (length == 0 || i2c_start() != 0) //I2C START
444  return 1; //Stop and return 0 if START fail
445 
446  ret |= i2c_write((slvAddr << 1) | I2C_WRITE_BIT); // Write 7 bit address with W bit
447  ret |= i2c_write(regAddr); // Set register address
448  ret |= i2c_repeated_start();
449  ret |= i2c_write((slvAddr << 1) | I2C_READ_BIT); // Write 7 bit address with R bit
450 
451  if (ret != 0) //If NACK return 1
452  {
453  i2c_stop(); //I2C STOP
454  return 1;
455  }
456 
457  length--;
458  while (length > 0)
459  {
460  *values = i2c_read(WITH_ACK); //Read from bus with ACK
461  values++;
462  length--;
463  }
464 
465  *values = i2c_read(WITH_NACK); //Read from bus with NACK for the last one;
466 
467  i2c_stop(); //I2C STOP
468 
469  return 0; // Success!
470 }

536 uint8_t LTC2947_Ara(uint8_t *svlAddr)
537 {
538  *svlAddr = 0;
539  //! Send I2C start bit
540  if (i2c_start() != 0)
541  return LTC2947_ARA_ERROR;
542 
543  //! send the ALERT RESPONSE ADDRESS with read bit
544  if (i2c_write((LTC2947_ALERT_RESP_ADDR << 1) | I2C_READ_BIT))
545  {
546  //! NACK means no device response!
547  return LTC2947_ARA_NO_RESPONSE;
548  }
549 
550  //! Read device address from the responding device
551  *svlAddr = i2c_read(WITH_NACK); //! read with NACK
552  i2c_stop(); //! I2C STOP
553  //! check for the expected write bit of the response
554  boolean response_wr_bit = bitMaskClrChk(*svlAddr, 0x1);
555  //! right shift to get 7-bit slave address
556  *svlAddr = ((*svlAddr) >> 1) & 0x7F;
557  if (response_wr_bit)
558  {
559  //! got expected write bit, compare slave address
560  //! with LTC2947's address
561  return ((*svlAddr) == LTC2947_I2C_Slave_Addr)
562  ? LTC2947_ARA_LTC2947_RESPONSE
563  : LTC2947_ARA_OTHER_RESPONSE;
564  }
565 
566  //! missing write bit within response!
567  return LTC2947_ARA_RESPONSE_WO_WR;
568 }

570 int16_t LTC2947_wake_up()
571 {
572  byte data[1];
573  unsigned long wakeupStart = millis(), wakeupTime;
574  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, 0);//! any serial transaction will wakeup LTC2947
575  do
576  {
577  delay(1);
578  LTC2947_RD_BYTE(LTC2947_REG_OPCTL, data); //! wake up polling by reading OPCTL
579  wakeupTime = millis() - wakeupStart;
580  if (data[0] == 0) //! check if we are in idle mode
581  {
582  //! wake up successful, return wakeup time in milliseconds
583  return wakeupTime;
584  }
585  if (wakeupTime > 200)
586  {
587  //! failed to wake up due to timeout, return -1
588  return -1;
589  }
590  }
591  while (true);
592 }

606 void LTC2947_Read_I_P_V_TEMP_VCC(float *I, float *P, float *V, float *TEMP, float *VCC)
607 {
608  // byte array to store register values
609  byte bytes[12];
610 
611  // read measurement results from device
612  LTC2947_RD_BYTES(LTC2947_VAL_I, 6, bytes); // I[23:0] P[23:0] starting at data[0]
613  LTC2947_RD_BYTES(LTC2947_VAL_V, 6, bytes + 6); // V[15:0] TEMP[15:0] VDVCC[15:0] starting at data[6]
614 
615  // convert to floating point values
616  // Note: definitions in LTC2947.h are given in mA, mW, mV
617  // so they have to be mutliplied by 1e-3 to get A, W, V respectively
618  *I = LTC2947_3BytesToInt32(bytes) * LTC2947_LSB_I * 1e-3; // calc current in amps
619  *P = LTC2947_3BytesToInt32(bytes + 3) * LTC2947_LSB_P * 1e-3; // calc power in watts
620  *V = LTC2947_2BytesToInt16(bytes + 6) * LTC2947_LSB_V * 1e-3; // calc voltage in volts
621  *TEMP = LTC2947_2BytesToInt16(bytes + 6 + 2) * LTC2947_LSB_TEMP + LTC2947_OFFS_TEMP; // calc temperature in degree celcius
622  *VCC = LTC2947_2BytesToInt16(bytes + 6 + 4) * LTC2947_LSB_VDVCC * 1e-3; // calc supply voltage in volts
623 }

625 void LTC2947_Read_Abs_C_E_TB(boolean accuSet1, double *C, boolean *signC, double *E, boolean *signE, double *TB)
626 {
627  // byte array to store register values
628  byte bytes[16];
629

631  if (accuSet1)
632  // read accumulated quantities set 1: C1[47:0] E1[47:0] TB1[31:0]
633  LTC2947_RD_BYTES(LTC2947_VAL_C1, 16, bytes);
