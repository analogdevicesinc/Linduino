61 //! Initialize Linduino once on power-up or reset.
62 void setup()
63 {
64 
65  Serial.begin(115200); //! Initializes the serial port to the PC.
66 
67  output_high(LTC2422_CS); //! Pull LTC2442 Chip Select High.
68 
69  quikeval_I2C_init(); //! Initializes Linduino I2C port.
70  quikeval_I2C_connect();
71 
72  pinMode(2, OUTPUT); //! Configure Arduino Pins 2, 3, and 4 as outputs.
73  pinMode(3, OUTPUT);
74  pinMode(4, OUTPUT);
75 
76  uint16_t dac_code;
77  //! Find LTC2607 dac_code that corresponds to 4.5V
78  dac_code = LTC2607_voltage_to_code(4.5, LTC2607_TYPICAL_lsb, LTC2607_TYPICAL_OFFSET);
79  //! Set DAC_A output to 4.5V
80  LTC2607_write(LTC2607_I2C_GLOBAL_ADDRESS, LTC2607_WRITE_UPDATE_COMMAND, LTC2607_DAC_B, dac_code);
81  //! Make DAC_B output high impedance (power down)
82  LTC2607_write(LTC2607_I2C_GLOBAL_ADDRESS, LTC2607_POWER_DOWN_COMMAND, LTC2607_DAC_A, 0);
83 
84  quikeval_SPI_init(); //! Configures the SPI port for 4Mhz SCK.
85  quikeval_SPI_connect(); //! Connects SPI to QuikEval port.
86 
87  uint8_t adc_channel;
88  uint32_t adc_code;
89 
90  if (LTC2422_EOC_timeout(LTC2422_CS, LTC2422_timeout)) // Check for EOC
91  return;
92  LTC2422_adc_read(LTC2422_CS, &adc_channel, &adc_code); // Align channel by reading again if adc_channel=0
93 
94  if (adc_channel == 0)
95  {
96  if (LTC2422_EOC_timeout(LTC2422_CS, LTC2422_timeout)) // Check for EOC
97  return;
98  LTC2422_adc_read(LTC2422_CS, &adc_channel, &adc_code);
99  }
100 }

102 void loop()
103 {
104  static int32_t previous_adc_code; //stores ADC code from previous time through loop
105  uint32_t adc_code;
106  int32_t delta_adc_code;
107  uint8_t adc_channel;
108 
109  if (LTC2422_EOC_timeout(LTC2422_CS, LTC2422_timeout)) // Check for EOC
110  return;
111  LTC2422_adc_read(LTC2422_CS, &adc_channel, &adc_code); // Read channel 0 and throw it away
112 
113  if (LTC2422_EOC_timeout(LTC2422_CS, LTC2422_timeout)) // Check for EOC
114  return;
115  LTC2422_adc_read(LTC2422_CS, &adc_channel, &adc_code); // Read channel 1 into adc_code
116  delay(300);
117 
118  delta_adc_code = adc_code - previous_adc_code;
119 
120  uint8_t pin2=0, pin3=0, pin4=0;
121  if (delta_adc_code > 30) pin2 = 1;
122  if ((delta_adc_code <= 30) && (delta_adc_code >=-30)) pin3=1;
123  if (delta_adc_code < -30) pin4=1;
124 
125  digitalWrite(2,pin2);
126  digitalWrite(3,pin3);
127  digitalWrite(4,pin4);
128 
129  Serial.println();
130  Serial.print("RED: "); //! Print "RED: " to serial port
131  Serial.println(pin4);
132  Serial.print("GREEN: ");
133  Serial.println(pin3);
134  Serial.print("BLUE: ");
135  Serial.println(pin2);
136 
137  previous_adc_code=adc_code;
138 }
