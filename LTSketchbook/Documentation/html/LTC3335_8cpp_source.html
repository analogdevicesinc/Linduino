164 // Initializes the LTC3335 driver.
165 int8_t LTC3335_Init(void)
166 {
167  int8_t result;
168  uint8_t ltc3335_data;
169 
170  // Init all of the microprocessor hardware used by the LTC3335 driver.
171  LTC3335_MICRO_INIT();
172 
173  // Init to INVALID, so that subaddress will be sent for first right. After this, subaddress will only be written if needed.
174  ltc3335_subaddress_last = LTC3335_INVALID;
175 
176  // Send command to set the voltage and prescaler set in LTC3335_Config.h with one i2c transaction.
177  ltc3335_data = ltc3335_encode_register_a(true, (LTC3335_OUTPUT_VOLTAGE_TYPE) LTC3335_OUTPUT_VOLTAGE, LTC3335_PRESCALER);
178  result = ltc3335_set_register(LTC3335_REGISTER_A, &ltc3335_data);
179 
180  // Set cached values in case voltage or prescaler are changed during runtime.
181  ltc3335_voltage_selection_last = (LTC3335_OUTPUT_VOLTAGE_TYPE) LTC3335_OUTPUT_VOLTAGE;
182  ltc3335_prescaler_last = LTC3335_PRESCALER;
183  ltc3335_voltage_selection_enabled_last = true;
184 
185  // Set the alarm threshold. Round down so that set value is always <= to the desired value.
186 #if LTC3335_USE_SOFTWARE_CORRECTION == false
187  ltc3335_data = (1LL * LTC3335_ALARM_CAPACITY * SEC_PER_HR)/LTC3335_RESOLUTION(LTC3335_PRESCALER);
188 #else
189  ltc3335_data = (1LL * LTC3335_ALARM_CAPACITY * SEC_PER_HR)/(LTC3335_RESOLUTION(LTC3335_PRESCALER) * ((1LL << 16) + LTC3335_CORRECTION_FACTOR_TYP) / (1L << 16));
190 #endif // #if LTC3335_USE_SOFTWARE_CORRECTION == false
191  result |= ltc3335_set_register(LTC3335_REGISTER_B, &ltc3335_data);
192 
193  // Turn counter test on or off given setting in in LTC3335_Config.h.
194  result |= LTC3335_Set_Counter_Test(LTC3335_USE_CURRENT_MEASUREMENT);
195 
196 #if LTC3335_USE_SOFTWARE_CORRECTION == true
197  // Initialize the corrected discharged capacity and timer for quiescent current accumulation..
198  ltc3335_discharged_capacity = 0;
199  ltc3335_accumulator_last = 0;
200  ltc3335_quiescent_current_timer = LTC3335_TIMER_GET();
201  result |= LTC3335_Get_Discharged_Capacity(&ltc3335_discharged_capacity, LTC3335_VBAT_TYP);
202 #endif // #if LTC3335_USE_SOFTWARE_CORRECTION == true
203 
204  return result;
205 }

207 // Enables/Disables software control of the LTC3335 output voltage.
208 // If software control is enabled, the voltage is set to the specified setting.
209 int8_t LTC3335_Set_Voltage(boolean enabled, LTC3335_OUTPUT_VOLTAGE_TYPE voltage)
210 {
211  int8_t result;
212  uint8_t ltc3335_data;
213 
214  // Can not set out of range.
215  if (LTC3335_NUM_OUTPUT_VOLTAGES <= voltage)
216  {
217  return 1;
218  }
219 
220  // Send command to set the voltage.
221  ltc3335_data = ltc3335_encode_register_a(enabled, voltage, ltc3335_prescaler_last);
222  result = ltc3335_set_register(LTC3335_REGISTER_A, &ltc3335_data);
223 
224  return result;

227 // Gets the alarms active from the LTC3335.
228 int8_t LTC3335_Get_Alarms(LTC3335_ALARM_TYPE *alarms)
229 {
230  int8_t result;
231  uint8_t ltc3335_data;
232 
233  result = ltc3335_get_register(LTC3335_REGISTER_D, &ltc3335_data);
234 
235  if (result == 0)
236  {
237  // Decode the register into its alarm bits.
238  if (ltc3335_data & MASK(1, 0))
239  {
240  alarms->ac_on_time_overflow = 1;
241  }
242  else
243  {
244  alarms->ac_on_time_overflow = 0;
245  }
246 
247  if (ltc3335_data & MASK(1, 1))
248  {
249  alarms->coulomb_counter_overflow = 1;
250  }
251  else
252  {
253  alarms->coulomb_counter_overflow = 0;
254  }
255 
256  if (ltc3335_data & MASK(1, 2))
257  {
258  alarms->alarm_trip = 1;
259  }
260  else
261  {
262  alarms->alarm_trip = 0;
263  }
264  }
265 
266  return result;
267 }

269 // Sends the command to clear the INT condition.
270 // NOTE! Additional registers are rewritten in order for the INT condition to be reset.
271 int8_t LTC3335_Clear_Int(LTC3335_ALARM_TYPE *alarms)
272 {
273  int8_t result = 0;
274  uint8_t ltc3335_data;
275 
276  // Write to other registers required depending upon the specific type of alarm
277  if (alarms->coulomb_counter_overflow)
278  {
279  // The accumulator must be rewritten in order for the clear int command to clear these alarms.
280  result |= ltc3335_get_register(LTC3335_REGISTER_C, &ltc3335_data);
281  if (result == 0) result |= ltc3335_set_register(LTC3335_REGISTER_C, &ltc3335_data);
282  }
283 
284  if (alarms->alarm_trip)
285  {
286  // Note that if the accumulator is >= the alarm value, the alarm_trip bit will immediately trip again.
287  }
288 
289  if (alarms->ac_on_time_overflow)
290  {
291  // Note AC ON Time Alarm can not be cleared until PK[2:0] (Pin 15, 14, 13) are changed to match inductor value.
292  }
293 
294  // If currently doing a Counter Test, do not turn it off when clearing the int.
295  ltc3335_data = 1;
296  if (ltc3335_counter_test_last == true)
297  {
298  ltc3335_data |= 2;
299  }
300 
301  result |= ltc3335_set_register(LTC3335_REGISTER_E, &ltc3335_data);
302  ltc3335_data++;
303 
304  return result;
305 }

307 #if LTC3335_USE_SOFTWARE_CORRECTION == false
308 // Reads the accumulator and translates into discharged capacity.
309 int8_t LTC3335_Get_Discharged_Capacity(uint32_t *discharged_capacity)
310 {
311  int8_t result = true;
312  uint8_t ltc3335_data;
313 
314  result = ltc3335_get_register(LTC3335_REGISTER_C, &ltc3335_data);
315 
316  if (result == 0)
317  {
318  *discharged_capacity = ltc3335_data * LTC3335_RESOLUTION(LTC3335_PRESCALER);
319  }
320 
321  return result;
322 }
323 #else
324 // Tracks changes in the accumulator and translates changes in discharged capacity, accounting
325 // for the resolution available at the current battery voltage, output voltage, and ipeak.
326 // The quiescent current is accumulated over time and used to update the discharged capacity.
327 int8_t LTC3335_Get_Discharged_Capacity(uint32_t *discharged_capacity, uint16_t vbat)
328 {
329  int8_t result;
330  uint8_t ltc3335_data;
331 
332  result = ltc3335_get_register(LTC3335_REGISTER_C, &ltc3335_data);
333 
334  if (result == 0)
335  {
336  if (ltc3335_data != ltc3335_accumulator_last)
337  {
338  // Add in changes in the accumulator, adjusted by the software correction factor.
339  int16_t temp16 = LTC3335_Get_Software_Correction_Factor(vbat);
340  uint32_t mAs = (uint8_t)(ltc3335_data - ltc3335_accumulator_last) * LTC3335_RESOLUTION(LTC3335_PRESCALER);
341  mAs += ((mAs * temp16 + (1L << 15)) >> 16) | (temp16 < 0 ? 0xFFFF0000LL : 0);
342  ltc3335_discharged_capacity += mAs;
343  ltc3335_accumulator_last = ltc3335_data;
344  }
345  }
346 
347  // Look for enough time to pass to increment discharged capacity by 1mAs.
348  if (LTC3335_TIMER_GET() - ltc3335_quiescent_current_timer > LTC3335_TIMER_COUNTS_PER_IQ_MAS)
349  {
350  ltc3335_discharged_capacity++;
351  ltc3335_quiescent_current_timer += LTC3335_TIMER_COUNTS_PER_IQ_MAS;
352  }
353 
354  *discharged_capacity = ltc3335_discharged_capacity;
355 
356  return result;
357 
358 }
359 #endif // LTC3335_USE_SOFTWARE_CORRECTION == false

361 // Enables/Disables the LTC3335 Counter Test feature. This can be used to verify the
362 // LTC3335 is functional, or it can be used to measure the battery current.
363 int8_t LTC3335_Set_Counter_Test(boolean enabled)
364 {
365  int8_t result;
366  uint8_t ltc3335_data;
367 
368  // Send command to start the test.
369  ltc3335_data = ((enabled == true) ? MASK(1, 1) : 0);
370  result = ltc3335_set_register(LTC3335_REGISTER_E, &ltc3335_data);
371 
372  // Store results of the test so that we don't need to poll the IC for that info.
373  if (result == 0)
374  {
375  ltc3335_counter_test_last = enabled;
376  }
377 
378 #if LTC3335_USE_CURRENT_MEASUREMENT == true
379  // Initialize after the command is sent, as the first edge that occurs
380  // when the test is started should not be counted.
381  if (enabled == true)
382  {
383  // Take initial readings when test is started.
384  LTC3335_Counter_Test_Current_Task();
385  }
386  else
387  {
388  // Reset test results when test is stopped.
389  LTC3335_Reset_Counter_Test_Current();
390  }
391 #endif
392 
393  return result;
394 }

409 // Accumulates the number of edges on the /IRQ pins provided by the LTC3335 Counter Test feature over a period of time,
410 // the ratio of which can be used to calculate battery current.
411 // Task that must be run periodically at a rate faster than LTC3335_MIN_CURRENT_TASK_RATE.
412 void LTC3335_Counter_Test_Current_Task(void)
413 {
414  if (ltc3335_counter_test_last == true)
415  {
416  // Get new timer and counter values
417  uint16_t hw_timer_new = LTC3335_TIMER_GET();
418  uint16_t hw_counter_new = LTC3335_COUNTER_GET();
419 
420  // Add difference between last timer and counter to summations.
421  ltc3335_counter_test_edge_count += (hw_counter_new - ltc3335_hw_counter_last);
422  ltc3335_counter_test_time += (hw_timer_new - ltc3335_hw_timer_last);
423 
424  // Save new timer and counter values for next execution of task.
425  ltc3335_hw_timer_last = hw_timer_new;
426  ltc3335_hw_counter_last = hw_counter_new;
427  }
428 
429  return;
430 }

432 #if LTC3335_USE_SOFTWARE_CORRECTION == false
433 // Calculates the battery current from the number of edges on the /IRQ pins over a period of time accumulated in LTC3335_Counter_Test_Current_Task.
434 int8_t LTC3335_Get_Counter_Test_Current(uint16_t *microamps)
435 {
436  if ((ltc3335_counter_test_last == true) && ((ltc3335_counter_test_time != 0)))
437  {
438  if (ltc3335_counter_test_edge_count != 0)
439  {
440  uint64_t uAs = (uint64_t)(1LL * LTC3335_IPEAK_MA * UA_PER_MA * LTC3335_TFS * LTC3335_TIMER_COUNTS_PER_SEC) * ltc3335_counter_test_edge_count;
441  uAs += ltc3335_counter_test_time/2;
442  uAs /= ltc3335_counter_test_time;
443  if (uAs >= (1L << 16))
444  {
445  *microamps = (1L << 16) - 1;
446  }
447  else
448  {
449  *microamps = (uint16_t) uAs;
450  }
451  }
452  else
453  {
454  *microamps = 0;
455  }
456  return 0;
457  }
458  else
459  {
460  return 1;
461  }
462 }

505 // Writes to an LTC3335 register.
506 static int8_t ltc3335_set_register(uint8_t subaddress, uint8_t *ltc3335_data_ptr)
507 {
508  int8_t result = 0;
509  result |= i2c_write_byte_data(LTC3335_BASE_ADDRESS, subaddress, *ltc3335_data_ptr);
510  return result;
511 }

513 // Reads from a LTC3335 register.
514 static int8_t ltc3335_get_register(uint8_t subaddress, uint8_t *ltc3335_data_ptr)
515 {
516  int8_t result = 0;
517  if (subaddress != ltc3335_subaddress_last)
518  {
519  result |= i2c_write_byte(LTC3335_BASE_ADDRESS, subaddress);
520  }
521 
522  result |= i2c_read_byte(LTC3335_BASE_ADDRESS, ltc3335_data_ptr);
523  return result;
524 }

526 // Encodes elements of register A.
527 static uint8_t ltc3335_encode_register_a(boolean enabled, LTC3335_OUTPUT_VOLTAGE_TYPE voltage, uint8_t prescaler)
528 {
529  return ((enabled == true) ? (1 << 7) : 0) | (voltage << 4) | prescaler;
530 }
531 
532 // Decodes elements of register A.
533 static void ltc3335_decode_register_a(uint8_t register_a)
534 {
535  ltc3335_voltage_selection_enabled_last = ((register_a & MASK(1, 7)) == MASK(1, 7));
536  ltc3335_voltage_selection_last = (LTC3335_OUTPUT_VOLTAGE_TYPE) ((register_a & MASK(3, 4)) >> 4);
537  ltc3335_prescaler_last = register_a & MASK(4, 0);
538  return;
