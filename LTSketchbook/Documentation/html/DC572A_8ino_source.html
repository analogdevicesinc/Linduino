5 @verbatim
6 NOTES
7  Setup:
8  Set the terminal baud rate to 115200 and select the newline terminator.
9 
10  An external +/- 15V power supply is required to power the circuit.
11 
12  Explanation of Commands:
13  1- Voltage Output
14  Displays the calculated voltage depending on the code input from user and
15  voltage range selected.
16 
17  2- Square Wave Output
18  Generates a square wave on the output pin. This function helps to measure
19  settling time and glitch impulse.
20 
21  3- Change Range
22  | Command | Range Selected |
23  | C3 C2 C1 C0 | |
24  |------------------------------|
25  | 1 0 0 0 | 0V - 5V |
26  | 1 0 0 1 | 0V - 10V |
27  | 1 0 1 0 | -5V - +5V |
28  | 1 0 1 1 | -10V - +10V |
29  | 1 1 0 0 | -2.5V - +2.5V |
30  | 1 1 0 1 | -2.5V - 7.5V |
31 
32 
33 USER INPUT DATA FORMAT:
34  decimal : 1024
35  hex : 0x400
36  octal : 02000 (leading 0 "zero")

286 
287  Serial.print("\nEnter voltage_low: ");

290  Serial.print(voltage_low);

293 
294  Serial.print("\nEnter the required frequency in Hz: ");

303 
304  //! Converting data into voltage

306  data_voltage_low = LTC1592_voltage_to_code(voltage_low, LTC1592_RANGE_HIGH, LTC1592_RANGE_LOW);
307 
308  switch (range_selected)
309  {
310  case 0:
311  dac_command = LTC1592_RANGE_0_TO_5V;
312  break;
313 
314  case 1:
315  dac_command = LTC1592_RANGE_0_TO_10V;
316  break;
317 
318  case 2:
319  dac_command = LTC1592_RANGE_PLUS_MINUS_5V;
320  break;
321 
322  case 3:
323  dac_command = LTC1592_RANGE_PLUS_MINUS_10V;
324  break;
325 
326  case 4:
327  dac_command = LTC1592_RANGE_PLUS_MINUS_2_5V;
328  break;
329 
330  case 5:
331  dac_command = LTC1592_RANGE_MINUS_2_5V_TO_PLUS_7_5V;
332  break;
333  }
334 
335  while (!Serial.available()) //! Generate square wave until a key is pressed
336  {
337  LTC1592_write(LTC1592_CS, dac_command, data_voltage_high);

339  LTC1592_write(LTC1592_CS, dac_command, data_voltage_low);
340  delayMicroseconds(time * 500);
341  }
342  receive_enter = read_int();
343  return 0;
344 }
