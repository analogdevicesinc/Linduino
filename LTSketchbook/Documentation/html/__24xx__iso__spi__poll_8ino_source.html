137 //! Repeats Linduino loop
138 void loop()
139 {
140  uint8_t user_command; // The user input command
141  uint8_t ack = 0;
142  if (demo_board_connected)
143  {
144  if (Serial.available()) // Check for user input
145  {
146  user_command = read_int(); // Read the user command
147  if (user_command == 'm');
148  else
149  Serial.println(user_command);
150  delay(50); // Allow the print to finish
151  switch (user_command)
152  {
153  case 1:
154  ack |= menu_1_read_single_ended();
155  break;
156  case 2:
157  ack |= menu_2_read_differential();
158  break;
159  default:
160  Serial.println("Incorrect Option");
161  break;
162  }
163  if (ack)
164  Serial.println(F("***** SPI ERROR *****"));
165  Serial.println(F("*****************************************************************"));
166  print_prompt();
167  }
168  }
169 }

199 //! read from ADC single-ended
200 //! @return 0 if successful, 1 if failure
201 //! @return 0 when m is entered into menu, 1 if timeout for EOC
202 uint8_t menu_1_read_single_ended()
203 {
204  uint8_t adc_command; // The LTC24XX command byte
205  int16_t user_command; // The user input command
206  int32_t adc_code=0; // The LTC24XX code
207  float adc_voltage; // The LTC24XX voltage
208 
209  while (1)
210  {
211  Serial.print(F("*************************\n\n")); // Display single-ended menu
212  Serial.print(F("0-CH0 8-CH8\n"));
213  Serial.print(F("1-CH1 9-CH9\n"));
214  Serial.print(F("2-CH2 10-CH10\n"));
215  Serial.print(F("3-CH3 11-CH11\n"));
216  Serial.print(F("4-CH4 12-CH12\n"));
217  Serial.print(F("5-CH5 13-CH13\n"));
218  Serial.print(F("6-CH6 14-CH14\n"));
219  Serial.print(F("7-CH7 15-CH15\n"));
220  Serial.print(F("16-ALL\n"));
221  Serial.print(F("m-Main Menu\n"));
222  Serial.print(F("\nEnter a Command: "));
223  user_command = read_int(); // Read the single command
224  if (user_command == 'm')
225  return(0);
226  else
227  Serial.println(user_command);
228  Serial.println();
229 
230  if (user_command == 16)
231  {
232  Serial.print(F("ALL\n"));
233  adc_command = BUILD_COMMAND_SINGLE_ENDED[0]; // Build ADC command for channel 0
234  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
235  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
236  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
237  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
238 
239  for (int8_t x = 0; x <= 15; x++)
240  {
241  adc_command = BUILD_COMMAND_SINGLE_ENDED[(x + 1) % 16]; // Read all channels in single-ended mode
242  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
243  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code);
244  adc_voltage = LTC24XX_diff_code_to_voltage(adc_code, reference_voltage);
245  Serial.print(F(" ****"));
246  Serial.print(F("CH"));
247  Serial.print(x);
248  Serial.print(F(": "));
249  Serial.print(adc_voltage, 4);
250  Serial.print(F("V\n"));
251  }
252  }
253  else if (user_command<16 && user_command >=0) // Read selected channel
254  {
255  adc_command = BUILD_COMMAND_SINGLE_ENDED[user_command];
256  Serial.print(F("\nADC Command: h"));
257  Serial.println(adc_command, HEX);
258  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
259  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
260  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
261  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code);
262  Serial.print(F("Received Code: 0x"));
263  Serial.println(adc_code, HEX);
264  adc_voltage = LTC24XX_diff_code_to_voltage(adc_code, reference_voltage);
265  Serial.print(F(" ****"));
266  Serial.print(F("CH"));
267  Serial.print(user_command);
268  Serial.print(F(": "));
269  Serial.print(adc_voltage, 4);
270  Serial.print(F("V\n"));
271  }
272  }
273 }

275 //! Read channels in differential mode
276 //! @return 0 when m is entered into menu, 1 if timeout for EOC
277 uint8_t menu_2_read_differential()
278 {
279  int8_t y; // Offset into differential channel array to select polarity
280  uint8_t adc_command; // The LTC24XX command byte
281  int16_t user_command; // The user input command
282  int32_t adc_code=1; // The LTC24XX code
283  float adc_voltage; // The LTC24XX voltage
284 
285  while (1)
286  {
287  // Display differential menu
288  Serial.print(F("\n*************************\n\n"));
289  Serial.print(F("0-0P-1N 8-1P-0N\n"));
290  Serial.print(F("1-2P-3N 9-3P-2N\n"));
291  Serial.print(F("2-4P-5N 10-5P-4N\n"));
292  Serial.print(F("3-6P-7N 11-7P-6N\n"));
293  Serial.print(F("4-8P-9N 12-9P-8N\n"));
294  Serial.print(F("5-10P-11N 13-11P-10N\n"));
295  Serial.print(F("6-12P_13N 14-13P-12N\n"));
296  Serial.print(F("7-14P-15N 15-15P-14N\n"));
297  Serial.print(F("16-ALL Even_P-Odd_N\n"));
298  Serial.print(F("17-ALL Odd_P-Even_N\n"));
299  Serial.print(F("m-Main Menu\n"));
300  Serial.print(F("\nEnter a Command: "));
301  user_command = read_int();
302  if (user_command == 'm')
303  return(0);
304  Serial.println(user_command);
305  Serial.println();
306 
307  if ((user_command == 16) || (user_command == 17))
308  {
309  if (user_command == 16)
310  {
311  Serial.print(F("ALL Even_P-Odd_N\n")); // Cycles through options 0-7
312  y = 0;
313  adc_command = BUILD_COMMAND_DIFF[y];
314  }
315  if (user_command == 17)
316  {
317  Serial.print(F("ALL Odd_P-Even_N\n")); // Cycles through options 8-15
318  y = 8;
319  adc_command = BUILD_COMMAND_DIFF[y];
320  }
321  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
322  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
323  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
324  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
325  for (int8_t x = 0; x <= 7; x++) // Read all channels. All even channels are positive and odd channels are negative
326  {
327  adc_command = BUILD_COMMAND_DIFF[((x+1) % 8) + y];
328  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
329  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code);
330  adc_voltage = LTC24XX_diff_code_to_voltage(adc_code, reference_voltage);
331  Serial.println();
332  Serial.print(F("**** "));
333  print_user_command(x + y);
334  Serial.print(F(": "));
335  Serial.print(adc_voltage, 4);
336  Serial.print(F("V\n"));
337  }
338  }
339  else // Read selected channels
340  {
341  // Reads and displays a selected channel
342  adc_command = BUILD_COMMAND_DIFF[user_command];
343  Serial.print(F("ADC Command: 0b"));
344  Serial.println(adc_command, BIN);
345  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
346  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
347  while (LTC24XX_EOC_timeout(LTC24XX_CS, MISO_TIMEOUT));
348  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code);
349  Serial.print(F("Received Code: 0x"));
350  Serial.println(adc_code, HEX);
351  adc_voltage = LTC24XX_diff_code_to_voltage(adc_code, reference_voltage);
352  Serial.println();
353  Serial.print(F("**** "));
354  Serial.print(F("CH"));
355  print_user_command(user_command);
356  Serial.print(F(": "));
357  Serial.print(adc_voltage, 4);
358  Serial.print(F("V"));
359  Serial.println();
360  }
361  }
362 }
