128 //! Repeats Linduino loops
129 void loop()
130 {
131  int16_t user_command;
132  static int16_t selected_dac =0; // The selected DAC to be updated (0=A, 1=B ... 5=All). Initialized to "A".
133  // The main control loop
134  if (demo_board_connected) // Do nothing if the demo board is not connected
135  {
136  if (Serial.available()) // Check for user input
137  {
138  user_command = read_int(); // Read the user command
139  Serial.println(user_command);
140  Serial.flush();
141  switch (user_command)
142  {
143  case 1:
144  menu_1_select_dac(&selected_dac);
145  break;
146  case 2:
147  menu_2_write_to_input_register(selected_dac);
148  break;
149  case 3:
150  menu_3_write_and_update_dac(selected_dac);
151  break;
152  case 4:
153  menu_4_update_power_up_dac(selected_dac);
154  break;
155  case 5:
156  menu_5_power_down_dac(selected_dac);
157  break;
158  }
159  Serial.println("\n*****************************************************************");
160  print_prompt(selected_dac);
161  }
162  }
163 }

184 //! Write data to input register, but do not update DAC output
185 void menu_2_write_to_input_register(int16_t selected_dac) //!< what DAC to operate on
186 {
187  uint16_t dac_code;
188 
189  if (prompt_voltage_or_code() == PROMPT_VOLTAGE)
190  dac_code = get_voltage(LTC2600_lsb, LTC2600_offset);
191  else
192  dac_code = get_code();
193 
194  LTC2600_write(LTC2600_CS, LTC2600_CMD_WRITE, address_map[selected_dac], dac_code << shift_count);
195 }

243 uint16_t get_voltage(float LTC2600_lsb, //!< the voltage LSB of the DAC
