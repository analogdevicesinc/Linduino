93 // calculate current time and convert to millivolts for display in voltmeter
94 void loop()
95 {
96  quikeval_I2C_connect(); //! Connects I2C port to the QuikEval connector
97 
98  uint32_t elapsedTime = millis();
99  Serial.print("elapsedTime: ");
100  Serial.println(elapsedTime);
101 
102  // convert milliseconds to minutes
103  uint32_t elapsedMinutes = elapsedTime / (1000L*60L);
104  // current TOTAL minutes (minutes in the hour plus 60 per hour)
105  uint32_t minutes = startMinutes + elapsedMinutes;
106  // get the current hour from the minutes, make it go from 1-12 (not 0-11)
107  uint32_t hours = 1 + (((minutes / 60) - 1) % 12);
108  // now minutes is just minutes in the current hour
109  minutes %= 60;
110 
111  Serial.print("minutes: ");
112  Serial.println(minutes);
113  Serial.print("hours: ");
114  Serial.println(hours);
115 
116  // make the hours in the 1000's and 100's column, minutes in 10's and 1's on a millivolt display
117  uint32_t mV = hours * 100L + minutes;
118 
119  Serial.print("Outputting ");
120  Serial.print(mV, 4);
121  Serial.println(" mV on DAC 0");
122 
123  //! Write a code to the LTC2607 internal register and update the output voltage. (convert millivolts to volts)
124  LTC2607_write(LTC2607_I2C_GLOBAL_ADDRESS, LTC2607_WRITE_UPDATE_COMMAND, address_map[0],
125  LTC2607_voltage_to_code((float)mV/1000.0F, MY_LSB, MY_OFF));
126 }
