336  * Read the efficiency of a polyphase rail
337  *
338  * polling: poll if true
339  * return: the efficiency
340  */
341 float LT_PMBusRail::readEfficiency(bool polling)
342 {
343  float val1, val2 = 0.0;
344  float efficiency = 0.0;
345 
346 // Serial.print("eff "); Serial.println(getCapabilities(), HEX);
347  if (hasCapability(HAS_POUT | HAS_PIN))
348  {
349 // Serial.println("1");
350  val1 = readPin(polling);
351  val2 += readPout(polling);
352  efficiency = 100.0 * val2/val1;
353  }
354  else if (hasCapability(HAS_POUT | HAS_IIN | HAS_VIN))
355  {
356 // Serial.println("2");
357  val1 = readVin(polling);
358  val1 *= readIin(polling);
359  val2 = readPout(polling);
360  efficiency = 100.0 * val2/val1;
361  }
362  else if (hasCapability(HAS_VOUT | HAS_IOUT | HAS_PIN))
363  {
364 // Serial.println("3");
365  val1 = readPin(polling);
366  val2 = readVout(polling);
367  val2 *= readIout(polling);
368  efficiency = 100.0 * val2/val1;
369  }
370  else if (hasCapability(HAS_VOUT | HAS_IOUT | HAS_VIN | HAS_IIN))
371  {
372 // Serial.println("4");
373  val1 = readVin(polling);
374  val1 *= readIin(polling);
375  val2 = readVout(polling);
376  val2 *= readIout(polling);
377  efficiency = 100.0 * val2/val1;
378  }
379  return efficiency;
380 }

443 float LT_PMBusRail::readTransient(bool polling)
444 {
445  float max = 0.0;
446  float vout;
447  float vout_uv;
448  uint8_t vout_response;
449  uint8_t status;
450  tRailDef **rail = railDef_;
451  float v;
452  bool is_controller;
453  uint16_t pads;
454  PsmDeviceType t;
455 
456  while (*rail != NULL)
457  {
458  is_controller = isController();
459  max = 0.0;
460  for (int j = 0; j < (*rail)->noOfPages; j++)
461  {
462  // Skip LTC3882 Slave phases
463  t = pmbus_->deviceType((*rail)->address);
464  if (is_controller && (t == LTC3882 || t == LTC3882_1))
465  {
466 // Serial.println(j, DEC);
467  pads = pmbus_->smbus()->readWord((*rail)->address, MFR_PADS);
468 // Serial.println(pads, HEX);
469  if (pads & (1 << (14 + j)))
470  continue;
471  }
472 
473  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
474  if (polling) pmbus_->waitForNotBusy((*rail)->address);
475  pmbus_->setPage((*rail)->address, (*rail)->pages[j]);
476  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
477  if (polling) pmbus_->waitForNotBusy((*rail)->address);
478  vout = pmbus_->getVout((*rail)->address, polling);
479  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
480  if (polling) pmbus_->waitForNotBusy((*rail)->address);
481  vout_uv = pmbus_->getVoutUv((*rail)->address, polling);
482  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
483  if (polling) pmbus_->waitForNotBusy((*rail)->address);
484  vout_response = pmbus_->smbus()->readByte((*rail)->address, VOUT_UV_FAULT_RESPONSE);
485  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
486  if (polling) pmbus_->waitForNotBusy((*rail)->address);
487  pmbus_->smbus()->writeByte((*rail)->address, VOUT_UV_FAULT_RESPONSE, 0);
488  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
489  if (polling) pmbus_->waitForNotBusy((*rail)->address);
490 
491  status = pmbus_->readVoutStatusByte((*rail)->address);
492  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
493  if (polling) pmbus_->waitForNotBusy((*rail)->address);
494  if (is_controller)
495  pmbus_->smbus()->writeByte((*rail)->address, STATUS_VOUT, status | (1 << 4));
496  else
497  pmbus_->clearFaults((*rail)->address);
498  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
499  if (polling) pmbus_->waitForNotBusy((*rail)->address);
500 
501 // Serial.print("N "); Serial.println(vout, DEC);
502 
503  // Generating a fault, even if ignored, can make things busy, so poll.
504  for (v = 0.95 * vout; v < 1.05 * vout; v = v + 0.001)
505  {
506  pmbus_->setVoutUvFaultLimit((*rail)->address, v);
507  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
508  if (polling) pmbus_->waitForNotBusy((*rail)->address);
509  status = pmbus_->readVoutStatusByte((*rail)->address);
510  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
511  if (polling) pmbus_->waitForNotBusy((*rail)->address);
512  if (status & (1 << 4))
513  {
514 // Serial.print("V "); Serial.print(v,DEC); Serial.print(" S "); Serial.println(status, HEX);
515  break;
516  }
517  delay(50);
518  }
519 
520  pmbus_->setVoutUvFaultLimit((*rail)->address, vout_uv);
521  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
522  if (polling) pmbus_->waitForNotBusy((*rail)->address);
523  pmbus_->smbus()->writeByte((*rail)->address, VOUT_UV_FAULT_RESPONSE, vout_response);
524  if (polling) pmbus_->smbus()->waitForAck((*rail)->address, 0x00);
525  if (polling) pmbus_->waitForNotBusy((*rail)->address);
526 
527  if (is_controller)
528  pmbus_->smbus()->writeByte((*rail)->address, STATUS_VOUT, status | (1 << 4));
529  else
530  pmbus_->clearFaults((*rail)->address);
531 
532 // Serial.println(vout, DEC);
533 // Serial.println(v, DEC);
534 // Serial.println();
535 
536  max = max(max, vout-v);
537 // Serial.print("transient "); Serial.println(vout-v, DEC);
538 // Serial.print("uv "); Serial.println(vout_uv, DEC);
539 // Serial.print("resp "); Serial.println(vout_response, HEX);
540 
541  }
542  rail++;
543  }
544 
545  return max;
546 }
