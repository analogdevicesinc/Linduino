4 @verbatim
5 
6 
7 tinyDC590 is used to give basic DC590B compatibility to e.g. DC2574A_KIT sketch so
8 it can also be operated together with the LTC2947 GUI
9 
10 DC590B compatibility:
11 tinyDC590 provides a subset of the DC590B commands.
12 This way the e.g. the LTC2947 GUI is also able to communicate with the DC2574A_KIT sketch which enables the
13 user to use the GUI to change LTC2947 configuration e.g. to set alarm thresholds etc.
14 The main limitations of this compatibility mode are:
15 - EEPROM read from the DC2334A demo board is faked (it will always report the right EEPROM string)
16 - SPI operation is not supported, due to incompatibility with the optional LCD Keypad Shield
17 
18 Command Function Notes
19 Q Receive Byte * I2C only, send ACK
20 R Receive Byte * I2C only, sends NACK
21 Sxx Send byte (2 hex chars) * SPI and I2C. Returns ‘N’ if slave NACKs.
22 s Send start condition I2C only
23 p Send stop condition I2C only
24 Z Send line feed to host (0x0A)
25 I Read demo board info.
26 i Read contoller ID and firmware rev
27 O Turn isolated power on
28 o Turn isolated power off
29 MI Switch to isolated I2C mode
30 Kxy Bang pin, x argument is ‘0’ or ‘1’ for low or high, y is pin address, see below.
31 jxy Set pin mode, x argument is ‘0’ or ‘1’ for INPUT or OUTPUT, y is pin address, see below.
32 ky Read pin, answer is ‘0’ or ‘1’ for low or high, y is pin address, see below.
33 Uxxy Write Port D bus, xx is ASCII Hex data, y is strobe pin address, strobed low.
34 D Delay milliseconds
35 P PingPause (echo P, wait 5 ms)
36 g IO pin low
37 G IO pin high
38 H Wait for SDO=1 (SPI only) ‘T’ on timeout
39 L Wait for SDO=0 (SPI only) ‘T’ on timeout
40 0x80 DC590B pseudo Reset
41 
42 
43 the following DC590B commands are NOT implemented:
44 r Receive 7 bits In SPI mode only.
45 t Begin recording loop
46 u Stop recording loop
47 v Echo loop
48 Txx Tranceive (send and receive) * (SPI only)
49 w Loop until another command rec’d
50 X CS=1
51 x CS=0
52 MS Switch to isolated SPI mode
53 MX Switch to auxillary I2C mode

154 /** \fn void EnqueueSerialData()
155 \brief Enqueue serial data. */
156 void EnqueueSerialData()
157 {
158  uint8_t help, help2;
159  do
160  {
161  help = Serial.available(); // max 64 bytes!
162  }
163  while ((help == 0) && (QueueIn == QueueOut)); // no new data and queue is empty
164 
165  // helper is needed to check if queue is full
166  help2 = ((QueueOut + QUEUE_ELEMENTS) % QUEUE_SIZE);
167 
168  while (
169  (help != 0) && // data available
170  (QueueIn != help2) // queue not full
171  )
172  {
173  // store byte to queue
174  Queue[QueueIn] = Serial.read();
175  QueueIn = (QueueIn + 1) % QUEUE_SIZE;
176  help = Serial.available(); // update number of remaining bytes
177  }
178 }

180 /** \fn char DequeueChar()
181 \brief Dequeue character.
182 \return A char. */
183 char DequeueChar()
184 {
185  // dequeue oldest byte from queue
186  uint8_t help = Queue[QueueOut];
187  QueueOut = (QueueOut + 1) % QUEUE_SIZE;
188  return help;
189 }

252 /** \fn boolean tinyDC590B(char command)
253 \brief process commands received via serial interface. Those are the m,l commands and the basic DC590B commands
254 used for DC590B compatibility mode that allows the GUI to communicate with this sketch.
255 (basic set of DC590 commands to allow I2C communication to the LTC2947)
256 \param command the command to be decoded
257 \return true if command was recognized and decoded, false for unknown command */
258 boolean tinyDC590B(char command)
259 {
260  dc590Busy = true;// DC590 is busy
261  switch (command)
262  {
263  case 't':
264  case 'v':
265  case 'w':
266  // non supported DC590B commands
267  //'u': // this is a sub-command of 't', so no need to define it here!!!
268  break;
269  case 'x':
270  output_low(QUIKEVAL_CS);
271  break;
272  case 'X':
273  output_high(QUIKEVAL_CS);
274  break;
275  case 'D': // delay milliseconds
276  delay(read_hex() << 8 | read_hex());
277  break;
278  case 'g': // IO pin low
279  output_low(QUIKEVAL_GPIO);
280  break;
281  case 'G': // IO pin high
282  output_high(QUIKEVAL_GPIO);
283  break;
284  case 'H': // wait for MISO to go high with a timeout
285  {
286  long delay_count = 0;
287  while (1)
288  {
289  if (input(MISO) == 1) break; // MISO is high so quit
290  if (delay_count++ > MISO_TIMEOUT)
291  {
292  //Serial.print('T'); // timeout occurred. Print 'T'
293  break;
294  }
295  else delay(1);
296  }
297  }
298  break;
299  case 'L': // wait for MISO to go low with a timeout
300  {
301  long delay_count = 0;
302  while (1)
303  {
304  if (input(MISO) == 0) break; // MISO is low so quit
305  if (delay_count++ > MISO_TIMEOUT)
306  {
307  //Serial.print('T'); // timeout occurred. Print 'T'
308  break;
309  }
310  else delay(1);
311  }
312  }
313  break;
314  case 'P': // ping
315  Serial.print('P');
316  delay(5);
317  break;
318  case 'i': // send controller id string
319  pseudo_reset = 0;
320  ////////////////////////////////////////////////////
321  // CHANGED MAJOR VERSION to 2 FOR ENHANCED VERSION//
322  // CHANGED MINOR VERSION to 2 FOR TINY VERSION //
323  // added suffix to identify tiny version //
324  ////////////////////////////////////////////////////
325  //Serial.print(F("USBSPI,PIC,01,01,DC,DC590,----------------------\n"));
326  Serial.print(F("USBSPI,PIC,02,02,DC,DC590,tinyDC590-DC2574A_KIT-\n"));
327  Serial.print('\0'); // quikeval wants to read the null termination character!
328  Serial.flush();
329  dc590Busy = false;
330  break;
331  case 'I': // get controller id string
332  pseudo_reset = 0;
333  // this is DC2334 only, so we will just send its string (no read from demoboard's EEPROM here!)
334  Serial.print(F("LTC2947,Cls,D2947,01,01,DC,DC2334A,-------------\n"));
335  Serial.print('\0'); // quikeval wants to read the null termination character!
336  Serial.flush();
337  dc590Busy = false;
338  break;
339  case 'K': // Set pin state. The pin assignments are :
340  {
341  // 0: PIND2, Arduino 2
342  // 1: PIND3, Arduino 3
343  // 2: PIND4, Arduino 4
344  // 3: PIND5, Arduino 5
345  // 4: PIND6, Arduino 6
346  // 5: PIND6, Arduino 7
347  char pin_value = get_char(); // read the value
348  char pin = get_char(); // read the pin
349  digitalWrite(pin - 0x30 + 2, pin_value == '0' ? LOW : HIGH);
350  }
351  break;
352  case 'k': // Get pin state. The pin assignments are the same as for 'K':
353  {
354  char pin = get_char(); // read the pin
355  Serial.print(digitalRead(pin - 0x30 + 2));
356  }
357  break;
358  case 'j': // Set pin mode. The pin assignments are the same as for 'K':
359  {
360  char pin_value = get_char(); // read the value
361  char pin = get_char(); // read the pin
362  pinMode(pin - 0x30 + 2, pin_value == '0' ? INPUT : OUTPUT);
363  }
364  break;
365  case 'M': // change the serial mode (I2C only!)
366  switch (get_char())
367  {
368  case 'I': // I2C mode
369  serial_mode = i2c_mode;
370  quikeval_I2C_connect();
371  break;
372  case 'S':// spi mode
373  serial_mode = spi_mode;
374  quikeval_SPI_connect();
375  break;
376  case 'X':// axillary I2C mode - no hardware action necessary, always available.
377  serial_mode = i2c_auxiliary_mode;
378  quikeval_SPI_connect();
379  break;
380  }
381  break;
382  case 'p': // I2C stop //i2c_stop();
383  if (serial_mode != spi_mode) TWCR = (1 << TWINT) | (1 << TWEN) | (1 << TWSTO);
384  break;
385  case 'Q': // Read byte in I2C mode only. Add ACK
386  if (serial_mode != spi_mode)
387  {
388  byte_to_hex(i2c_read(WITH_ACK));
389  Serial.print(byte_to_hex_buffer);
390  }
391  break;
392  case 'r':
393  byte_to_hex(spi_read(0));
394  Serial.print(byte_to_hex_buffer);
395  break;
396  case 'R': // Read byte, add NACK in I2C mode
397  if (serial_mode != spi_mode)
398  {
399  byte_to_hex(i2c_read(WITH_NACK));
400  Serial.print(byte_to_hex_buffer);
401  }
402  else
403  {
404  byte_to_hex(spi_read(0));
405  Serial.print(byte_to_hex_buffer);
406  }
407  break;
408  case 's': // I2C start
409  if (serial_mode != spi_mode)
410  i2c_start();
411  break;
412  case 'S': // send byte
413  if (serial_mode != spi_mode)
414  {
415  if (i2c_write(read_hex()) == 1) Serial.print('N');
416  }
417  else
418  {
419  spi_write(read_hex());
420  }
421  break;
422  case 'T': // transceive byte
423  if (serial_mode == spi_mode)
424  {
425  byte_to_hex(spi_read(read_hex()));
426  Serial.print(byte_to_hex_buffer);
427  }
428  break;
429  case 'Z': // line feed
430  Serial.print('\n');
431  Serial.flush();
432  // The last character the GUI sends for every transaction is always Z
433  // this means if we reached this state the GUI finished its communication
434  // with DC590 and we are not busy anymore
435  dc590Busy = false;
436  break;
437  case '\x80': // 0x80:Reset
438  if (pseudo_reset == 0)
439  {
440  delay(500); // The delay is needed for older GUI's
441  Serial.print("hello\n");
442  pseudo_reset = 1;
443  }
444  break;
445  default:
446  dc590Busy = false; // if we don't set this to false here, we have to deassert it within processCommand
447  return false; // non-DC590B command
448  }
449  return true;// processed DC590B command
450 }
