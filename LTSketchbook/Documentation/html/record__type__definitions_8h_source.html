521 /********************************************************************
522  * Definitions: NAME (VALUE) NOTE
523  * BEFORE_BEGIN (0) This event is fired before any
524  * commands are issued to communicate
525  * with the system
526  * SYSTEM_BEFORE_PROGRAM (1) This event is fired before any
527  * commands are issued to program
528  * the NVM
529  * SYSTEM_BEFORE_VERIFY (2) This event is fired after programming
530  * the system before any commands are
531  * issued to verify the NVM
532  * AFTER_DONE (3) This event is fired after the end
533  * of the entire programming and
534  * verification sequence is complete
535  *******************************************************************/
536 #define BEFORE_BEGIN (0x00)
537 #define BEFORE_INSYSTEM_PROGRAMMING_BEGIN (0x10)
538 #define SYSTEM_BEFORE_PROGRAM (0x01)
539 #define INSYSTEM_CHIP_BEFORE_PROGRAM (0x11)
540 #define SYSTEM_BEFORE_VERIFY (0x02)
541 #define INSYSTEM_CHIP_BEFORE_VERIFY (0x12)
542 #define INSYSTEM_CHIP_AFTER_VERIFY (0x13)
543 #define SYSTEM_AFTER_VERIFY (0x04)
544 #define AFTER_DONE (0x03)
