1 /*!
2 Linear Technology DC1096B Demonstration Board.
3 LTC2642: 16-/14-/12-Bit VOUT DAC with SPI Interface
4 
5 LTC2641: 16-/14-/12-Bit VOUT DAC with SPI Interface
6 @verbatim
7 
8 NOTES
9 
10  Set the terminal baud rate to 115200 and select the newline terminator.
11  No external power supply is required. Two on-board reference voltages of
12  2.5V and 5V are available. The board features LTC2642A with direct and
13  buffered output.
14 
15  The program displays calculated voltages which are based on the reference
16  voltage used. A precision voltmeter is needed to verify the actual
17  measured voltages against the calculated voltage displayed.
18 
19  There is also an option of generating a square wave of required voltage
20  levels and frequency. This proves the fast settling time of the corresponding
21  parts. Setting the software to unipolar or bipolar will not set the hardware
22  to bipolar mode. The jumpers need to be placed in their proper positions in
23  order to accomplish this.
24 
25  Jumper settings:
26  JP3: Sets DAC2 to either unipolar or bipolar mode. In bipolar mode, set JP6
27  and JP7 to EXT and apply an appropriate supply to AMP V+, AMP V–, and GND
28  turret posts.
29 
30  JP4: Sets DAC3 to either unipolar or bipolar mode. In bipolar mode, set JP6
31  and JP7 to EXT and apply an appropriate supply to AMP V+, AMP V–, and GND
32  turret posts.
33 
34  Explanation of Commands:
35  Option 1: Enter a digital value or voltage to obtain analog voltage output.
36  Option 2: Enter the voltage levels to obtain a square wave.
37  Option 3: Vary the reference voltage used.
38  Option 4: Select Bipolar/Unipolar.
39 
40 USER INPUT DATA FORMAT:
41  decimal : 1024
42  hex : 0x400
43  octal : 02000 (leading 0 "zero")
44  binary : B10000000000
45  float : 1024.0

154 //! Repeats Linduino loop
155 void loop()
156 {
157  int8_t user_command; // The user input command
158  uint8_t acknowledge = 0;
159  if (Serial.available()) // Check for user input
160  {
161  user_command = read_int(); // Read the user command
162  if (user_command != 'm')
163  Serial.println(user_command); // Prints the user command to com port
164  Serial.flush();
165  switch (user_command)
166  {
167  case 1:
168  acknowledge |= menu1_voltage_output();
169  break;
170  case 2:
171  acknowledge |= menu2_square_wave_output();
172  break;
173  case 3:
174  menu3_change_reference_voltage();
175  break;
176  case 4:
177  menu4_select_range();
178  break;
179  default:
180  Serial.println(F("Incorrect Option"));
181  }
182  if (acknowledge)
183  Serial.println(F("***** I2C ERROR *****"));
184  print_prompt();
185  }
186 }

188 //! Function to enter a digital value and get the analog output
189 uint8_t menu1_voltage_output()

200  Serial.print("\nDac Code = 0x");

202  Serial.print("\nOutput voltage = ");
203  Serial.print(voltage, 3);
204  Serial.print(" V");
205  return 0;
206 }

208 //! Function to generate a square wave of desired frequency and voltage ranges
209 uint8_t menu2_square_wave_output()
210 {

214

217  voltage_high = read_float();
218  Serial.print(voltage_high);
219  Serial.println(" V");
220 
221  Serial.print("Enter voltage_low: ");
222  while (!Serial.available());
223  voltage_low = read_float();
224  Serial.print(voltage_low);
225  Serial.println(" V");
226 
227  //! Converting data into voltage
228  code_high = LTC2642_voltage_to_code(voltage_high, reference_voltage, range);
229  code_low = LTC2642_voltage_to_code(voltage_low, reference_voltage, range);
230 
231  Serial.print(F("Enter any character to stop. "));
232  while (!Serial.available()) //! Generate square wave until a key is pressed
233  {
234  LTC2642_write(LTC2642_CS, code_high);
235  delay(30); // 30 ms delay
236  LTC2642_write(LTC2642_CS, code_low);
237  delay(30); // 30 ms delay
238  }
239  read_int();
240  return 0;
241 }
