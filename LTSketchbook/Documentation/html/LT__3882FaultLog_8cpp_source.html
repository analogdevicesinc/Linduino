73  * Read LTC3882 fault log

77 void
78 LT_3882FaultLog::read(uint8_t address)
79 {
80 #ifdef RAW_EEPROM
81  uint8_t *data = (uint8_t *) malloc(sizeof(uint8_t) * 80 * 2); // Becuse CRC is stripped, the acutal data size is smaller
82 
83  getNvmBlock(address, 176, 80, 0x00, data);
84 #else
85  uint8_t *data = (uint8_t *) malloc(147);
86  data[0] = 0x00;
87 
88  pmbus_->smbus()->readBlock(address, MFR_FAULT_LOG, data, 147);
89 #endif
90 
91  faultLog3882 = (struct LT_3882FaultLog::FaultLogLtc3882 *) (data);
92 }
