64 void rotate(uint8_t *buf, int start, int len)
65 {
66 
67  int count = 0;
68  int offset = 0;
69 
70  while (count < len)
71  {
72 
73  int index = offset;
74  char tmp = buf[index];
75  int index2 = (start + index) % len;
76 
77  while (index2 != offset)
78  {
79 
80  buf[index] = buf[index2];
81  count++;
82 
83  index = index2;
84  index2 = (start + index) % len;
85  }
86 
87  buf[index] = tmp;
88  count++;
89 
90  offset++;
91  }
92 }

116  * Read LTC2974 fault log

120 void
121 LT_2974FaultLog::read(uint8_t address)
122 {
123  // Copy to RAM
124  pmbus_->smbus()->sendByte(address, MFR_FAULT_LOG_RESTORE);
125  // Monitor BUSY bit
126  while ((pmbus_->smbus()->readByte(address, MFR_COMMON) & (1 << 6)) == 0);
127 
128  uint16_t size = sizeof(struct LT_2974FaultLog::FaultLogLtc2974);
129  uint8_t *data = (uint8_t *) malloc(size);
130  if (data == 0)
131  Serial.print(F("bad malloc."));
132 #ifdef RAW_EEPROM
133  // For MFR_EE_DATA, but would require reversing cyclic data
134  getNvmBlock(address, 384, 128, 0x00, data);
135  // Points to the beginning of the cyclic data
136  uint8_t block_address = data[0];
137  // Shift the last data to the top
138  rotate(&data[72], block_address - 71, 167);
139  // Drop the block address
140  memmove(data, data+1, size-1);
141  // Reverse the order of cyclic data
142  for (int i = 71; i < 71 + 83; i++)
143  {
144  uint8_t d = data[237-(i-71)];
145  data[237-(i-71)] = data[i];
146  data[i] = d;
147  }
148 
149 #else
150  // Read block data with log
151  pmbus_->smbus()->readBlock(address, MFR_FAULT_LOG, data, 255);
152 #endif
153 
154  struct LT_2974FaultLog::FaultLogLtc2974 *log = (struct LT_2974FaultLog::FaultLogLtc2974 *)data;
155 
156  log->firstValidByte = 71;
157 
158  log->lastValidByte = 237;
159 
160  uint8_t cycle_start = log->preamble.position_last;
161 
162  log->loops = (LT_2974FaultLog::FaultLogReadLoopLtc2974 *) (log->telemetryData - 53 + cycle_start);
163 
164  faultLog2974 = log;
165 }
