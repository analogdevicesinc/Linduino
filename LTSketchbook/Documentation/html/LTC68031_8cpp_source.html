92 //Function that writes configuration of LTC6803-1/-3
93 void LTC6803_wrcfg(uint8_t total_ic,uint8_t config[][6])
94 {
95  uint8_t BYTES_IN_REG = 6;
96  uint8_t CMD_LEN = 2+(7*total_ic);
97  uint8_t *cmd;
98  uint16_t cfg_pec;
99  uint8_t cmd_index; //command counter
100 
101  cmd = (uint8_t *)malloc(CMD_LEN*sizeof(uint8_t));
102 
103  cmd[0] = 0x01;
104  cmd[1] = 0xc7;
105 
106  cmd_index = 2;
107  for (uint8_t current_ic = total_ic; current_ic > 0; current_ic--)
108  {
109 
110  for (uint8_t current_byte = 0; current_byte < BYTES_IN_REG; current_byte++)
111  {
112  cmd[cmd_index] = config[current_ic-1][current_byte];
113  cmd_index = cmd_index + 1;
114  }
115 
116  cfg_pec = pec8_calc(BYTES_IN_REG, &config[current_ic-1][0]); // calculating the PEC for each ICs configuration register data
117  cmd[cmd_index ] = (uint8_t)cfg_pec;
118  cmd_index = cmd_index + 1;
119  }
120 
121  output_low(LTC6803_CS);
122  spi_write_array(CMD_LEN, cmd);
123  output_high(LTC6803_CS);
124  free(cmd);
125 }

307 //Function that calculates PEC byte
308 uint8_t pec8_calc(uint8_t len, uint8_t *data)
309 {
310 
311  uint8_t remainder = 0x41;//PEC_SEED;
312 
313 
314  /*
315  * Perform modulo-2 division, a byte at a time.
316  */
317  for (int byte = 0; byte < len; ++byte)
318  {
319  /*
320  * Bring the next byte into the remainder.
321  */
322  remainder ^= data[byte];
323 
324  /*
325  * Perform modulo-2 division, a bit at a time.
326  */
327  for (uint8_t bit = 8; bit > 0; --bit)
328  {
329  /*
330  * Try to divide the current data bit.
331  */
332  if (remainder & 128)
333  {
334  remainder = (remainder << 1) ^ PEC_POLY;
335  }
336  else
337  {
338  remainder = (remainder << 1);
339  }
340  }
341  }
342 
343  /*
344  * The final remainder is the CRC result.
345  */
346  return (remainder);
347 
348 }
