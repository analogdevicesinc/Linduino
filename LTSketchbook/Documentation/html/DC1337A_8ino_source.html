8 @verbatim
9 
10 NOTES
11  Setup:
12  Set the terminal baud rate to 115200 and select the newline terminator. Equipment
13  required is a voltage source (preferably low-noise) and a precision voltmeter. No
14  external power supply is required. Ensure all jumpers on the demo board are
15  installed in their default positions from the factory. Refer to Demo Manual
16  DC1337A.
17 
18  How to test Single-Ended mode:
19  Bipolar Mode:
20  The voltage source should be connected to one of the inputs CH0,...CH7. Ensure the input is
21  within its specified absolute input voltage range. (It is easiest
22  to tie the voltage source negative terminal to COM.) Ensure the voltage
23  source is set within the range of 0V to +4.096V.
24  (Swapping input voltages results in a reversed polarity reading.)
25 
26  Unipolar Mode:
27  The voltage source should be connected to one of the inputs CH0,...CH7. Ensure the input is
28  within its specified absolute input voltage range. (It is easiest
29  to tie the voltage source negative terminal to COM.) Ensure the voltage
30  source is set within the range of 0V to +4.096V.
31 
32  How to test Differential Mode:
33  Bipolar Mode:
34  The voltage source should be connected between paired channels. Ensure both
35  inputs are within their specified absolute input voltage range. (It is easiest
36  to tie the voltage source negative terminal to COM or GND.) Ensure the voltage
37  source is set within the range of 0 to +4.096V (differential voltage range).
38  (Swapping input voltages results in a reversed polarity reading.)
39 
40  Unipolar Mode:
41  The voltage source should be connected between paired channels. (It is easiest
42  to tie the voltage source negative terminal to COM or GND.) Ensure both inputs
43  are within their specified absolute input voltage range. (It is easiest to tie
44  the voltage source negative terminal to COM or GND.) Ensure the voltage source
45  is set within the range of 0 to +4.096V (differential voltage range).
46

49  hex : 0x400
50  octal : 02000 (leading 0 "zero")
51  binary : B10000000000

155 {

187  if (acknowledge != 0) Serial.println(F("Error: No Acknowledge. Check I2C Address."));
188  Serial.println(F("*************************"));
189  print_prompt();
190  }

193 // Function Definitions
194 //! Read channels
195 //! @return 1 if successful, 0 if not
196 int8_t menu_1_read_input()
197 {
198  uint8_t user_command;
199  uint16_t adc_command; // The LTC2309 command byte
200  uint16_t adc_code = 0; // The LTC2309 code
201  uint8_t x, y, startcount, endcount;
202  uint16_t display_code;
203  float adc_voltage;
204  uint8_t acknowledge = 0; // I2C acknowledge bit
205 
206  if (uni_bipolar == LTC2309_UNIPOLAR_MODE)
207  {
208  if (single_ended_differential == LTC2309_SINGLE_ENDED_MODE)
209  {
210  Serial.println(F(" Read Input in Single-Ended, Unipolar mode:"));
211  Serial.println(F(" Note that in Unipolar mode, input voltages less than zero are reported as 0.0V"));
212  user_command = print_user_command_single_ended();
213  }
214  else
215  {
216  Serial.println(F(" Read Input in Differential, Unipolar mode:"));
217  Serial.println(F(" Note that in Unipolar mode, input voltages less than zero are reported as 0.0V"));
218  user_command = print_user_command_differential();
219  }
220  }
221  else
222  {
223  if (single_ended_differential == LTC2309_SINGLE_ENDED_MODE)
224  {
225  Serial.println(F(" Read Input in Single-Ended, Bipolar mode:"));
226  user_command = print_user_command_single_ended();
227  }
228  else
229  {
230  Serial.println(F(" Read Input in Differential, Bipolar mode:"));
231  user_command = print_user_command_differential();
232  }
233  }
234 
235  if (user_command == 8) //read all channels
236  {
237  startcount = 0;
238  endcount = 7;
239  }
240  else
241  {
242  startcount = user_command;
243  endcount = user_command;
244  }
245 
246  if (single_ended_differential == LTC2309_SINGLE_ENDED_MODE) //read single-ended
247  {
248  adc_command = BUILD_COMMAND_SINGLE_ENDED[0] | uni_bipolar;
249  acknowledge |= LTC2309_read(LTC2309_I2C_ADDRESS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
250  for (x = startcount; x <= endcount; x++)
251  {
252  adc_command = BUILD_COMMAND_SINGLE_ENDED[x % 8] | uni_bipolar; // Send channel config for the NEXT conversion to take place
253  acknowledge |= LTC2309_read(LTC2309_I2C_ADDRESS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
254  acknowledge |= LTC2309_read(LTC2309_I2C_ADDRESS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
255 
256  display_code = adc_code >> (16 - LTC2309_bits);
257  display_code = display_code & 0xFFF;
258 
259  Serial.print(F(" Received Code: b"));
260  Serial.println(display_code, BIN);
261 
262  adc_voltage = LTC2309_code_to_voltage(adc_code, LTC2309_vref, uni_bipolar);
263  Serial.print(F(" Voltage read on "));
264  Serial.print(F("Ch"));
265  Serial.print(x);
266  Serial.print(F(": "));
267  Serial.print(adc_voltage, 4);
268  Serial.println(F("V"));
269  Serial.println();
270  }
271  }
272 
273  else //read differential
274  {
275  adc_command = BUILD_COMMAND_DIFF[x % 8] | uni_bipolar; // Send channel config for the NEXT conversion to take place
276  acknowledge |= LTC2309_read(LTC2309_I2C_ADDRESS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
277  acknowledge |= LTC2309_read(LTC2309_I2C_ADDRESS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
278 
279  display_code = adc_code >> (16 - LTC2309_bits);
280  display_code = display_code & 0xFFF;
281 
282  Serial.print(F(" Received Code: b"));
283  Serial.println(display_code, BIN);
284 
285  adc_voltage = LTC2309_code_to_voltage(adc_code, LTC2309_vref, uni_bipolar);
286  Serial.print(F(" Voltage read between Chs "));
287  print_channel_selection(x + y);
288  Serial.print(adc_voltage, 4);
289  Serial.println(F("V"));
290  Serial.println();
291  }
292  return(acknowledge);
293 }
