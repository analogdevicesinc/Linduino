88 //! Repeats Linduino loop
89 void loop()
90 {
91  uint8_t user_command; // The user input command
92  uint16_t output_register; // The register value read back from the LTC5566
93  uint8_t byte_1; // Dummy variable for decoding Mixer 1's register value
94  uint8_t byte_2; // Dummy variable for decoding Mixer 2's register value
95  if (Serial.available()) // Check for user input
96  {
97  user_command = read_int(); // Read the user command
98  Serial.println(user_command); // Print the command to the screen
99  switch (user_command)
100  {
101  case 1:
102  output_register = LTC5566_dupl_settings(); // Run the duplicate settings function
103  Serial.println(F("\n\n\nLTC5566 settings:\n"));
104  LTC5566_decode_output(output_register); // Decode the resulting value
105  break;
106  case 2:
107  output_register = LTC5566_diff_settings(); // Run the different settings function
108  byte_1 = output_register & 0xFF; // Pick off least significant 8 bits for decoding
109  byte_2 = output_register >> 8; // Pick off most significant 8 bits for decoding
110  Serial.println(F("\n\n\nMixer 1 settings:\n"));
111  LTC5566_decode_output(byte_1); // Decode the resulting value
112  Serial.println(F("\n\n\nMixer 2 settings:\n"));
113  LTC5566_decode_output(byte_2); // Decode the resulting value
114  break;
115  default:
116  Serial.println(F("\n\nIncorrect Option\n"));
117  }
118  Serial.println(F("\n*****************************************************************\n\n\n\n\n\n\n\n\n\n\n\n\n\n\n\n"));
119  LTC5566_print_prompt();
120  }
121 }
