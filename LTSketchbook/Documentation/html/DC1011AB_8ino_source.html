1 /*!
2 Linear Technology DC1011A-B Demonstration Board.
3 LTC2496: 16-Bit 8-/16-Channel Delta Sigma ADC with Easy Drive Input Current Cancellation
4 
5 @verbatim
6 
7 NOTES
8  Setup:
9  Set the terminal baud rate to 115200 and select the newline terminator.
10  Equipment required is a precision voltage source and a precision
11  voltmeter. No external power supply is required. Ensure all jumpers on
12  the demo board are installed in their default positions from the
13  factory.
14 
15  How to test Single-Ended mode:
16  The voltage source should be connected to the ADC such that the negative
17  lead is connected to COM. The positive lead may be connected to any
18  channel input. Ensure voltage is within analog input voltage range -2.5V to
19  +2.5V.
20 
21  How to test Differential Mode:
22  The voltage source should be connected with positive and negative leads to
23  paired channels. The voltage source negative output must also be connected to
24  the COM pin in order to provide a ground-referenced voltage. Ensure voltage is
25  within analog input voltage range -2.5V to +2.5V. Swapping input voltages
26  results in a reversed polarity reading.
27 
28 USER INPUT DATA FORMAT:
29  decimal : 1024
30  hex : 0x400
31  octal : 02000 (leading 0 "zero")
32  binary : B10000000000
33  float : 1024.0

168 //! Repeats Linduino loop
169 void loop()
170 {
171  uint8_t user_command; // The user input command
172  uint8_t ack = 0;
173  if (demo_board_connected)
174  {
175  if (Serial.available()) // Check for user input
176  {
177  user_command = read_int(); // Read the user command
178  if (user_command == 'm');
179  else
180  Serial.println(user_command);
181  delay(50); // Allow the print to finish
182  switch (user_command)
183  {
184  case 1:
185  ack |= menu_1_read_single_ended();
186  break;
187  case 2:
188  ack |= menu_2_read_differential();
189  break;
190  default:
191  Serial.println("Incorrect Option");
192  break;
193  }
194  if (ack)
195  Serial.println(F("***** SPI ERROR *****"));
196  Serial.println(F("*****************************************************************"));
197  print_prompt();
198  }
199  }
200 }

230 //! read from ADC single-ended
231 //! @return 0 if successful, 1 if failure
232 //! @return 0 when m is entered into menu, 1 if timeout for EOC
233 uint8_t menu_1_read_single_ended()
234 {
235  uint8_t adc_command; // The LTC2496 command byte
236  int16_t user_command; // The user input command
237  int32_t adc_code=0; // The LTC2496 code
238  float adc_voltage; // The LTC2496 voltage
239 
240  while (1)
241  {
242  Serial.print(F("*************************\n\n")); // Display single-ended menu
243  Serial.print(F("0-CH0 8-CH8\n"));
244  Serial.print(F("1-CH1 9-CH9\n"));
245  Serial.print(F("2-CH2 10-CH10\n"));
246  Serial.print(F("3-CH3 11-CH11\n"));
247  Serial.print(F("4-CH4 12-CH12\n"));
248  Serial.print(F("5-CH5 13-CH13\n"));
249  Serial.print(F("6-CH6 14-CH14\n"));
250  Serial.print(F("7-CH7 15-CH15\n"));
251  Serial.print(F("16-ALL\n"));
252  Serial.print(F("m-Main Menu\n"));
253  Serial.print(F("\nEnter a Command: "));
254  user_command = read_int(); // Read the single command
255  if (user_command == 'm')
256  return(0);
257  else
258  Serial.println(user_command);
259  Serial.println();
260 
261  if (user_command == 16)
262  {
263  Serial.print(F("ALL\n"));
264  adc_command = LTC2496_ENABLE | BUILD_COMMAND_SINGLE_ENDED[0]; // Build ADC command for channel 0
265 
266  if (LTC2496_EOC_timeout(LTC2496_CS, MISO_TIMEOUT)) // Check for EOC
267  return(1);
268  LTC2496_read(LTC2496_CS, adc_command, &adc_code); // Throws out last reading
269 
270  for (int8_t x = 0; x < 15; x++)
271  {
272  adc_command = LTC2496_ENABLE | BUILD_COMMAND_SINGLE_ENDED[(x + 1) % 16]; // Read all channels in single-ended mode
273  if (LTC2496_EOC_timeout(LTC2496_CS, MISO_TIMEOUT)) // Check for EOC
274  return(1);
275  LTC2496_read(LTC2496_CS, adc_command, &adc_code);
276  adc_voltage = LTC2496_code_to_voltage(adc_code, reference_voltage);
277  Serial.print(F(" ****"));
278  Serial.print(F("CH"));
279  Serial.print(x);
280  Serial.print(F(": "));
281  Serial.print(adc_voltage, 4);
282  Serial.print(F("V\n"));
283  }
284  }
285  else if (user_command<16 && user_command >=0) // Read selected channel
286  {
287  adc_command = LTC2496_ENABLE | BUILD_COMMAND_SINGLE_ENDED[user_command];
288  Serial.print(F("\nADC Command: B"));
289  Serial.println(adc_command, BIN);
290  if (LTC2496_EOC_timeout(LTC2496_CS, MISO_TIMEOUT)) // Check for EOC
291  return(1);
292  LTC2496_read(LTC2496_CS, adc_command, &adc_code); // Throws out last reading
293  if (LTC2496_EOC_timeout(LTC2496_CS, MISO_TIMEOUT)) // Check for EOC
294  return(1);
295  LTC2496_read(LTC2496_CS, adc_command, &adc_code);
296  Serial.print(F("Received Code: 0x"));
297  Serial.println(adc_code, HEX);
298  adc_voltage = LTC2496_code_to_voltage(adc_code, reference_voltage);
299  Serial.print(F(" ****"));
300  Serial.print(F("CH"));
301  Serial.print(user_command);
302  Serial.print(F(": "));
303  Serial.print(adc_voltage, 4);
304  Serial.print(F("V\n"));
305  }
306  }
307 }

309 //! Read channels in differential mode
310 //! @return 0 when m is entered into menu, 1 if timeout for EOC
311 uint8_t menu_2_read_differential()
312 {
313  int8_t y; // Offset into differential channel array to select polarity
314  uint8_t adc_command; // The LTC2496 command byte
315  int16_t user_command; // The user input command
316  int32_t adc_code=1; // The LTC2496 code
317  float adc_voltage; // The LTC2496 voltage
318 
319  while (1)
320  {
321  // Display differential menu
322  Serial.print(F("\n*************************\n\n"));
323  Serial.print(F("0-0P-1N 8-1P-0N\n"));
324  Serial.print(F("1-2P-3N 9-3P-2N\n"));
325  Serial.print(F("2-4P-5N 10-5P-4N\n"));
326  Serial.print(F("3-6P-7N 11-7P-6N\n"));
327  Serial.print(F("4-8P-9N 12-9P-8N\n"));
328  Serial.print(F("5-10P-11N 13-11P-10N\n"));
329  Serial.print(F("6-12P_13N 14-13P-12N\n"));
330  Serial.print(F("7-14P-15N 15-15P-14N\n"));
331  Serial.print(F("16-ALL Even_P-Odd_N\n"));
332  Serial.print(F("17-ALL Odd_P-Even_N\n"));
333  Serial.print(F("m-Main Menu\n"));
334  Serial.print(F("\nEnter a Command: "));
335  user_command = read_int();
336  if (user_command == 'm')
337  return(0);
338  Serial.println(user_command);
339  Serial.println();
340 
341  if ((user_command == 16) || (user_command == 17))
342  {
343  if (user_command == 16)
344  {
345  Serial.print(F("ALL Even_P-Odd_N\n")); // Cycles through options 0-7
346  y = 0;
347  }
348  if (user_command == 17)
349  {
350  Serial.print(F("ALL Odd_P-Even_N\n")); // Cycles through options 8-15
351  y = 8;
352  }
353  adc_command = LTC2496_ENABLE | BUILD_COMMAND_DIFF[y]; // Set up first channel
354  if (LTC2496_EOC_timeout(LTC2496_CS, MISO_TIMEOUT)) // check for EOC
355  return(1);
356  LTC2496_read(LTC2496_CS, adc_command, &adc_code); // Throws out last reading
357  for (int8_t x = 0; x <= 7; x++) // Read all channels. All even channels are positive and odd channels are negative
358  {
359  adc_command = LTC2496_ENABLE | BUILD_COMMAND_DIFF[((x+1) % 8) + y];
360  if (LTC2496_EOC_timeout(LTC2496_CS, MISO_TIMEOUT)) // Check for EOC
361  return(1);
362  LTC2496_read(LTC2496_CS, adc_command, &adc_code);
363  adc_voltage = LTC2496_code_to_voltage(adc_code, reference_voltage);
364  Serial.println();
365  Serial.print(F("**** "));
366  print_user_command(x + y);
367  Serial.print(F(": "));
368  Serial.print(adc_voltage, 4);
369  Serial.print(F("V\n"));
370  }
371  }
372  else // Read selected channels
373  {
374  // Reads and displays a selected channel
375  adc_command = LTC2496_ENABLE | BUILD_COMMAND_DIFF[user_command];
376  Serial.print(F("ADC Command: 0b"));
377  Serial.println(adc_command, BIN);
378  if (LTC2496_EOC_timeout(LTC2496_CS, MISO_TIMEOUT)) // Check for EOC
379  return(1);
380  LTC2496_read(LTC2496_CS, adc_command, &adc_code); // Throws out last reading
381  if (LTC2496_EOC_timeout(LTC2496_CS, MISO_TIMEOUT)) // Check for EOC
382  return(1);
383  LTC2496_read(LTC2496_CS, adc_command, &adc_code);
384  Serial.print(F("Received Code: 0x"));
385  Serial.println(adc_code, HEX);
386  adc_voltage = LTC2496_code_to_voltage(adc_code, reference_voltage);
387  Serial.println();
388  Serial.print(F("**** "));
389  Serial.print(F("CH"));
390  print_user_command(user_command);
391  Serial.print(F(": "));
392  Serial.print(adc_voltage, 4);
393  Serial.print(F("V"));
394  Serial.println();
395  }
396  }
397 }
